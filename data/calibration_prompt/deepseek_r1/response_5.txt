class WakeModelOptimizer:
    """Modular wake model optimizer with enhanced analysis capabilities."""
    
    def __init__(self, model_type=1, downstream=True):
        """
        Initialize the optimizer.
        
        Parameters:
        -----------
        model_type : int
            1 for BlondelSuperGaussianDeficit2020, 2 for TurboGaussianDeficit
        downstream : bool
            True for downstream wake, False for upstream blockage
        """
        self.model_type = model_type
        self.downstream = downstream
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        self.site = Hornsrev1Site()
        
        # Load and prepare reference data
        self._load_reference_data()
        self._setup_region_of_interest()
        
        # Optimization state
        self.optimizer = None
        self.best_params = None
        self.best_rmse = None
        
    def _load_reference_data(self):
        """Load and prepare the reference dataset."""
        self.dat = xr.load_dataset('./DTU10MW.nc')
        self.dat = self.dat.assign_coords(x=self.dat.x * self.D, y=self.dat.y * self.D)
        
    def _setup_region_of_interest(self):
        """Setup region of interest based on downstream/upstream setting."""
        if self.downstream:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        self.flow_roi = self.dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
        # Store for plotting
        self.X_LB, self.X_UB = X_LB, X_UB
        
    def _prepare_operating_conditions(self):
        """Prepare operating conditions for evaluation."""
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
        full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
        
        assert full_ws.size == full_ti.size
        
        # Store for later use
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.num_conditions = full_ws.size
        
        # Extract observed deficits
        self._extract_observed_deficits()
        
    def _extract_observed_deficits(self):
        """Extract observed deficits for all operating conditions."""
        # Create a baseline simulation to get CT values for each condition
        baseline_wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        sim_res = baseline_wfm(
            [0], [0], 
            ws=self.full_ws, 
            TI=self.full_ti, 
            wd=[270] * self.num_conditions, 
            time=True
        )
        
        # Extract observed deficits for each condition
        obs_values = []
        for t in range(sim_res.time.size):
            this_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_sim.CT, 
                ti=this_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
            
        self.observed_deficits = xr.concat(obs_values, dim='time')
        
    def create_wfm(self, params):
        """
        Create wind farm model with given parameters.
        
        Parameters:
        -----------
        params : dict
            Dictionary of model parameters
            
        Returns:
        --------
        wfm : All2AllIterative
            Configured wind farm model
        """
        # Split parameters based on model type and direction
        if self.downstream:
            if self.model_type == 1:
                # BlondelSuperGaussianDeficit2020 for downstream
                wake_deficitModel = BlondelSuperGaussianDeficit2020(
                    a_s=params.get('a_s', 0.17),
                    b_s=params.get('b_s', 0.005),
                    c_s=params.get('c_s', 0.2),
                    b_f=params.get('b_f', -0.68),
                    c_f=params.get('c_f', 2.41)

                turb_args = {'c': np.array([
                    params.get('ch1', 0.73),

                ])}
                blockage_args = {}
            else:
                # TurboGaussianDeficit for downstream
                wake_deficitModel = TurboGaussianDeficit(
                    A=params.get('A', 0.04),
                    cTI=[params.get('cti1', 1.5), params.get('cti2', 0.8)],
                    ctlim=params.get('ctlim', 0.999),
                    ceps=params.get('ceps', 0.25),
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficitModel.WS_key = 'WS_jlk'
                turb_args = {'c': np.array([
                    params.get('ch1', 0.73),
                    params.get('ch2', 0.8325),
                    params.get('ch3', -0.0325),
                    params.get('ch4', -0.3)
                ])}
                blockage_args = {}
        else:
            # Upstream blockage
            wake_deficitModel = BlondelSuperGaussianDeficit2020()
            turb_args = {}
            blockage_args = {
                'ss_alpha': params.get('ss_alpha', 0.8888888888888888),
                'ss_beta': params.get('ss_beta', 1.4142135623730951),
                'r12p': np.array([params.get('rp1', -0.672), params.get('rp2', 0.4897)]),
                'ngp': np.array([
                    params.get('ng1', -1.381),
                    params.get('ng2', 2.627),
                    params.get('ng3', -1.524),
                    params.get('ng4', 1.336)
                ])
            }
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
        
        # Create wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficitModel,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def evaluate_model(self, params):
        """
        Evaluate model with given parameters.
        
        Parameters:
        -----------
        params : dict
            Dictionary of model parameters
            
        Returns:
        --------
        rmse : float
            Root mean squared error (negative for Bayesian optimization)
        flow_map : xarray.DataArray
            Flow map for all conditions
        pred_deficits : xarray.DataArray
            Predicted deficits
        """
        # Create wind farm model
        wfm = self.create_wfm(params)
        
        # Run simulation for all conditions
        sim_res = wfm(
            [0], [0],
            ws=self.full_ws,
            TI=self.full_ti,
            wd=[270] * self.num_conditions,
            time=True
        )
        
        # Compute flow maps and deficits
        flow_maps = []
        for t in range(self.num_conditions):
            fm = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y),
                time=[t]
            )['WS_eff']
            flow_maps.append(fm)
        
        flow_map = xr.concat(flow_maps, dim='time')
        
        # Compute predicted deficits (velocity deficits)
        # Note: flow_map.WS_eff is the wake velocity
        pred_deficits = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
        
        # Compute RMSE
        rmse = float(np.sqrt(((self.observed_deficits - pred_deficits) ** 2)
                           .mean(['x', 'y'])).mean('time'))
        
        if np.isnan(rmse):
            return -0.5, flow_map, pred_deficits
            
        return -rmse, flow_map, pred_deficits

    def optimize(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        # Prepare operating conditions
        self._prepare_operating_conditions()
        
        # Get bounds and defaults
        pbounds = self.get_parameter_bounds()
        defaults = self.get_default_parameters()
        
        # Define optimization function
        def evaluate_rmse(**params):
            rmse, _, _ = self.evaluate_model(params)
            return rmse
        
        # Setup optimizer
        self.optimizer = BayesianOptimization(
            f=evaluate_rmse,
            pbounds=pbounds,
            random_state=1
        )
        
        # Add default point
        self.optimizer.probe(params=defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        print(f"Optimization complete. Best RMSE: {self.best_rmse:.4f}")
        
        return self.best_params, self.best_rmse

    def analyze_best_model(self, save_plots=True):
        """Analyze the best model with enhanced error metrics."""
        if self.best_params is None:
            raise ValueError("Run optimization first")
        
        # Evaluate best model
        _, flow_map, pred_deficits = self.evaluate_model(self.best_params)
        
        # Calculate errors for each condition
        errors = []
        rmse_values = []
        error_stats = []
        
        for t in range(self.num_conditions):
            # Get observed and predicted deficits for this condition
            obs = self.observed_deficits.isel(time=t)
            pred = pred_deficits.isel(time=t)
            
            # Calculate error field
            error = obs - pred
            errors.append(error)
            
            # Calculate RMSE for this condition
            rmse = float(np.sqrt((error**2).mean(['x', 'y'])))
            rmse_values.append(rmse)
            
            # Calculate error statistics
            error_flat = error.values.flatten()
            error_stats.append({
                'mean': np.mean(error_flat),
                'std': np.std(error_flat),
                'p50': np.percentile(error_flat, 50),
                'p90': np.percentile(error_flat, 90),
                'p95': np.percentile(error_flat, 95),
                'rmse': rmse
            })
        
        # Convert to arrays for analysis
        errors_array = xr.concat(errors, dim='time')
        
        # Calculate overall statistics
        all_errors = errors_array.values.flatten()
        overall_stats = {
            'mean_error': np.mean(all_errors),
            'std_error': np.std(all_errors),
            'rmse_mean': np.mean(rmse_values),
            'rmse_std': np.std(rmse_values),
            'rmse_p90': np.percentile(rmse_values, 90),
            'max_error': np.max(np.abs(all_errors))
        }
        
        # Print summary
        print("\n" + "="*50)
        print("MODEL PERFORMANCE ANALYSIS")
        print("="*50)
        print(f"Overall RMSE: {overall_stats['rmse_mean']:.4f} ± {overall_stats['rmse_std']:.4f}")
        print(f"90th percentile RMSE: {overall_stats['rmse_p90']:.4f}")
        print(f"Mean error: {overall_stats['mean_error']:.4f}")
        print(f"Error standard deviation: {overall_stats['std_error']:.4f}")
        print(f"Maximum absolute error: {overall_stats['max_error']:.4f}")
        
        # Create enhanced flow field plots
        if save_plots:
            self._create_enhanced_plots(
                flow_map, pred_deficits, errors_array,
                error_stats, overall_stats
            )
        
        return overall_stats, error_stats
