        Initialize the optimizer.
        
        Args:
            model_type: 1 for BlondelSuperGaussian, 2 for TurboGaussian
            downstream: True for downstream analysis, False for upstream/blockage

        self.model_type = model_type
        self.downstream = downstream

    def load_reference_data(self):
        """Load reference data from NetCDF file."""
        self.dat = xr.load_dataset('./DTU10MW.nc')

            x=self.dat.x * self.D,
            y=self.dat.y * self.D

    def setup_roi(self):
        """Setup region of interest based on downstream/upstream flag."""
        if self.downstream:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        self.flow_roi = self.dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
        # Store for plotting
        self.X_LB = X_LB
        self.X_UB = X_UB
    
    def setup_conditions(self):

        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        # Create all combinations
        full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()

        assert full_ws.size == full_ti.size
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.wd = [270] * full_ti.size
        
        # Pre-compute observed values
        self.precompute_observed_values()
    
    def precompute_observed_values(self):

        obs_values = []

        wfm = All2AllIterative(
            Hornsrev1Site(),
            DTU10MW(),
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                      wd=self.wd, time=True)
        
        flow_map = sim_res.flow_map(HorizontalGrid(x=self.target_x, y=self.target_y))
        
        for t in range(flow_map.time.size):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            obs_values.append(observed_deficit.T)
        
        self.all_obs = xr.concat(obs_values, dim='time')
    
    def evaluate_model(self, **params):
        """
        Evaluate a model with given parameters.
        
        Returns negative RMSE for Bayesian optimization.
        """
        try:
            # Create model
            wfm = self.model_factory.create_wake_model(**params)
            
            # Run simulation
            sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                         wd=self.wd, time=True)
            
            # Compute flow maps for all time steps
            flow_map_list = []
            for t in range(len(self.full_ws)):

                    HorizontalGrid(x=self.target_x, y=self.target_y),

            flow_map = xr.concat(flow_map_list, dim='time')
            
            # Compute predicted deficit

    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        pbounds = self.get_parameter_bounds()
        defaults = self.get_default_parameters()
        
        self.optimizer = BayesianOptimization(
            f=self.evaluate_model,
            pbounds=pbounds,

        # Probe with default values
        self.optimizer.probe(params=defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
