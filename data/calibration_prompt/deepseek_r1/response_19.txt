import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr
from bayes_opt import BayesianOptimization

# PyWake imports
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum

from py_wake.examples.data.hornsrev1 import Hornsrev1Site

class WakeModelCalibrator:
    """Modular wake model calibration framework"""
    
    def __init__(self, model_type=2, downstream=True):
        """
        Initialize calibrator
        
        Parameters
        ----------
        model_type : int, optional
            1: BlondelSuperGaussianDeficit2020
            2: TurboGaussianDeficit
        downstream : bool, optional
            True for downstream wake, False for upstream blockage
        """
        self.MODEL = model_type
        self.DOWNWIND = downstream
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        self.site = Hornsrev1Site()
        
        # Load reference data
        self._load_reference_data()
        self._setup_evaluation_grid()
        
        # Store optimization results
        self.optimizer = None
        self.best_params = None
        self.best_rmse = None
        
    def _load_reference_data(self):
        """Load and prepare reference data"""
        dat = xr.load_dataset('./DTU10MW.nc')
        dat = dat.assign_coords(x=dat.x * self.D, y=dat.y * self.D)
        
        # Define region of interest
        if self.DOWNWIND:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        self.reference_data = dat.sel(x=roi_x, y=roi_y)
        self.X_LB = X_LB
        self.X_UB = X_UB
        
    def _setup_evaluation_grid(self):
        """Setup evaluation grid and operating conditions"""
        self.target_x = self.reference_data.x
        self.target_y = self.reference_data.y
        
        # Define operating conditions
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.tile(TIs, len(WSs))
        full_ws = np.repeat(WSs, len(TIs))
        
        self.operating_conditions = {
            'ws': full_ws,
            'ti': full_ti,
            'wd': [270] * len(full_ws)
        }
        
        # Prepare reference values for all conditions
        self._prepare_reference_values()
        
    def _prepare_reference_values(self):
        """Prepare reference deficit values for all operating conditions"""
        # Baseline simulation for interpolation
        sim_res = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )([0], [0], ws=self.operating_conditions['ws'], 
           TI=self.operating_conditions['ti'], 
           wd=self.operating_conditions['wd'], time=True)
        
        flow_map = sim_res.flow_map(HorizontalGrid(x=self.target_x, y=self.target_y))
        
        obs_values = []
        for t in range(flow_map.time.size):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.reference_data.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            obs_values.append(observed_deficit.T)
        
        self.reference_deficits = xr.concat(obs_values, dim='time')
        
    def create_wake_model(self, params):
        """
        Create wind farm model with given parameters
        
        Parameters
        ----------
        params : dict
            Model parameters
        
        Returns
        -------
        wfm : All2AllIterative
            Configured wind farm model
        """
        # Extract turbulence parameters (common to all models)
        turb_args = {'c': np.array([params.get('ch1', 0.73),
                                   params.get('ch2', 0.8325),
                                   params.get('ch3', -0.0325),
                                   params.get('ch4', -0.32)])}
        
        # Configure based on model type and direction
        if self.DOWNWIND:
            if self.MODEL == 1:
                # BlondelSuperGaussianDeficit2020 for downstream
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
                blockage_args = {}
            else:
                # TurboGaussianDeficit for downstream
                wake_deficitModel = TurboGaussianDeficit(
                    A=params.get('A', 0.04),
                    cTI=[params.get('cti1', 1.5), params.get('cti2', 0.8)],
                    ctlim=params.get('ctlim', 0.999),
                    ceps=params.get('ceps', 0.25),
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficitModel.WS_key = 'WS_jlk'
                blockage_args = {}
        else:
            # Upstream blockage
            if self.MODEL == 1:
                wake_deficitModel = BlondelSuperGaussianDeficit2020()
                blockage_args = {}
            else:
                wake_deficitModel = BlondelSuperGaussianDeficit2020()
                blockage_args = {
                    'ss_alpha': params.get('ss_alpha', 0.8888888888888888),
                    'ss_beta': params.get('ss_beta', 1.4142135623730951),
                    'r12p': np.array([params.get('rp1', -0.672), params.get('rp2', 0.4897)]),
                    'ngp': np.array([params.get('ng1', -1.381), params.get('ng2', 2.627),
                                   params.get('ng3', -1.524), params.get('ng4', 1.336)])
                }
                if self.MODEL == 2:
                    blockage_args['groundModel'] = Mirror()
        
        # Create wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficitModel,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def evaluate_model(self, wfm):
        """
        Evaluate model performance against reference data
        
        Parameters
        ----------
        wfm : All2AllIterative
            Wind farm model to evaluate
        
        Returns
        -------
        rmse : float
            Root mean square error
        predictions : xarray.DataArray
            Model predictions
        """
        # Run simulation for all operating conditions
        sim_res = wfm(
            [0], [0],
            ws=self.operating_conditions['ws'],
            TI=self.operating_conditions['ti'],
            wd=self.operating_conditions['wd'],
            time=True
        )
        
        # Calculate velocity deficits from flow map
        flow_maps = []
        for t in range(len(self.operating_conditions['ws'])):
            fm = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y),
                time=[t]
            )['WS_eff']
            flow_maps.append(fm)
        
        flow_map = xr.concat(flow_maps, dim='time')
        
        # Calculate velocity deficit: (U_inf - U_wake) / U_inf
        # flow_map.WS_eff is wake velocity, sim_res.WS is freestream velocity
        predictions = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
        
        # Calculate RMSE
        rmse = float(np.sqrt(((self.reference_deficits - predictions) ** 2)
                            .mean(['x', 'y'])).mean('time'))
        
        return rmse, predictions, flow_map
    
    def optimization_objective(self, **params):
        """
        Objective function for Bayesian optimization
        
        Parameters
        ----------
        params : dict
            Model parameters
        
        Returns
        -------
        neg_rmse : float
            Negative RMSE (for maximization)
        """
        try:
            wfm = self.create_wake_model(params)
            rmse, _, _ = self.evaluate_model(wfm)
            
            if np.isnan(rmse):
                return -0.5  # Penalize NaN results
                
            return -rmse
        except Exception as e:
            print(f"Error in objective function: {e}")
            return -1.0
    
    def get_parameter_bounds(self):

        if self.MODEL == 1:
            if self.DOWNWIND:
                return {
                    'a_s': (0.001, 0.5),

                'cti1': (0.01, 5),
                'cti2': (0.01, 5),
                'ceps': (0.01, 3),
                'ctlim': (0.01, 1),
                'ch1': (-1, 2),
                'ch2': (-1, 2),
                'ch3': (-1, 2),
                'ch4': (-1, 2),
            }
    
    def get_default_parameters(self):
        """Get default parameters for current model configuration"""
        if self.MODEL == 1:
            if self.DOWNWIND:
                return {
                    'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2,
                    'b_f': -0.68, 'c_f': 2.41,
                    'ch1': 0.73, 'ch2': 0.8325,
                    'ch3': -0.0325, 'ch4': -0.32
                }
            else:
                return {

                'ctlim': 0.999,

    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization"""
        pbounds = self.get_parameter_bounds()
        defaults = self.get_default_parameters()
        
        self.optimizer = BayesianOptimization(
            f=self.optimization_objective,
            pbounds=pbounds,
            random_state=1
        )
        
        # Probe with default values
        self.optimizer.probe(params=defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        return self.optimizer
    
    def create_optimization_animation(self, filename):

        if self.optimizer is None:
            raise ValueError("Run optimization first")
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))

            ax2.clear()
            
            # Track best RMSE so far
            best_so_far_rmses = []
            best_so_far_rmse = float('inf')
            
            for i in range(frame + 1):
                current_rmse = -self.optimizer.space.target[i]
                if current_rmse < best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot optimization history
            ax1.plot(-np.array(self.optimizer.space.target[:frame+1]),

            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Plot parameter values for best solution so far

            best_vals = [self.best_params[k] for k in keys]

            ax2.bar(keys, best_vals, label='Optimized')

                   linewidth=2, color='none', label='Default')
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.tick_params(axis='x', rotation=45)
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(self.optimizer.space.target),
