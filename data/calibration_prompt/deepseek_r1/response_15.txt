    def calculate_deficits(self, sim_res, flow_map):
        """Calculate velocity deficits from simulation results"""
        # Note: flow_map.WS_eff is the effective velocity (u_inf - deficit)
        # So deficit = (u_inf - WS_eff) / u_inf
        u_inf = sim_res.WS  # Incoming wind speed
        deficits = (u_inf - flow_map.WS_eff) / u_inf
        return deficits
    
    def evaluate_objective(self, **kwargs) -> float:
        """Objective function for optimization"""
        try:
            # Create wind farm model with current parameters
            wfm = self.create_wind_farm_model(**kwargs)
            
            # Run simulation
            sim_res = wfm(
                [0], [0],
                ws=self.operating_conditions['ws'],
                TI=self.operating_conditions['ti'],
                wd=self.operating_conditions['wd'],
                time=True

            # Calculate flow field
            grid = HorizontalGrid(x=self.target_x, y=self.target_y)
            flow_map = sim_res.flow_map(grid)
            
            # Calculate predicted deficits
            pred_deficits = self.calculate_deficits(sim_res, flow_map)
            
            # Calculate error metrics
            errors = self.all_obs - pred_deficits.isel(h=0)
            
            # Multiple error metrics
            rmse = float(np.sqrt((errors**2).mean(['x', 'y'])).mean('time'))
            
            # Also calculate other statistics
            mae = float(np.abs(errors).mean(['x', 'y']).mean('time'))
            max_error = float(np.abs(errors).max(['x', 'y']).mean('time'))
            
            # Combine metrics (you can adjust weights)
            # Negative because BayesianOptimization maximizes
            score = -rmse  # - 0.1 * mae - 0.05 * max_error
            
            # Store additional metrics
            if not hasattr(self, 'additional_metrics'):
                self.additional_metrics = []
            self.additional_metrics.append({
                'rmse': rmse,
                'mae': mae,
                'max_error': max_error,
                'params': kwargs.copy()
            })
            
            if np.isnan(score):
                return -10.0  # Penalize NaN results
                
            return score
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -10.0  # Penalize failed evaluations
    
    def run_optimization(self, n_iter: int = 100, init_points: int = 20):
        """Run Bayesian optimization"""
        self.optimizer = BayesianOptimization(
            f=self.evaluate_objective,
            pbounds=self.pbounds,
            random_state=1,
            verbose=2
        )
        
        # Add default parameters as initial point
        self.optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_score = -self.optimizer.max['target']
        
        return self.best_params, self.best_score
