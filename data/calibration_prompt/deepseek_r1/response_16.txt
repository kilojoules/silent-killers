    def __init__(self, data_path='./DTU10MW.nc', downwind=True, model=2):

    def load_reference_data(self, data_path):
        """Load and preprocess reference data."""
        dat = xr.load_dataset(data_path)
        D = DTU10MW().diameter()
        dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
        
        # Define region of interest
        if self.downwind:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * D, X_UB * D)
        roi_y = slice(-2 * D, 2 * D)
        
        self.flow_roi = dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        self.X_LB, self.X_UB = X_LB, X_UB
        
    def setup_wind_conditions(self):
        """Setup wind speed and turbulence intensity combinations."""
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        # Create all combinations
        full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
        full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.full_wd = [270] * full_ti.size
        
        # Pre-compute observed deficits for all conditions
        self.prepare_observed_deficits()
        
    def prepare_observed_deficits(self):
        """Extract observed deficits for all wind conditions."""
        obs_values = []
        
        # Create a baseline simulation to get CT values
        baseline_wfm = self.create_wake_model({})
        sim_res = baseline_wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                              wd=self.full_wd, time=True)
        
        for t in range(sim_res.time.size):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            obs_values.append(observed_deficit.T)
            
        self.observed_deficits = xr.concat(obs_values, dim='time')
        
    def create_wake_model(self, params):
        """
        Create wake model with given parameters.
        
        Parameters:
        -----------
        params : dict
            Dictionary of model parameters
        
        Returns:
        --------
        wfm : WindFarmModel
            Configured wake model
        """
        if self.downwind:
            if self.model == 1:
                wake_deficit = self.create_blondel_model(params)
                turb_args = {'c': np.array([params.get('ch1', 0.73),
                                           params.get('ch2', 0.8325),
                                           params.get('ch3', -0.0325),
                                           params.get('ch4', -0.32)])}
                blockage_args = {}
            else:  # model == 2
                wake_deficit = self.create_turbogaussian_model(params)
                turb_args = {'c': np.array([params.get('ch1', 0.73),
                                           params.get('ch2', 0.8325),
                                           params.get('ch3', -0.0325),
                                           params.get('ch4', -0.32)])}
                blockage_args = {}
        else:
            # Upstream/blockage configuration
            wake_deficit = BlondelSuperGaussianDeficit2020()
            turb_args = {}
            blockage_args = {
                'ss_alpha': params.get('ss_alpha', 0.8889),
                'ss_beta': params.get('ss_beta', 1.4142),
                'r12p': np.array([params.get('rp1', -0.672),
                                 params.get('rp2', 0.4897)]),
                'ngp': np.array([params.get('ng1', -1.381),
                                params.get('ng2', 2.627),
                                params.get('ng3', -1.524),
                                params.get('ng4', 1.336)])
            }
            if self.model == 2:
                blockage_args['groundModel'] = Mirror()
        
        # Create wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficit,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def create_blondel_model(self, params):
        """Create BlondelSuperGaussianDeficit2020 model."""
        def_args = {
            'a_s': params.get('a_s', 0.17),
            'b_s': params.get('b_s', 0.005),
            'c_s': params.get('c_s', 0.2),
            'b_f': params.get('b_f', -0.68),
            'c_f': params.get('c_f', 2.41)
        }
        return BlondelSuperGaussianDeficit2020(**def_args)
    
    def create_turbogaussian_model(self, params):
        """Create TurboGaussianDeficit model."""
        model = TurboGaussianDeficit(
            A=params.get('A', 0.04),
            cTI=[params.get('cti1', 1.5), params.get('cti2', 0.8)],
            ctlim=params.get('ctlim', 0.999),
            ceps=params.get('ceps', 0.25),
            ct2a=ct2a_mom1d,
            groundModel=Mirror(),
            rotorAvgModel=GaussianOverlapAvgModel()
        )
        model.WS_key = 'WS_jlk'
        return model
    
    def define_parameter_bounds(self):

        if self.model == 1:
            if self.downwind:

    def evaluate_model(self, params):
        """
        Evaluate model with given parameters.
        
        Returns negative RMSE for Bayesian optimization.
        """
        # Create wake model with current parameters
        wfm = self.create_wake_model(params)
        
        # Run simulation
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                     wd=self.full_wd, time=True)
        
        # Calculate velocity deficits
        grid = HorizontalGrid(x=self.target_x, y=self.target_y)
        flow_map = sim_res.flow_map(grid)
        
        # Compute predicted deficits
        # Note: flow_map.WS_eff is the velocity at grid points
        # We compute deficit as: (U_inf - U_wake) / U_inf
        predicted_deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        
        # Calculate RMSE
        rmse = float(np.sqrt(((self.observed_deficits - predicted_deficits) ** 2)
                            .mean(['x', 'y'])).mean('time'))
        
        return -rmse if not np.isnan(rmse) else -0.5
    
    def optimize(self, n_iter=200, init_points=50):
        """Run Bayesian optimization."""
        optimizer = BayesianOptimization(
            f=self.evaluate_model,
            pbounds=self.pbounds,
            random_state=1,
            verbose=2
        )
        
        # Start with default parameters
        optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        self.optimizer = optimizer
        self.best_params = optimizer.max['params']
        self.best_rmse = -optimizer.max['target']
        
        return optimizer

            best_so_far_rmse = float('inf')
            
            for i in range(frame + 1):
                current_rmse = -self.optimizer.space.target[i]
                if current_rmse <= best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(self.optimizer.space.target[:frame+1]), 
                    color='gray', alpha=0.5, label='All evaluations')
            ax1.plot(np.array(best_so_far_rmses), color='black', 
                    linewidth=2, label='Best so far')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            ax1.legend()
            
            # Plot current best parameters
            keys = list(self.best_params.keys())
            best_vals = [self.best_params[k] for k in keys]
            default_vals = [self.defaults.get(k, 0) for k in keys]
            
            x_pos = np.arange(len(keys))

            ax2.bar(x_pos + 0.2, default_vals, 0.4, label='Default', alpha=0.7)
            ax2.set_xticks(x_pos)
            ax2.set_xticklabels(keys, rotation=45)
            ax2.set_title(f'Best RMSE: {self.best_rmse:.4f}')
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(self.optimizer.space.target),
            repeat=False
        )
        
        # Save animation
        writer = animation.FFMpegWriter(fps=15)
        ani.save(filename, writer=writer)
        plt.close(fig)
    
    def analyze_results(self, save_path='.'):
        """Analyze and visualize optimization results."""
        # Create wake model with best parameters
        best_wfm = self.create_wake_model(self.best_params)
        sim_res = best_wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                          wd=self.full_wd, time=True)
        
        # Calculate errors for each time step
        grid = HorizontalGrid(x=self.target_x, y=self.target_y)
        flow_map = sim_res.flow_map(grid)
        predicted_deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        
        # Calculate error statistics
        errors = []
        rmse_values = []
        abs_errors = []
        
        for t in range(sim_res.time.size):
            obs = self.observed_deficits.isel(time=t)
            pred = predicted_deficits.isel(time=t)
            error = obs - pred
            
            errors.append(error)
            rmse = float(np.sqrt(np.mean(error**2)))
            rmse_values.append(rmse)
            abs_errors.append(np.abs(error).values.flatten())
        
        # Overall statistics
        all_errors = np.concatenate([e.values.flatten() for e in errors])
        mean_error = np.mean(all_errors)
        std_error = np.std(all_errors)
        p90_error = np.percentile(np.abs(all_errors), 90)
        
        print(f"Overall RMSE: {np.mean(rmse_values):.4f}")
        print(f"Mean absolute error: {np.mean(np.abs(all_errors)):.4f}")
        print(f"Error standard deviation: {std_error:.4f}")
        print(f"90th percentile absolute error: {p90_error:.4f}")
        
        # Create detailed error plots
        self.create_error_plots(errors, rmse_values, save_path)
        
        # Plot parameter comparison
        self.plot_parameter_comparison(save_path)
        
        return rmse_values, errors

            'Value': [np.mean(rmse_values), np.std(rmse_values),

                     np.percentile(np.abs(all_errors), 90),
                     np.max(np.abs(all_errors))]
        })
        
        summary_stats.to_csv(f'{save_path}/summary_stats_{self.X_LB}_{self.X_UB}.csv', 
                            index=False)
