        """

        """

    def load_reference_data(self, data_path):
        """Load and preprocess reference data."""
        dat = xr.load_dataset(data_path)
        dat = dat.assign_coords(x=dat.x * self.D, y=dat.y * self.D)
        
        # Define region of interest
        if self.downwind:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        self.X_LB, self.X_UB = X_LB, X_UB
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        self.flow_roi = dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
        # Setup operating conditions
        self.TIs = np.arange(0.05, 0.45, 0.05)
        self.WSs = np.arange(4, 11)
        
        full_ti = np.tile(self.TIs, len(self.WSs))
        full_ws = np.repeat(self.WSs, len(self.TIs))
        
        assert full_ws.size == full_ti.size
        self.full_ws = full_ws
        self.full_ti = full_ti
        
        # Extract observed deficits
        self.prepare_observed_deficits()
        
    def prepare_observed_deficits(self):
        """Extract observed deficits from reference data using a baseline simulation."""
        # Run baseline simulation to get CT values for each condition
        baseline_wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        baseline_res = baseline_wfm(
            [0], [0], 
            ws=self.full_ws, 
            TI=self.full_ti, 
            wd=[270] * self.full_ti.size, 
            time=True
        )
        
        flow_map = baseline_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        # Extract observed deficits for each time step
        obs_values = []
        for t in range(flow_map.time.size):
            this_pred_sim = baseline_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, 
                ti=this_pred_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
            
        self.observed_deficits = xr.concat(obs_values, dim='time')

                    'fg1': (-2, 2),
                    'fg2': (-2, 2),
                    'fg3': (-2, 2),

        else:  # model == 2

                'cti1': (.01, 5),
                'cti2': (0.01, 5),
                'ceps': (0.01, 3),
                'ctlim': (0.01, 1),
                'ch1': (-1, 2),
                'ch2': (-1, 2),
                'ch3': (-1, 2),
                'ch4': (-1, 2),
            }
            self.defaults = {

                'ch1': 0.73,

    def create_wind_farm_model(self, params):
        """Create a wind farm model with given parameters."""
        # Extract turbulence parameters (common for all models)
        turb_params = {'c': np.array([
            params.get('ch1', self.defaults['ch1']),
            params.get('ch2', self.defaults['ch2']),
            params.get('ch3', self.defaults['ch3']),
            params.get('ch4', self.defaults['ch4'])

        # Create wake deficit model based on model type
        if self.downwind:

                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}

                    ceps=params['ceps'],

        else:  # upwind
            if self.model == 1:
                wake_deficit_model = BlondelSuperGaussianDeficit2020()
                blockage_args = {}
            else:  # model == 2
                wake_deficit_model = BlondelSuperGaussianDeficit2020()
                blockage_args = {
                    'ss_alpha': params['ss_alpha'],

    def evaluate_model(self, params):
        """Evaluate model with given parameters and return RMSE."""
        try:
            # Create wind farm model
            wfm = self.create_wind_farm_model(params)
            
            # Run simulation
            sim_res = wfm(
                [0], [0],
                ws=self.full_ws,
                TI=self.full_ti,
                wd=[270] * self.full_ti.size,
                time=True
            )
            
            # Calculate predicted deficits
            flow_map = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y)
            )
            
            # Note: flow_map.WS_eff is velocity, convert to deficit
            predicted_deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
            
            # Calculate RMSE
            rmse = float(np.sqrt(
                ((self.observed_deficits - predicted_deficits) ** 2).mean(['x', 'y'])
            ).mean('time'))
            
            # Return negative RMSE for minimization
            return -rmse if not np.isnan(rmse) else -0.5
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5  # Return poor score on error
    
    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        # Create optimizer
        optimizer = BayesianOptimization(
            f=self.evaluate_model,
            pbounds=self.pbounds,
            random_state=1
        )
        
        # Probe with defaults
        optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        self.optimizer = optimizer
        self.best_params = optimizer.max['params']
        self.best_rmse = -optimizer.max['target']
        
        print(f"Optimization complete. Best RMSE: {self.best_rmse:.4f}")
        
        return optimizer

        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Track best RMSE so far
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                current_rmse = -self.optimizer.space.target[i]
                if current_rmse <= best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(self.optimizer.space.target[:frame+1]), 
                    color='gray', alpha=0.5, label='All evaluations')
            ax1.plot(best_so_far_rmses, color='black', label='Best so far')
            ax1.set_title('Optimization Convergence')

            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            ax1.legend()
            
            # Plot current best parameters
            current_best_idx = np.argmin(-self.optimizer.space.target[:frame+1])
            current_params = self.optimizer.res[current_best_idx]['params']
            
            keys = list(current_params.keys())
            best_vals = [current_params[k] for k in keys]
            default_vals = [self.defaults.get(k, 0) for k in keys]
            
            ax2.bar(keys, best_vals, alpha=0.7, label='Current best')
            ax2.bar(keys, default_vals, edgecolor='black',

            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.tick_params(axis='x', rotation=45)
            ax2.legend()
            
            plt.tight_layout()

        ani = animation.FuncAnimation(

        )
        
        # Save animation

        plt.close(fig)
        
        return ani
    
    def evaluate_best_model(self, save_plots=True):
        """Evaluate the best model and create detailed plots."""
        if not hasattr(self, 'best_params'):
            raise ValueError("Run optimization first")
        
        # Create model with best parameters
        wfm = self.create_wind_farm_model(self.best_params)
        
        # Run simulation
        sim_res = wfm(
            [0], [0],

            wd=[270] * self.full_ti.size,

        flow_map = sim_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        # Calculate statistics
        predicted_deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        
        # Calculate errors per time step
        rmse_values = []
        mae_values = []
        max_error_values = []
        
        for t in range(len(self.full_ws)):
            obs = self.observed_deficits.isel(time=t)
            pred = predicted_deficits.isel(time=t)
            
            error = obs - pred
            rmse = np.sqrt(np.mean(error**2))
            mae = np.mean(np.abs(error))
            max_err = np.max(np.abs(error))
            
            rmse_values.append(rmse)
            mae_values.append(mae)
            max_error_values.append(max_err)
            
            if save_plots:
                self.create_error_plots(t, obs, pred, error, sim_res.isel(time=t))
        
        # Create summary statistics
        stats = {
            'overall_rmse': np.mean(rmse_values),
            'rmse_90th': np.percentile(rmse_values, 90),
            'rmse_values': rmse_values,
            'mae_values': mae_values,
            'max_error_values': max_error_values,
            'average_mae': np.mean(mae_values),
            'average_max_error': np.mean(max_error_values)
        }
        
        print(f"Overall RMSE: {stats['overall_rmse']:.4f}")
        print(f"90th percentile RMSE: {stats['rmse_90th']:.4f}")
        print(f"Average MAE: {stats['average_mae']:.4f}")
        
        # Create parameter comparison plot
        self.create_parameter_comparison_plot()
        
        return stats, sim_res, flow_map
