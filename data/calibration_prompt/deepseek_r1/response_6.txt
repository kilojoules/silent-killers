class DataManager:
    """Manages loading and preparing data for optimization"""
    
    def __init__(self, config: Config):
        self.config = config
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        
    def load_data(self, filepath: str = './DTU10MW.nc') -> xr.Dataset:
        """Load and prepare reference data"""
        dat = xr.load_dataset(filepath)
        dat = dat.assign_coords(x=dat.x * self.D, y=dat.y * self.D)
        return dat
    
    def get_region_of_interest(self, dat: xr.Dataset) -> xr.Dataset:
        """Extract region of interest from data"""
        roi_x = slice(self.config.x_lb * self.D, self.config.x_ub * self.D)
        roi_y = slice(self.config.y_lb * self.D, self.config.y_ub * self.D)
        return dat.sel(x=roi_x, y=roi_y)
    
    def get_operating_conditions(self) -> Tuple[np.ndarray, np.ndarray]:
        """Generate operating conditions (WS, TI) for simulation"""
        TIs = np.arange(self.config.ti_range[0], self.config.ti_range[1] + 0.001, 
                       self.config.ti_step)
        WSs = np.arange(self.config.ws_range[0], self.config.ws_range[1] + 0.001,
                       self.config.ws_step)
        
        full_ti = np.tile(TIs, len(WSs))
        full_ws = np.repeat(WSs, len(TIs))
        
        return full_ws, full_ti
    
    def extract_observed_deficits(self, flow_roi: xr.Dataset, 
                                 sim_results: xr.Dataset) -> xr.Dataset:
        """Extract observed deficits for all simulation conditions"""
        obs_values = []
        
        for t in range(sim_results.time.size):
            this_pred_sim = sim_results.isel(time=t, wt=0)
            observed_deficit = flow_roi.deficits.interp(
                ct=this_pred_sim.CT, 
                ti=this_pred_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
        
        return xr.concat(obs_values, dim='time')


class Optimizer:
    """Handles Bayesian optimization process"""
    
    def __init__(self, config: Config, model_factory: WakeModelFactory,
                 data_manager: DataManager):
        self.config = config
        self.model_factory = model_factory
        self.data_manager = data_manager
        
        # Load and prepare data
        self.dat = data_manager.load_data()
        self.flow_roi = data_manager.get_region_of_interest(self.dat)
        self.full_ws, self.full_ti = data_manager.get_operating_conditions()
        
        # Get target grid
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
        # Create grid for flow maps
        self.grid = HorizontalGrid(x=self.target_x, y=self.target_y)
        
    def _compute_rmse(self, params: Dict) -> float:
        """Compute RMSE for given parameters"""
        try:
            # Create WFM with current parameters
            wfm = self.model_factory.create_wfm(params)
            
            # Run simulation for all conditions
            sim_res = wfm(
                [0], [0],  # Single turbine at origin
                ws=self.full_ws,
                TI=self.full_ti,
                wd=[270] * len(self.full_ws),
                time=True
            )
            
            # Get flow map for all time steps at once
            flow_map = sim_res.flow_map(self.grid)
            
            # Calculate predicted deficit
            # Note: flow_map.WS_eff is the effective velocity, sim_res.WS is freestream
            predicted_deficit = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
            
            # Extract observed deficits
            observed_deficits = self.data_manager.extract_observed_deficits(
                self.flow_roi, sim_res
            )
            
            # Calculate RMSE
            rmse = float(np.sqrt(((observed_deficits - predicted_deficit) ** 2)
                               .mean(['x', 'y'])).mean('time'))
            
            return -rmse if not np.isnan(rmse) else -0.5
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def get_parameter_bounds(self) -> Dict:
        """Get parameter bounds based on configuration"""
        if self.config.downstream:
            if self.config.model_type == 1:
                return {
                    'a_s': (0.001, 0.5),
                    'b_s': (0.001, 0.01),
                    'c_s': (0.001, 0.5),
                    'b_f': (-2, 1),
                    'c_f': (0.1, 5),
                    'ch1': (-1, 2),
                    'ch2': (-1, 2),
                    'ch3': (-1, 2),
                    'ch4': (-1, 2),
                }
            else:  # Model 2
                return {

                    'cti1': (0.01, 5),
                    'cti2': (0.01, 5),
                    'ceps': (0.01, 3),
                    'ctlim': (0.01, 1),
                    'ch1': (-1, 2),
                    'ch2': (-1, 2),
                    'ch3': (-1, 2),
                    'ch4': (-1, 2),
                }
        else:  # Upstream
            return {
                'ss_alpha': (0.05, 3),
                'ss_beta': (0.05, 3),
                'rp1': (-2, 2),
                'rp2': (-2, 2),
                'ng1': (-3, 3),
                'ng2': (-3, 3),
                'ng3': (-3, 3),
                'ng4': (-3, 3),
            }
    
    def get_default_parameters(self) -> Dict:
        """Get default parameters based on configuration"""
        if self.config.downstream:
            if self.config.model_type == 1:
                return {
                    'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2,
                    'b_f': -0.68, 'c_f': 2.41,
                    'ch1': 0.73, 'ch2': 0.8325,
                    'ch3': -0.0325, 'ch4': -0.32
                }
            else:  # Model 2
                return {
                    'A': 0.04, 'cti1': 1.5, 'cti2': 0.8,

                    'ch1': 0.73, 'ch2': 0.8325,
                    'ch3': -0.0325, 'ch4': -0.3

        else:  # Upstream

                'ss_beta': 1.4142135623730951,
                'rp1': -0.672, 'rp2': 0.4897,
                'ng1': -1.381, 'ng2': 2.627,
                'ng3': -1.524, 'ng4': 1.336,
            }
    
    def run_optimization(self, init_points: int = 50, n_iter: int = 200) -> Dict:
        """Run Bayesian optimization"""
        pbounds = self.get_parameter_bounds()
        defaults = self.get_default_parameters()
        
        optimizer = BayesianOptimization(
            f=self._compute_rmse,
            pbounds=pbounds,
            random_state=1
        )
        
        optimizer.probe(params=defaults, lazy=True)
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        best_params = optimizer.max['params']
        best_rmse = -optimizer.max['target']
        
        return {
            'optimizer': optimizer,
            'best_params': best_params,
            'best_rmse': best_rmse,
            'defaults': defaults
        }
