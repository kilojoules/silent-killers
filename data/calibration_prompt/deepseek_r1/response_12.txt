        dat = xr.load_dataset(self.data_path)
        turbine = DTU10MW()
        D = turbine.diameter()
        
        # Scale coordinates by turbine diameter
        dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
        
        # Define region of interest
        if self.downwind:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
        
        roi_x = slice(X_LB * D, X_UB * D)
        roi_y = slice(-2 * D, 2 * D)

    def _setup_wind_conditions(self):

        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.tile(TIs, WSs.size)
        full_ws = np.repeat(WSs, TIs.size)
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.wd = np.full_like(full_ws, 270)  # Fixed wind direction
        
        # Prepare reference values
        self._prepare_reference_values()
    
    def _prepare_reference_values(self):
        """Prepare reference values for comparison."""
        obs_values = []
        
        # Run a baseline simulation to get reference values
        sim_res = All2AllIterative(

            wake_deficitModel=BlondelSuperGaussianDeficit2020(),

            deflectionModel=None,

    def create_wfm(self, params):

    def evaluate_rmse(self, **params):
        """
        Evaluate RMSE for given parameters.
        
        Parameters
        ----------
        params : dict
            Parameters for the wake model
            
        Returns
        -------
        negative_rmse : float
            Negative RMSE (for minimization)
        """
        wfm = self.create_wfm(params)
        
        # Run simulation
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                      wd=self.wd, time=True)
        
        # Calculate flow map for all time steps
        flow_maps = []
        for t in range(len(self.full_ws)):
            fm = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y),
                time=[t]
            )['WS_eff']
            flow_maps.append(fm)
        
        flow_map = xr.concat(flow_maps, dim='time')
        
        # Calculate velocity deficit
        pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
        
        # Calculate RMSE
        rmse = float(np.sqrt(((self.all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
        
        if np.isnan(rmse):
            return -0.5  # Penalize NaN results
        
        return -rmse  # Negative for minimization
    
    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        self.optimizer = BayesianOptimization(
            f=self.evaluate_rmse,
            pbounds=self.pbounds,
            random_state=1
        )
        
        # Probe with defaults first
        self.optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        return self.best_params, self.best_rmse

            for i in range(frame + 1):
                if -self.optimizer.space.target[i] <= best_so_far_rmse:

            ax1.set_title('Optimization Convergence')

            ax1.set_ylabel('RMSE')
