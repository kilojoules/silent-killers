import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr
from py_wake import HorizontalGrid
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020

class WakeModelOptimizer:
    """Class for optimizing wake model parameters using Bayesian Optimization."""
    
    def __init__(self, 
                 downwind: bool = True, 
                 model_type: int = 2,
                 x_lb: float = 2,
                 x_ub: float = 10):
        """
        Initialize the optimizer.
        
        Args:
            downwind: Whether to optimize for downwind (True) or upstream (False) region
            model_type: Model type (1 for BlondelSuperGaussian, 2 for TurboGaussian)
            x_lb: Lower bound for x-coordinate in diameters
            x_ub: Upper bound for x-coordinate in diameters
        """
        self.downwind = downwind
        self.model_type = model_type
        self.x_lb = x_lb
        self.x_ub = x_ub
        
        # Load and prepare data
        self.load_data()
        self.prepare_target_grid()
        self.prepare_wind_conditions()
        
        # Initialize site and turbine
        self.site = Hornsrev1Site()
        self.turbine = DTU10MW()
        
        # Default parameters
        self.default_params = self.get_default_params()
        self.pbounds = self.get_parameter_bounds()
        
        # Results storage
        self.optimizer = None
        self.best_params = None
        self.best_rmse = None
        
    def load_data(self):
        """Load and preprocess the reference data."""
        dat = xr.load_dataset('./DTU10MW.nc')
        D = self.turbine.diameter() if hasattr(self, 'turbine') else 178.3  # DTU10MW diameter
        dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
        self.reference_data = dat
        
    def prepare_target_grid(self):
        """Prepare the target grid for flow mapping."""
        D = self.turbine.diameter() if hasattr(self, 'turbine') else 178.3
        
        if self.downwind:
            roi_x = slice(self.x_lb * D, self.x_ub * D)
        else:
            roi_x = slice(self.x_lb * D, self.x_ub * D)
            
        roi_y = slice(-2 * D, 2 * D)
        
        self.flow_roi = self.reference_data.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
    def prepare_wind_conditions(self):
        """Prepare the wind conditions for evaluation."""
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.repeat(TIs, len(WSs))
        full_ws = np.tile(WSs, len(TIs))
        
        # Ensure same ordering
        idx = np.argsort(np.repeat(range(len(WSs)), len(TIs)) * 100 + np.tile(range(len(TIs)), len(WSs)))
        full_ti = full_ti[idx]
        full_ws = full_ws[idx]
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.wd = [270] * len(full_ws)
        
        # Prepare observed deficits
        self.prepare_observed_deficits()
        
    def prepare_observed_deficits(self):
        """Prepare observed deficit values from reference data."""
        # Create a baseline simulation to get CT and TI values
        sim_res = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )([0], [0], ws=self.full_ws, TI=self.full_ti, wd=self.wd, time=True)
        
        obs_values = []
        for t in range(len(self.full_ws)):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, 
                ti=this_pred_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
            
        self.all_obs = xr.concat(obs_values, dim='time')
        
    def get_default_params(self) -> Dict:
        """Get default parameters based on model type and direction."""
        if self.model_type == 1:
            if self.downwind:
                return {
                    'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2, 
                    'b_f': -0.68, 'c_f': 2.41,
                    'ch1': 0.73, 'ch2': 0.8325, 'ch3': -0.0325, 'ch4': -0.32
                }
            else:
                return {
                    'ss_alpha': 0.8888888888888888,
                    'ss_beta': 1.4142135623730951,
                    'rp1': -0.672, 'rp2': 0.4897,
                    'ng1': -1.381, 'ng2': 2.627, 'ng3': -1.524, 'ng4': 1.336,
                    'fg1': -0.06489, 'fg2': 0.4911, 'fg3': 1.116, 'fg4': -0.1577
                }
        else:  # model_type == 2
            return {
                'A': 0.04, 'cti1': 1.5, 'cti2': 0.8,
                'ceps': 0.25, 'ctlim': 0.999,
                'ch1': 0.73, 'ch2': 0.8325, 'ch3': -0.0325, 'ch4': -0.3
            }
            
    def get_parameter_bounds(self) -> Dict:
        """Get parameter bounds for Bayesian optimization."""
        if self.model_type == 1:
            if self.downwind:
                return {
                    'a_s': (0.001, 0.5), 'b_s': (0.001, 0.01), 'c_s': (0.001, 0.5),
                    'b_f': (-2, 1), 'c_f': (0.1, 5),
                    'ch1': (-1, 2), 'ch2': (-1, 2), 'ch3': (-1, 2), 'ch4': (-1, 2)
                }
            else:
                return {
                    'ss_alpha': (0.05, 3), 'ss_beta': (0.05, 3),
                    'rp1': (-2, 2), 'rp2': (-2, 2),
                    'ng1': (-3, 3), 'ng2': (-3, 3), 'ng3': (-3, 3), 'ng4': (-3, 3),
                    'fg1': (-2, 2), 'fg2': (-2, 2), 'fg3': (-2, 2), 'fg4': (-2, 2)
                }
        else:  # model_type == 2
            return {
                'A': (0.001, 0.5), 'cti1': (0.01, 5), 'cti2': (0.01, 5),
                'ceps': (0.01, 3), 'ctlim': (0.01, 1),
                'ch1': (-1, 2), 'ch2': (-1, 2), 'ch3': (-1, 2), 'ch4': (-1, 2)
            }
            
    def create_wake_model(self, params: Dict) -> All2AllIterative:
        """Create a wake farm model with given parameters."""
        
        # Extract turbulence parameters
        turb_args = {'c': np.array([params['ch1'], params['ch2'], 
                                   params['ch3'], params['ch4']])}
        
        if self.downwind:
            if self.model_type == 1:
                # BlondelSuperGaussian for downwind
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
                blockage_args = {}
            else:  # model_type == 2
                # TurboGaussian for downwind
                wake_deficit_model = TurboGaussianDeficit(
                    A=params['A'],
                    cTI=[params['cti1'], params['cti2']],
                    ctlim=params['ctlim'],
                    ceps=params['ceps'],
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficit_model.WS_key = 'WS_jlk'
                blockage_args = {}
        else:  # upstream
            # SelfSimilarityDeficit for upstream
            wake_deficit_model = BlondelSuperGaussianDeficit2020()
            blockage_args = {
                'ss_alpha': params.get('ss_alpha', 0.89),
                'ss_beta': params.get('ss_beta', 1.41),
                'r12p': np.array([params.get('rp1', -0.67), params.get('rp2', 0.49)]),
                'ngp': np.array([params.get('ng1', -1.38), params.get('ng2', 2.63),
                               params.get('ng3', -1.52), params.get('ng4', 1.34)])
            }
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
        
        # Create the wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficit_model,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def evaluate_model(self, params: Dict) -> Tuple[float, xr.DataArray]:
        """Evaluate a model with given parameters and return RMSE and predictions."""
        wfm = self.create_wake_model(params)
        
        # Run simulation
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                      wd=self.wd, time=True)
        
        # Create flow map
        flow_map = sim_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        # Calculate velocity deficits (note: flow_map.WS_eff is velocity)
        # Convert to deficit: (WS_ref - WS_eff) / WS_ref
        # Where WS_ref is the upstream velocity (sim_res.WS)
        pred_deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        
        # Calculate RMSE
        rmse = float(np.sqrt(((self.all_obs - pred_deficits) ** 2).mean(['x', 'y'])).mean('time'))
        
        if np.isnan(rmse):
            rmse = 0.5  # Penalty for NaN results
            
        return rmse, pred_deficits
    
    def objective_function(self, **kwargs) -> float:
        """Objective function for Bayesian optimization (negative RMSE)."""
        rmse, _ = self.evaluate_model(kwargs)
        return -rmse
    
    def run_optimization(self, init_points: int = 50, n_iter: int = 200):
        """Run Bayesian optimization."""
        self.optimizer = BayesianOptimization(
            f=self.objective_function,
            pbounds=self.pbounds,
            random_state=1
        )
        
        # Probe with default parameters
        self.optimizer.probe(params=self.default_params, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        return self.best_params, self.best_rmse
    
    def create_optimization_animation(self, filename: str = 'optimization_animation.mp4'):
        """Create an animation of the optimization convergence."""
        if self.optimizer is None:
            raise ValueError("Optimization must be run first")
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Get best parameters and RMSE up to current frame
            best_so_far_rmse = float('inf')
            best_so_far_params = {}
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                current_rmse = -self.optimizer.space.target[i]
                if current_rmse <= best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                    best_so_far_params = self.optimizer.res[i]['params']
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(self.optimizer.space.target), color='gray', alpha=0.5)
            ax1.plot(best_so_far_rmses, color='black')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Plot parameter comparison
            keys = list(best_so_far_params.keys())
            best_vals = [best_so_far_params[k] for k in keys]
            default_vals = [self.default_params.get(k, 0) for k in keys]
            
            ax2.bar(keys, best_vals, label='Optimized')
            ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, 
                   color='none', label='Default')
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.tick_params(axis='x', rotation=45)
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(self.optimizer.space.target), 
            repeat=False
        )
        
        writer = animation.FFMpegWriter(fps=15)
        ani.save(filename, writer=writer)
        plt.close(fig)
        
    def plot_flow_field_comparison(self, save_dir: str = 'figs'):
        """Plot flow field comparisons with error statistics."""
        if self.best_params is None:
            raise ValueError("Optimization must be run first")
        
        rmse_values = []
        all_errors = []
        
        # Create best model
        wfm = self.create_wake_model(self.best_params)
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                      wd=self.wd, time=True)
        
        flow_map = sim_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        for t in range(len(self.full_ws)):
            # Get predictions
            pred_deficit = (sim_res.isel(time=t).WS - 
                          flow_map.WS_eff.isel(h=0, time=t)) / sim_res.isel(time=t).WS
            
            # Get observations
            observed_deficit = self.flow_roi.deficits.interp(
                ct=sim_res.isel(time=t).CT,
                ti=sim_res.isel(time=t).TI,
                z=0
            ).isel(wt=0)
            
            # Calculate errors
            error = observed_deficit.T - pred_deficit
            rmse = np.sqrt(np.mean(error**2))
            
            rmse_values.append(rmse)
            all_errors.append(error.values.flatten())
            
            # Create comparison plot
            fig, axes = plt.subplots(2, 2, figsize=(12, 10))
            
            # Observed deficit
            im0 = axes[0, 0].contourf(self.target_x, self.target_y, observed_deficit.T)
            axes[0, 0].set_title('Observed Velocity Deficit')
            axes[0, 0].set_ylabel('y/D')
            plt.colorbar(im0, ax=axes[0, 0])
            
            # Predicted deficit
            im1 = axes[0, 1].contourf(self.target_x, self.target_y, pred_deficit)
            axes[0, 1].set_title('Predicted Velocity Deficit')
            plt.colorbar(im1, ax=axes[0, 1])
            
            # Error
            im2 = axes[1, 0].contourf(self.target_x, self.target_y, error)
            axes[1, 0].set_title(f'Error (RMSE: {rmse:.4f})')
            axes[1, 0].set_xlabel('x/D')
            axes[1, 0].set_ylabel('y/D')
            plt.colorbar(im2, ax=axes[1, 0])
            
            # Error histogram
            axes[1, 1].hist(error.values.flatten(), bins=50, density=True, alpha=0.7)
            axes[1, 1].axvline(0, color='k', linestyle='--', alpha=0.5)
            axes[1, 1].set_title('Error Distribution')
            axes[1, 1].set_xlabel('Error')
            axes[1, 1].set_ylabel('Density')
            
            # Add statistics
            error_mean = np.mean(error)
            error_p90 = np.percentile(error, 90)
            error_n90 = np.percentile(error, 10)
            axes[1, 1].text(0.05, 0.95, f'Mean: {error_mean:.4f}\nP90: {error_p90:.4f}\nP10: {error_n90:.4f}',
                          transform=axes[1, 1].transAxes, verticalalignment='top',
                          bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))
            
            plt.tight_layout()
            plt.savefig(f'{save_dir}/flow_comparison_{t:03d}.png', dpi=150, bbox_inches='tight')
            plt.close()
        
        # Calculate overall statistics
        all_errors_flat = np.concatenate(all_errors)
        overall_mean = np.mean(all_errors_flat)
        overall_p90 = np.percentile(all_errors_flat, 90)
        overall_n90 = np.percentile(all_errors_flat, 10)
        
        print(f"RMSE values per time step: {rmse_values}")
        print(f"Average RMSE: {np.mean(rmse_values):.4f}")
        print(f"Overall error mean: {overall_mean:.4f}")
        print(f"Overall error P90: {overall_p90:.4f}")
        print(f"Overall error P10: {overall_n90:.4f}")
        
        return rmse_values
