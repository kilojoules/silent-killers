    def run_optimization(self, init_points: int = 50, n_iter: int = 200) -> Dict[str, Any]:
        """
        Run Bayesian optimization.
        
        Parameters:
        -----------
        init_points : int
            Number of initial random points
        n_iter : int
            Number of optimization iterations
        
        Returns:
        --------
        Dict containing optimization results
        """

        defaults = self.get_default_params()
        
        optimizer = BayesianOptimization(
            f=self.optimization_function,
            pbounds=pbounds,
            random_state=1
        )
        
        optimizer.probe(params=defaults, lazy=True)
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        best_params = optimizer.max['params']
        best_rmse = -optimizer.max['target']
        
        self.optimizer = optimizer
        self.best_params = best_params
        self.best_rmse = best_rmse
        
        return {
            'optimizer': optimizer,
            'best_params': best_params,
            'best_rmse': best_rmse

        # Calculate detailed error statistics

        # Statistical metrics

            'rmse': float(rmse),
            'mae': float(np.abs(errors).mean(['x', 'y']).mean('time')),
            'bias': float(errors.mean(['x', 'y']).mean('time')),
            'std': float(errors.std(['x', 'y']).mean('time')),

            error_t = errors.isel(time=t)
            rmse_t = np.sqrt(np.mean(error_t**2))

        stats['mean_rmse'] = np.mean(rmse_per_step)
