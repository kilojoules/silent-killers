import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr
from bayes_opt import BayesianOptimization

# PyWake imports
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.deficit_models.utils import ct2a_mom1d
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter

from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW


class WakeModelOptimizer:
    """Main class for wake model parameter optimization."""
    
    def __init__(self, model_type=1, downwind=True, data_path='./DTU10MW.nc'):
        """
        Initialize the optimizer.
        
        Parameters:
        -----------
        model_type : int
            Model type (1 or 2)
        downwind : bool
            If True, optimize for downstream region; if False, for upstream
        data_path : str
            Path to reference data file
        """
        if model_type not in [1, 2]:
            raise ValueError("model_type must be 1 or 2")
            
        self.model_type = model_type
        self.downwind = downwind
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        
        # Load and prepare reference data
        self.load_reference_data(data_path)
        self.setup_region_of_interest()
        self.prepare_observation_data()
        
        # Initialize site
        self.site = Hornsrev1Site()
        
        # Parameter bounds and defaults
        self.setup_parameters()
        
        # Results storage
        self.optimizer = None
        self.best_params = None
        self.best_rmse = None
        
    def load_reference_data(self, data_path):
        """Load and prepare reference data."""
        self.dat = xr.load_dataset(data_path)
        self.dat = self.dat.assign_coords(x=self.dat.x * self.D, y=self.dat.y * self.D)
        
    def setup_region_of_interest(self):
        """Define the region of interest for analysis."""
        if self.downwind:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        self.X_LB = X_LB
        self.X_UB = X_UB
        
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        self.flow_roi = self.dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        
    def prepare_observation_data(self):
        """Prepare observation data for all operating conditions."""
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        # Create full grid of TI and WS combinations
        full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
        full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
        
        self.full_ti = full_ti
        self.full_ws = full_ws
        self.wd = [270] * full_ti.size
        
        # Get observed deficits for all conditions
        obs_values = []
        for t in range(full_ti.size):
            # Get a simulation result for reference CT/TI values
            sim_res = All2AllIterative(
                self.site, self.turbine,
                wake_deficitModel=BlondelSuperGaussianDeficit2020(),
                superpositionModel=LinearSum(),
                deflectionModel=None,
                turbulenceModel=CrespoHernandez(),
                blockage_deficitModel=SelfSimilarityDeficit2020()
            )([0], [0], ws=full_ws[t], TI=full_ti[t], wd=270, time=True)
            
            flow_map = sim_res.flow_map(HorizontalGrid(x=self.target_x, y=self.target_y))
            this_pred_sim = sim_res.isel(time=0, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            obs_values.append(observed_deficit.T)
        
        self.all_obs = xr.concat(obs_values, dim='time')
        
    def setup_parameters(self):
        """Setup parameter bounds and defaults based on model and region."""
        if self.model_type == 1:
            if self.downwind:
                self.pbounds = {
                    'a_s': (0.001, 0.5),
                    'b_s': (0.001, 0.01),
                    'c_s': (0.001, 0.5),
                    'b_f': (-2, 1),
                    'c_f': (0.1, 5),
                    'ch1': (-1, 2),
                    'ch2': (-1, 2),
                    'ch3': (-1, 2),
                    'ch4': (-1, 2),
                }
                self.defaults = {
                    'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2,
                    'b_f': -0.68, 'c_f': 2.41,
                    'ch1': 0.73, 'ch2': 0.8325,
                    'ch3': -0.0325, 'ch4': -0.32
                }
            else:
                self.pbounds = {
                    'ss_alpha': (0.05, 3),
                    'ss_beta': (0.05, 3),
                    'rp1': (-2, 2),
                    'rp2': (-2, 2),
                    'ng1': (-3, 3),
                    'ng2': (-3, 3),
                    'ng3': (-3, 3),
                    'ng4': (-3, 3),
                    'fg1': (-2, 2),
                    'fg2': (-2, 2),
                    'fg3': (-2, 2),
                    'fg4': (-2, 2)
                }
                self.defaults = {
                    'ss_alpha': 0.8888888888888888,
                    'ss_beta': 1.4142135623730951,
                    'rp1': -0.672, 'rp2': 0.4897,
                    'ng1': -1.381, 'ng2': 2.627,
                    'ng3': -1.524, 'ng4': 1.336,
                    'fg1': -0.06489, 'fg2': 0.4911,
                    'fg3': 1.116, 'fg4': -0.1577
                }
        else:  # MODEL == 2
            self.defaults = {
                'A': 0.04,
                'cti1': 1.5, 'cti2': 0.8,
                'ceps': 0.25, 'ctlim': 0.999,
                'ch1': 0.73, 'ch2': 0.8325,
                'ch3': -0.0325, 'ch4': -0.3
            }
            self.pbounds = {
                'A': (0.001, 0.5),
                'cti1': (0.01, 5), 'cti2': (0.01, 5),
                'ceps': (0.01, 3), 'ctlim': (0.01, 1),
                'ch1': (-1, 2), 'ch2': (-1, 2),
                'ch3': (-1, 2), 'ch4': (-1, 2),
            }
    
    def create_wake_model(self, params):
        """
        Create a wind farm model with given parameters.
        
        Returns:
        --------
        wfm : WindFarmModel
            Configured wind farm model
        """
        # Extract turbulence parameters
        turb_params = {'c': np.array([params.get('ch1', 0.73),
                                      params.get('ch2', 0.8325),
                                      params.get('ch3', -0.0325),
                                      params.get('ch4', -0.32)])}
        
        if self.downwind:
            if self.model_type == 1:
                # BlondelSuperGaussian model
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
                blockage_args = {}
            else:  # MODEL == 2
                # TurboGaussian model
                wake_deficit_model = TurboGaussianDeficit(
                    A=params['A'],
                    cTI=[params['cti1'], params['cti2']],
                    ctlim=params['ctlim'],
                    ceps=params['ceps'],
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficit_model.WS_key = 'WS_jlk'
                blockage_args = {}
        else:  # Upstream
            wake_deficit_model = BlondelSuperGaussianDeficit2020()
            blockage_args = {
                'ss_alpha': params['ss_alpha'],
                'ss_beta': params['ss_beta'],
                'r12p': np.array([params['rp1'], params['rp2']]),
                'ngp': np.array([params['ng1'], params['ng2'], 
                                params['ng3'], params['ng4']])
            }
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
        
        # Create wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficit_model,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_params),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def evaluate_rmse(self, **params):
        """Evaluate RMSE for given parameters."""
        try:
            # Create model with current parameters
            wfm = self.create_wake_model(params)
            
            # Run simulation for all conditions
            sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                         wd=self.wd, time=True)
            
            # Compute flow maps for all time steps
            flow_maps = []
            for t in range(len(self.full_ws)):
                fm = sim_res.flow_map(
                    HorizontalGrid(x=self.target_x, y=self.target_y),
                    time=[t]
                )['WS_eff']
                flow_maps.append(fm)
            
            flow_map = xr.concat(flow_maps, dim='time')
            
            # Compute predicted deficits
            pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
            
            # Compute RMSE
            rmse = float(np.sqrt(((self.all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
            
            return -rmse if not np.isnan(rmse) else -0.5
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        self.optimizer = BayesianOptimization(
            f=self.evaluate_rmse,
            pbounds=self.pbounds,
            random_state=1
        )
        
        # Probe default parameters
        self.optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        return self.best_params, self.best_rmse
    
    def create_animation(self, output_path):
        """Create animation of optimization progress."""
        if self.optimizer is None:
            raise ValueError("Run optimization first")
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Get best parameters up to current frame
            best_so_far_params = {}
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                if -self.optimizer.space.target[i] <= best_so_far_rmse:
                    best_so_far_rmse = -self.optimizer.space.target[i]
                    best_so_far_params = self.optimizer.res[i]['params']
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot RMSE history
            ax1.plot(-np.array(self.optimizer.space.target), color='gray', alpha=0.5)
            ax1.plot(np.array(best_so_far_rmses), color='black')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Plot parameter comparison
            keys = list(best_so_far_params.keys())
            best_vals = [best_so_far_params[k] for k in keys]
            default_vals = [self.defaults.get(k, 0) for k in keys]
            
            ax2.bar(keys, best_vals, label='Optimized')
            ax2.bar(keys, default_vals, edgecolor='black', linewidth=2,
                   color='none', capstyle='butt', label='Default')
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.tick_params(axis='x', rotation=45)
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        # Create animation
        ani = animation.FuncAnimation(
            fig, update_plot,
            frames=len(self.optimizer.space.target),
            repeat=False
        )
        
        # Save animation
        writer = animation.FFMpegWriter(fps=15)
        ani.save(output_path, writer=writer)
        plt.close(fig)
    
    def analyze_results(self):
        """Analyze and visualize optimization results."""
        if self.best_params is None:
            raise ValueError("Run optimization first")
        
        # Create model with best parameters
        wfm_best = self.create_wake_model(self.best_params)
        
        # Run simulation
        sim_res = wfm_best([0], [0], ws=self.full_ws, TI=self.full_ti,
                          wd=self.wd, time=True)
        
        # Compute flow maps
        flow_maps = []
        for t in range(len(self.full_ws)):
            fm = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y),
                time=[t]
            )['WS_eff']
            flow_maps.append(fm)
        
        flow_map = xr.concat(flow_maps, dim='time')
        
        # Analyze each time step
        rmse_values = []
        error_stats = []
        
        for t in range(flow_map.time.size):
            this_pred_sim = sim_res.isel(time=t)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            ).isel(wt=0)
            
            pred = (this_pred_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_pred_sim.WS
            diff = observed_deficit.T - pred
            
            # Compute RMSE
            rmse = np.sqrt(np.mean(diff**2))
            rmse_values.append(rmse)
            
            # Compute error statistics
            abs_errors = np.abs(diff.values.flatten())
            error_stats.append({
                'time': t,
                'rmse': rmse,
                'mean_error': np.mean(abs_errors),
                'p90_error': np.percentile(abs_errors, 90),
                'max_error': np.max(abs_errors)
            })
            
            # Create flow field plots with error statistics
            self.plot_flow_field_comparison(
                t, observed_deficit.T, pred, diff,
                error_stats[-1], save=True
            )
        
        # Print statistics
        error_df = pd.DataFrame(error_stats)
        print("\n=== Error Statistics ===")
        print(f"Overall RMSE: {np.mean(rmse_values):.4f}")
        print(f"\nAverage error statistics across all cases:")
        print(f"  Mean absolute error: {error_df['mean_error'].mean():.4f}")
        print(f"  90th percentile error: {error_df['p90_error'].mean():.4f}")
        print(f"  Maximum error: {error_df['max_error'].max():.4f}")
        
        # Plot parameter comparison
        self.plot_parameter_comparison()
        
        return error_df
    
    def plot_flow_field_comparison(self, time_idx, observed, predicted, diff, 
                                  stats, save=False):
        """Plot flow field comparison with error statistics."""
        fig, axes = plt.subplots(3, 1, figsize=(10, 12))
        
        # Observed
        im0 = axes[0].contourf(self.target_x, self.target_y, observed)
        axes[0].set_title(f'Observed Deficits (Time {time_idx})')
        axes[0].set_ylabel('y/D')
        fig.colorbar(im0, ax=axes[0])
        
        # Predicted
        im1 = axes[1].contourf(self.target_x, self.target_y, predicted)
        axes[1].set_title('Predicted Deficits')
        axes[1].set_ylabel('y/D')
        fig.colorbar(im1, ax=axes[1])
        
        # Difference with error statistics
        im2 = axes[2].contourf(self.target_x, self.target_y, diff, cmap='RdBu_r')
        axes[2].set_title('Difference (Observed - Predicted)')
        axes[2].set_xlabel('x/D')
        axes[2].set_ylabel('y/D')
        
        # Add error statistics to plot
        stats_text = (f'RMSE: {stats["rmse"]:.4f}\n'
                     f'Mean Abs Error: {stats["mean_error"]:.4f}\n'
                     f'P90 Abs Error: {stats["p90_error"]:.4f}')
        axes[2].text(0.02, 0.98, stats_text, transform=axes[2].transAxes,
                    verticalalignment='top', bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.8))
        
        fig.colorbar(im2, ax=axes[2])
        plt.tight_layout()
        
        if save:
            plt.savefig(f'figs/flow_field_comparison_t{time_idx}.png', dpi=150)
            plt.close(fig)
        else:
            plt.show()
    
    def plot_parameter_comparison(self):
        """Plot comparison between optimized and default parameters."""
        if self.best_params is None:
            raise ValueError("No optimized parameters available")
        
        keys = list(self.best_params.keys())
        best_vals = [self.best_params[k] for k in keys]
        default_vals = [self.defaults.get(k, 0) for k in keys]
        
        fig, ax = plt.subplots(figsize=(10, 6))
        x_pos = np.arange(len(keys))
        
        width = 0.35
        ax.bar(x_pos - width/2, default_vals, width, label='Default', alpha=0.8)
        ax.bar(x_pos + width/2, best_vals, width, label='Optimized', alpha=0.8)
        
        ax.set_xlabel('Parameters')
        ax.set_ylabel('Value')
        ax.set_title(f'Parameter Comparison (RMSE: {self.best_rmse:.4f})')
        ax.set_xticks(x_pos)
        ax.set_xticklabels(keys, rotation=45, ha='right')
        ax.legend()
        ax.grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.savefig(f'parameter_comparison_model{self.model_type}_'
                   f'downwind{self.downwind}.png', dpi=150)
        plt.close()
