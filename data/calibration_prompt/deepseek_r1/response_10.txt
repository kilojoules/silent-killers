        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)

        full_ws = np.array([[self.WSs[ii]] * self.TIs.size for ii in range(self.WSs.size)]).flatten()
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        
        # Load observed deficits
        self.prepare_observed_deficits()
    
    def prepare_observed_deficits(self):
        """Prepare observed deficit values for all conditions."""
        # Create a baseline simulation to get CT values for each condition
        baseline_wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),

        )
        
        baseline_res = baseline_wfm(

            TI=self.full_ti,

            time=True
        )

        obs_values = []

            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_sim.CT, 
                ti=this_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)

    def create_wfm(self, params):
        """Create a WindFarmModel with given parameters."""
        
        # Setup wake deficit model
        if self.downwind:
            if self.model_type == 1:
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
                turb_args = {'c': np.array([params['ch1'], params['ch2'],

                blockage_args = {}
        else:  # upwind
            def_args = {}
            turb_args = {}
            wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
            blockage_args = {
                'ss_alpha': params['ss_alpha'],
                'ss_beta': params['ss_beta'],
                'r12p': np.array([params['rp1'], params['rp2']]),
                'ngp': np.array([params['ng1'], params['ng2'],

        wfm = All2AllIterative(

        return wfm
    
    def compute_velocity_deficit(self, sim_res, flow_map):
        """Compute velocity deficit from simulation results."""
        # Note: flow_map.WS_eff is the velocity, reference data has velocity deficits
        # Velocity deficit = (U_inf - U_wake) / U_inf
        # Since flow_map.WS_eff is U_wake and sim_res.WS is U_inf
        pred_deficit = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        return pred_deficit
    
    def evaluate_rmse(self, **params):
        """Evaluate RMSE for given parameters."""
        try:
            # Create WFM with current parameters
            wfm = self.create_wfm(params)
            
            # Run simulation
            sim_res = wfm(
                [0], [0], 
                ws=self.full_ws, 
                TI=self.full_ti, 
                wd=[270] * self.full_ti.size, 
                time=True
            )
            
            # Compute flow map for all time steps
            flow_maps = []
            for t in range(self.full_ws.size):
                fm = sim_res.flow_map(
                    HorizontalGrid(x=self.target_x, y=self.target_y),
                    time=[t]
                )['WS_eff']
                flow_maps.append(fm)
            
            flow_map = xr.concat(flow_maps, dim='time')
            
            # Compute predicted deficits
            pred_deficit = self.compute_velocity_deficit(sim_res, flow_map)
            
            # Calculate RMSE
            rmse = float(np.sqrt(((self.all_obs - pred_deficit) ** 2)
                                 .mean(['x', 'y']))
                                 .mean('time'))
            
            if np.isnan(rmse):
                return -0.5
            return -rmse  # Negative for minimization
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization."""
        self.optimizer = BayesianOptimization(
            f=self.evaluate_rmse, 
            pbounds=self.pbounds, 
            random_state=1
        )
        
        # Start with default parameters
        self.optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        print("Starting Bayesian optimization...")
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Store best results
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        print(f"Optimization complete. Best RMSE: {self.best_rmse:.4f}")
        
        return self.best_params, self.best_rmse
