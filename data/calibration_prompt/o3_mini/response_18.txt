#!/usr/bin/env python
"""
A modularized version of the turbine wake optimization script.
This version separates:
  • Data loading and ROI setup
  • Creation of the wake model based on the chosen deficit scheme (MODEL)
  • Evaluation of the RMSE objective (using observational deficit and predicted flow field)
  • Bayesian Optimization and plotting of results

Make sure to keep your dependency libraries installed.
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr

# Import wake and turbine classes
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from bayes_opt import BayesianOptimization
from py_wake.deficit_models.utils import ct2a_mom1d


# Global settings
DOWNWIND = True
MODEL = 2  # either 1 or 2. Error out if not valid.
if MODEL not in (1, 2):
    raise Exception("Bad Model Number")

# Data loading and ROI construction
def load_setup():
    # load the data and turbine
    dat = xr.load_dataset('./DTU10MW.nc')
    turbine = DTU10MW()
    D = turbine.diameter()
    dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
    
    # setup downwind/upstream region-of-interest boundaries
    if DOWNWIND:
        X_LB, X_UB = 2, 10
    else:
        X_LB, X_UB = -2, -1

    roi_x = slice(X_LB * D, X_UB * D)
    roi_y = slice(-2 * D, 2 * D)
    flow_roi = dat.sel(x=roi_x, y=roi_y)
    target_x, target_y = flow_roi.x, flow_roi.y

    # make a grid of inflow conditions 
    TIs = np.arange(0.05, 0.45, 0.05)
    WSs = np.arange(4, 11)
    full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
    full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
    assert full_ws.size == full_ti.size

    # get a site instance (Hornrev1 for now)
    site = Hornsrev1Site()

    return turbine, D, flow_roi, target_x, target_y, full_ws, full_ti, site, X_LB, X_UB

    
# Create the simulation result from a given wake model (wfm)
def run_simulation(wfm, target_x, target_y, full_ws, full_ti):
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    # build flow_map: note, we iterate through the time dimension for a proper xarray concat
    flow_map = None
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        if flow_map is None:
            flow_map = fm
        else:
            flow_map = xr.concat([flow_map, fm], dim='time')
    return sim_res, flow_map


# Objective function for Bayesian optimization
def evaluate_rmse(**kwargs):
    global DOWNWIND, MODEL, site, turbine, full_ws, full_ti, target_x, target_y, flow_roi

    # Setup the wake deficit and turbulence model based on DOWNWIND & MODEL
    if DOWNWIND:
        if MODEL == 1:
            def_args = {k: kwargs[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
            turb_args = {'c': np.array([kwargs['ch1'], kwargs['ch2'], kwargs['ch3'], kwargs['ch4']])}
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
            blockage_args = {}  # not used in this branch
        else:
            turb_args = {'c': np.array([kwargs['ch1'], kwargs['ch2'], kwargs['ch3'], kwargs['ch4']])}
            wake_deficitModel = TurboGaussianDeficit(
                A=kwargs['A'], cTI=[kwargs['cti1'], kwargs['cti2']],
                ctlim=kwargs['ctlim'], ceps=kwargs['ceps'],
                ct2a=ct2a_mom1d,
                groundModel=Mirror(),
                rotorAvgModel=GaussianOverlapAvgModel())
            wake_deficitModel.WS_key = 'WS_jlk'
            blockage_args = {}
    else:
        # For upstream the models have a different configuration
        turb_args = {}
        wake_deficitModel = BlondelSuperGaussianDeficit2020()
        blockage_args = {'ss_alpha': kwargs['ss_alpha'], 'ss_beta': kwargs['ss_beta'],
                         'r12p': np.array([kwargs['rp1'], kwargs['rp2']]),
                         'ngp': np.array([kwargs['ng1'], kwargs['ng2'], kwargs['ng3'], kwargs['ng4']])}
        if MODEL == 2:
            blockage_args['groundModel'] = Mirror()

    # Build the wake model (wfm)
    wfm = All2AllIterative(site, turbine,
                            wake_deficitModel=wake_deficitModel,
                            superpositionModel=LinearSum(),
                            deflectionModel=None,
                            turbulenceModel=CrespoHernandez(**turb_args),
                            blockage_deficitModel=SelfSimilarityDeficit2020(groundModel=Mirror()))
    # Run simulation (get predicted flow_map)
    sim_res, flow_map = run_simulation(wfm, target_x, target_y, full_ws, full_ti)
    # Compute relative error field (or deficit error)
    pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS

    # Get observed deficits: note interpolation to simulation CT/TI conditions, select wt=0
    obs_values = []
    for t in range(sim_res.time.size):
        this_sim = sim_res.isel(time=t, wt=0)
        # Interpolating the observed deficit (you might want to use average and p90 stats here)
        observed = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0)
        obs_values.append(observed.T)
    all_obs = xr.concat(obs_values, dim='time')
    
    # Calculate RMSE over the spatial domain, averaged over time
    rmse = float(np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
    if np.isnan(rmse):
        return -0.5
    return -rmse    # Note: the optimizer maximizes, so we return negative RMSE

        blockage_args = {'ss_alpha': best_params['ss_alpha'], 'ss_beta': best_params['ss_beta'],
                         'r12p': np.array([best_params['rp1'], best_params['rp2']]),

    return wfm


# Animation update function for showing convergence and parameter evolution
def update_plot(frame, optimizer, defaults, ax1, ax2):
    ax1.clear()
    ax2.clear()
    best_so_far_params = {}
    best_so_far_rmse = float('inf')
    best_so_far_rmses = []
    
    # Iterate through optimization history up to the current frame
    for i in range(frame + 1):
        current_rmse = -optimizer.space.target[i]  # note sign flip
        if current_rmse <= best_so_far_rmse:
            best_so_far_rmse = current_rmse
            best_so_far_params = optimizer.res[i]['params']
        best_so_far_rmses.append(best_so_far_rmse)
    
    # Plot full history and best so far RMSE
    ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5)
    ax1.plot(np.array(best_so_far_rmses), color='black')
    ax1.set_title('Optimization Convergence')
    ax1.set_xlabel('Iteration')
    ax1.set_ylabel('RMSE')
    ax1.grid(True)
    
    # Bar plot of parameter values (both optimized and defaults)
    keys = list(best_so_far_params.keys())
    best_vals = [best_so_far_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    ax2.bar(keys, best_vals, label='Optimized')
    ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, color='none', label='Default')
    ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
    ax2.tick_params(axis='x', rotation=45)
    ax2.legend()
    plt.tight_layout()
    return ax1, ax2
