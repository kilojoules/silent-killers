Below is one way to “rewrite” your code in a modular style. In the version below I’ve split the tasks into dedicated functions for loading data, setting up the region‐of-interest (ROI), building the wake farm model (wfm) and simulation results based on the MODEL and UPSTREAM (DOWNWIND) flags, defining the RMSE evaluation (which is used in BayesianOptimization), running the optimization and plotting routines (including convergence animation and post‐processing plots). You’ll find that all of the deficit settings (the various parameter dictionaries) as well as the MODEL and UPSTREAM logic are maintained. Feel free to adjust names or modularity as desired.

#!/usr/bin/env python
"""
Modular version of the wake modeling / optimization code.

This script:
  • Loads the reference data and sets up the ROI.
  • Provides functions to instantiate the wake farm model (wfm) according 
    to the MODEL & UPSTREAM (DOWNWIND) flags.
  • Implements the RMSE evaluation used by BayesianOptimization.
  • Runs the optimization.
  • Plots the optimization progress (including a convergence animation).
  • Generates flow field comparison plots (prediction error, observed deficit, etc).

Make sure to have your working directory set so that the './DTU10MW.nc' file
and the results folder (‘figs’) are available.
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr

from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from py_wake.deficit_models.utils import ct2a_mom1d

from bayes_opt import BayesianOptimization

##############################
# Global flags and settings
##############################
DOWNWIND = True
MODEL = 2  # Use 1 or 2
if MODEL not in {1, 2}:
    raise Exception("Bad Model Number")

# Data file and region bounds are defined here; 
# these can be adjusted as needed.
D_FILE = './DTU10MW.nc'


def load_and_prepare_data():
    """
    Loads the DTU10MW reference dataset and converts the coordinates from rotor diameters.
    """
    dat = xr.load_dataset(D_FILE)
    turbine = DTU10MW()
    D = turbine.diameter()
    dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)

    # Define region-of-interest in downwind and cross-stream directions
    if DOWNWIND:
        X_LB = 2
        X_UB = 10
    else:
        X_LB = -2
        X_UB = -1

    roi_x = slice(X_LB * D, X_UB * D)
    roi_y = slice(-2 * D, 2 * D)
    flow_roi = dat.sel(x=roi_x, y=roi_y)

    target_x = flow_roi.x
    target_y = flow_roi.y

    return turbine, D, flow_roi, target_x, target_y, X_LB, X_UB


def setup_simulation_params(D):
    """
    Sets up the wind speed, turbulence intensity and other parameter arrays.
    """
    TIs = np.arange(0.05, 0.45, 0.05)
    WSs = np.arange(4, 11)
    full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
    full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()

    assert full_ws.size == full_ti.size
    return full_ws, full_ti


def get_site():
    """Return the Hornsrev1 site (or other Site instance if needed)."""
    return Hornsrev1Site()


def initial_simulation(turbine, target_x, target_y, full_ws, full_ti):
    """
    Run an initial simulation with All2AllIterative using default settings
    to generate reference observed deficits.
    """
    site = get_site()

    # Only a small simulation is run here for initial reference
    sim_res = All2AllIterative(site, turbine,
                               wake_deficitModel=BlondelSuperGaussianDeficit2020(),
                               superpositionModel=LinearSum(),
                               deflectionModel=None,
                               turbulenceModel=CrespoHernandez(),
                               blockage_deficitModel=SelfSimilarityDeficit2020())(
                                   [0], [0],
                                   ws=full_ws,
                                   TI=full_ti,
                                   wd=[270] * full_ti.size,
                                   time=True)

    flow_map = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y))
    return sim_res, flow_map


def build_wfm(params, turbine, full_ws, full_ti, target_x, target_y):
    """
    Build the wake farm model (wfm) based on the input parameters, MODEL and DOWNWIND flags.
    
    For MODEL==1 (or MODEL==2) the appropriate deficit model is instantiated.
    """
    site = get_site()

    if DOWNWIND:
        if MODEL == 1:
            # Extract required parameters for Blondel model (MODEL 1)
            def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
            blockage_args = {}
        else:
            # MODEL 2: TurboGaussianDeficit with specific WS_key, plus turbulence settings.
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = TurboGaussianDeficit(A=params['A'], cTI=[params['cti1'], params['cti2']],
                                                    ctlim=params['ctlim'], ceps=params['ceps'],
                                                    ct2a=ct2a_mom1d,
                                                    groundModel=Mirror(),
                                                    rotorAvgModel=GaussianOverlapAvgModel())
            wake_deficitModel.WS_key = 'WS_jlk'
            blockage_args = {}
    else:
        # When UPSTREAM
        turb_args = {}
        blockage_args = {'ss_alpha': params['ss_alpha'], 'ss_beta': params['ss_beta'],
                         'r12p': np.array([params['rp1'], params['rp2']]),
                         'ngp': np.array([params['ng1'], params['ng2'], params['ng3'], params['ng4']])}
        wake_deficitModel = BlondelSuperGaussianDeficit2020()
        if MODEL == 2:
            blockage_args['groundModel'] = Mirror()

    wfm = All2AllIterative(site, turbine,
                           wake_deficitModel=wake_deficitModel,
                           superpositionModel=LinearSum(),
                           deflectionModel=None,
                           turbulenceModel=CrespoHernandez(**turb_args),
                           blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args))
    return wfm


def evaluate_rmse(**params):
    """
    Evaluation function to be optimized.
    
    For the given parameters the wake farm model (wfm) is built and simulated.
    Then the RMSE between predicted (error) and the reference observations is computed.
    """
    turbine, D, flow_roi, target_x, target_y, _, _ = load_and_prepare_data()
    full_ws, full_ti = setup_simulation_params(D)
    site = get_site()

    # Run simulation with default settings so that we have reference observed deficits.
    sim_res_ref, _ = initial_simulation(turbine, target_x, target_y, full_ws, full_ti)

    # Concatenate the observed deficits across time steps.
    obs_values = []
    for t in range(sim_res_ref.time.size):
        this_sim = sim_res_ref.isel(time=t, wt=0)
        observed_deficit = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0)
        obs_values.append(observed_deficit.T)
    all_obs = xr.concat(obs_values, dim='time')

    # Build new wfm according to the current parameters
    wfm = build_wfm(params, turbine, full_ws, full_ti, target_x, target_y)

    # Run simulation (note: time simulation)
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)

    # Rebuild the flow field; note that here we compute the flow map in a time loop
    flow_map = None
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        if flow_map is None:
            flow_map = fm
        else:
            flow_map = xr.concat([flow_map, fm], dim='time')

    pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS

    # Compute RMSE difference across x and y then average over time.
    rmse = float(np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
    if np.isnan(rmse):
        return -0.5
    return -rmse


def run_optimization(defaults, pbounds):
    """
    Run Bayesian optimization in order to minimize the RMSE (maximize -RMSE).
    """
    optimizer = BayesianOptimization(f=evaluate_rmse, pbounds=pbounds, random_state=1)
    # Probe default parameters; use lazy evaluation.
    optimizer.probe(params=defaults, lazy=True)
    optimizer.maximize(init_points=50, n_iter=200)
    best_params = optimizer.max['params']
    best_rmse = -optimizer.max['target']  # best observed RMSE
    return optimizer, best_params, best_rmse


def plot_convergence_animation(optimizer, defaults, save_name):
    """
    Create an animated convergence plot plus parameter bar chart.
    """
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))

    def update_plot(frame):
        ax1.clear()
        ax2.clear()
        best_so_far_params = {}
        best_so_far_rmse = float('inf')
        best_so_far_rmses = []
        for i in range(frame + 1):
            # optimizer.space.target[i] is negative RMSE
            if -optimizer.space.target[i] <= best_so_far_rmse:
                best_so_far_rmse = -optimizer.space.target[i]
                best_so_far_params = optimizer.res[i]['params']
            best_so_far_rmses.append(best_so_far_rmse)
    
        # Convergence plot: full trace in gray and best so far in black
        ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5)
        ax1.plot(np.array(best_so_far_rmses), color='black')
        ax1.set_title('Optimization Convergence')
        ax1.set_xlabel('Iteration')
        ax1.set_ylabel('RMSE')
        ax1.grid(True)
    
        # Bar plot with best parameters vs. defaults
        keys = list(best_so_far_params.keys())
        best_vals = [best_so_far_params[k] for k in keys]
        default_vals = [defaults[k] for k in keys]
        ax2.bar(keys, best_vals, label='Optimized')
        ax2.bar(keys, default_vals, edgecolor='black', linewidth=2,
                color='none', capstyle='butt', label='Default')
        ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
        ax2.tick_params(axis='x', rotation=45)
        ax2.legend()
        plt.tight_layout()
        return ax1, ax2

    ani = animation.FuncAnimation(fig, update_plot, frames=len(optimizer.space.target), repeat=False)
    writer = animation.FFMpegWriter(fps=15)
    ani.save(save_name, writer=writer)
    plt.close('all')


def generate_post_optimization_plots(best_params, defaults, turbine, full_ws, full_ti, target_x, target_y, flow_map, prefix):
    """
    Use best parameters to simulate and compare predicted vs observed deficits.
    Also generate bar chart comparing best and default parameters.
    """
    # Build model with best parameters
    wfm = build_wfm(best_params, turbine, full_ws, full_ti, target_x, target_y)
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)

    rmse_values = []
    for t in range(flow_map.time.size):
        this_pred_sim = sim_res.isel(time=t)
        observed_deficit = flow_roi.deficits.interp(ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0).isel(wt=0)
        pred = (this_pred_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_pred_sim.WS
        diff = observed_deficit.T - pred
        rmse = np.sqrt(np.mean(diff**2))
        rmse_values.append(rmse)

        fig, ax = plt.subplots(3, 1, figsize=(5, 15))
        co = ax[0].contourf(target_x, target_y, observed_deficit.T)
        cp = ax[1].contourf(target_x, target_y, pred)
        cd = ax[2].contourf(target_x, target_y, diff)
        for jj, c in enumerate([co, cp, cd]):
            fig.colorbar(c, ax=ax[jj])
        ax[0].set_ylabel('Observed')
        ax[1].set_ylabel('Prediction')
        ax[2].set_ylabel('Diff')
        plt.tight_layout()
        plt.savefig(f'{prefix}_downsream_err_{t}.png')
        plt.clf()

    overall_rmse = np.mean(rmse_values)
    print(f"RMSE values per time step: {rmse_values}")
    print(f"Overall RMSE: {overall_rmse}")

    # Bar plot to compare best parameters and defaults:
    keys = list(best_params.keys())
    best_vals = [best_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]

    plt.bar(keys, best_vals)
    plt.bar(keys, default_vals, edgecolor='black', linewidth=2,
            color='none', capstyle='butt')
    plt.title('Optimal RMSE: %.4f' % overall_rmse)
    plt.tight_layout()
    plt.savefig(f'{prefix}_bar.png')
    plt.clf()

Explanation:

1. load_and_prepare_data() reads the netCDF file, scales the x and y coordinates using turbine diameter, and selects an ROI based on the DOWNWIND flag.
2. setup_simulation_params() creates the arrays for wind speeds (WS) and turbulence intensities (TI).
3. get_site() returns the Hornsrev1Site.
4. initial_simulation() runs an initial simulation to extract the “observed” deficits.
5. build_wfm() encapsulates constructing the wake farm model with the MODEL and DOWNWIND logic.
6. evaluate_rmse() is the function passed to BayesianOptimization so that for given parameter sets it rebuilds the wfm, runs the simulation, computes the predicted deficits and RMSE.
7. run_optimization() is a thin wrapper to run BayesianOptimization.
8. plot_convergence_animation() creates and saves an animation of the optimization convergence.
9. generate_post_optimization_plots() runs a final simulation with the optimized parameters and generates the contour plots comparing observed vs prediction and a bar plot summary.
