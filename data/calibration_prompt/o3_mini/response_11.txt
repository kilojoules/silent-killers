Below is one way to rewrite your code into a more modular style. In the example below I’ve split things into functions that (a) build the wake‐farm model (wfm) given a set of parameters while preserving your deficit settings, MODEL and DOWNWIND logic, (b) run the simulation and extract flow maps, (c) evaluate the RMSE (with hooks for later adding extra error statistics such as the average and p90 error) and (d) update the optimization plots. (You may later change the flow field plotting section so that instead of plotting the full contours you compute and show the mean and 90th-percentile differences.)  

You can adjust function names, folder names and plotting details as needed. Note that this rewrite preserves your overall logic – the same pbounds, defaults, upstream/downwind logic, and the use of the models/settings you had.

#!/usr/bin/env python3
"""
Modularized wake optimization, simulation and reporting of flow field error metrics.
The code uses the following main functions:
  • build_wfm: create a wind farm model based on the MODEL and DOWNWIND settings.
  • run_simulation: runs a simulation for a given wfm, returning the simulated flow map.
  • evaluate_rmse: given the candidate parameters from the optimizer, compute RMSE error.
  • update_plot: update the convergence and parameter bar plot.
  • plot_flow_fields: (new version) plot flow field error metrics (mean and p90 of error) per time step.
All deficit settings, MODEL and UPSTREAM logic are kept.
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr

from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.deficit_models.gaussian import SelfSimilarityDeficit2020
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake.deficit_models.utils import ct2a_mom1d

from bayes_opt import BayesianOptimization

# ----------------------------------------------------------------------------
# Global simulation settings

DOWNWIND = True
MODEL = 2  # allowed values 1 and 2
if MODEL not in {1, 2}:
    raise Exception("Bad Model Number")

# Load turbine and dataset, scale coordinates as before.
dat = xr.load_dataset('./DTU10MW.nc')
turbine = DTU10MW()
D = turbine.diameter()
dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)

# Define region-of-interest based on DOWNWIND or not
if DOWNWIND:
    X_LB, X_UB = 2, 10
else:
    X_LB, X_UB = -2, -1

roi_x = slice(X_LB * D, X_UB * D)
roi_y = slice(-2 * D, 2 * D)
flow_roi = dat.sel(x=roi_x, y=roi_y)
target_x = flow_roi.x
target_y = flow_roi.y

# Setup parameter sweep for TI and WS used in simulation.
TIs = np.arange(0.05, 0.45, 0.05)
WSs = np.arange(4, 11)
full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
assert full_ws.size == full_ti.size

# Site object:
site = Hornsrev1Site()

# Precompute observed deficits from the data (for all time steps) using the DTU10MW.nc
obs_values = []
sim_res_initial = All2AllIterative(site, turbine,
                                   wake_deficitModel=BlondelSuperGaussianDeficit2020(),
                                   superpositionModel=LinearSum(),
                                   deflectionModel=None,
                                   turbulenceModel=CrespoHernandez(),
                                   blockage_deficitModel=SelfSimilarityDeficit2020())([0], [0],
                                                                                    ws=full_ws, TI=full_ti,
                                                                                    wd=[270] * full_ti.size,
                                                                                    time=True)
flow_map_initial = sim_res_initial.flow_map(HorizontalGrid(x=target_x, y=target_y))
for t in range(flow_map_initial.time.size):
    this_pred_sim = sim_res_initial.isel(time=t, wt=0)
    # Interpolate using CT and TI from the simulation to get the observed deficits.
    observed_deficit = flow_roi.deficits.interp(ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0)
    obs_values.append(observed_deficit.T)
all_obs = xr.concat(obs_values, dim='time')

# ----------------------------------------------------------------------------
# Function to instantiate the wind farm model (wfm) based on candidate parameters.
def build_wfm(params):
    """
    Build and return the All2AllIterative wind farm model (wfm) given candidate `params`.
    This function wraps the different settings for MODEL 1 or 2 and DOWNWIND/Upstream.
    """
    # Based on DOWNWIND flag and MODEL, define deficit, turbulence, and blockage arguments.
    if DOWNWIND:
        if MODEL == 1:
            def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
            blockage_args = {}
        else:
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = TurboGaussianDeficit(A=params['A'],
                                                     cTI=[params['cti1'], params['cti2']],
                                                     ctlim=params['ctlim'],
                                                     ceps=params['ceps'],
                                                     ct2a=ct2a_mom1d,
                                                     groundModel=Mirror(),
                                                     rotorAvgModel=GaussianOverlapAvgModel())
            wake_deficitModel.WS_key = 'WS_jlk'
            blockage_args = {}
    else:
        turb_args = {}
        blockage_args = {'ss_alpha': params['ss_alpha'],
                         'ss_beta': params['ss_beta'],
                         'r12p': np.array([params['rp1'], params['rp2']]),
                         'ngp': np.array([params['ng1'], params['ng2'], params['ng3'], params['ng4']])}
        wake_deficitModel = BlondelSuperGaussianDeficit2020()
        if MODEL == 2:
            blockage_args['groundModel'] = Mirror()

    # Instantiate the wake farm model.
    wfm = All2AllIterative(site, turbine,
                           wake_deficitModel=wake_deficitModel,
                           superpositionModel=LinearSum(),
                           deflectionModel=None,
                           turbulenceModel=CrespoHernandez(**turb_args),
                           blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args))
    return wfm

# ----------------------------------------------------------------------------
# Function to run the simulation using a given wind farm model.
def run_simulation(wfm):
    """
    Run a simulation for input wfm over all WS and TI conditions.
    Returns the simulation result and a full flow_map constructed over time steps.
    """
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    # Build a flow_map over all time indices.
    flow_map = None
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        if flow_map is None:
            flow_map = fm
        else:
            flow_map = xr.concat([flow_map, fm], dim='time')
    return sim_res, flow_map

# ----------------------------------------------------------------------------
# Modified evaluate_rmse function that instantiates a wfm via build_wfm and computes RMSE.
def evaluate_rmse(**params):
    wfm = build_wfm(params)
    sim_res, flow_map = run_simulation(wfm)
    # Compute predicted deficits: (WS - WS_eff) / WS.
    pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
    rmse = float(np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
    if np.isnan(rmse):
        return -0.5
    return -rmse

# ----------------------------------------------------------------------------
# Function to update the optimization plot animation.
def update_plot(frame, optimizer, ax1, ax2, defaults):
    # Clear axes for update.
    ax1.clear()
    ax2.clear()
    best_so_far_params = {}
    best_so_far_rmse = float('inf')
    best_so_far_rmses = []
    # Iterate over each optimization iteration up to the current frame.
    for i in range(frame + 1):
        current_rmse = -optimizer.space.target[i]  # since opt minimizes negative rmse
        if current_rmse <= best_so_far_rmse:
            best_so_far_rmse = current_rmse
            best_so_far_params = optimizer.res[i]['params']
        best_so_far_rmses.append(best_so_far_rmse)

    # Plot the entire history in gray and best so far in black.
    ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5, label='Iteration RMSE')
    ax1.plot(np.array(best_so_far_rmses), color='black', label='Best so far')
    ax1.set_title('Optimization Convergence')
    ax1.set_xlabel('Iteration')
    ax1.set_ylabel('RMSE (lower is better)')
    ax1.grid(True)
    ax1.legend()

    # For parameter comparison, create side-by-side bar plots.
    keys = list(best_so_far_params.keys())
    best_vals = [best_so_far_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    ax2.bar(keys, best_vals, label='Optimized')
    ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, color='none', label='Default')
    ax2.set_title(f'Best RMSE so far: {best_so_far_rmse:.4f}')
    ax2.tick_params(axis='x', rotation=45)
    ax2.legend()
    plt.tight_layout()
    return ax1, ax2

# ----------------------------------------------------------------------------
# New function: plot_flow_fields now computes average and p90 error (90th percentile)
def plot_flow_fields(sim_res, flow_map, flow_roi, target_x, target_y, time_index, out_folder='figs'):
    """
    For a given time index, compare the flow_map (prediction) with measured deficits.
    Compute the error field (observed - prediction) and report average and p90 error.
    Plots three panels: observed, prediction, and error field (with error metrics in title).
    """
    # Extract simulation result for this time.
    this_sim = sim_res.isel(time=time_index)
    # Get observed values via interpolation:
    observed_deficit = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0).isel(wt=0)
    # Prediction: (WS - WS_eff) / WS
    pred = (this_sim.WS - flow_map.WS_eff.isel(h=0, time=time_index)) / this_sim.WS
    error_field = observed_deficit.T - pred

    # Compute metrics.
    err_avg = float(np.mean(np.abs(error_field)))
    err_p90 = float(np.percentile(np.abs(error_field), 90))

    fig, axs = plt.subplots(3, 1, figsize=(6, 15))
    c0 = axs[0].contourf(target_x, target_y, observed_deficit.T, cmap='viridis')
    fig.colorbar(c0, ax=axs[0])
    axs[0].set_title("Observed Deficit")
    axs[0].set_ylabel("y")

    c1 = axs[1].contourf(target_x, target_y, pred, cmap='viridis')
    fig.colorbar(c1, ax=axs[1])
    axs[1].set_title("Predicted Deficit")
    axs[1].set_ylabel("y")

    c2 = axs[2].contourf(target_x, target_y, error_field, cmap='bwr')
    fig.colorbar(c2, ax=axs[2])
    axs[2].set_title(f"Error Field (avg: {err_avg:.3f}, p90: {err_p90:.3f})")
    axs[2].set_ylabel("y")
    axs[2].set_xlabel("x")
    plt.tight_layout()
    plt.savefig(f'{out_folder}/downsream_err_{time_index}.png')
    plt.close(fig)

# ----------------------------------------------------------------------------
# Run Bayesian Optimization
optimizer = BayesianOptimization(f=evaluate_rmse, pbounds=pbounds, random_state=1)
optimizer.probe(params=defaults, lazy=True)
optimizer.maximize(init_points=50, n_iter=200)

best_params = optimizer.max['params']
best_rmse = -optimizer.max['target']

# ----------------------------------------------------------------------------
# Create animation of convergence.
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
ani = animation.FuncAnimation(fig, update_plot,
                              frames=len(optimizer.space.target),
                              fargs=(optimizer, ax1, ax2, defaults), repeat=False)

# Save animation as MP4.
writer = animation.FFMpegWriter(fps=15)
ani.save('optimization_animation_%i_%i.mp4' % (X_LB, X_UB), writer=writer)

# ----------------------------------------------------------------------------
# Final simulation using best parameters.
wfm_best = build_wfm(best_params)
sim_res, flow_map = run_simulation(wfm_best)

# For each time step, compute error metrics and plot flow field errors.
rmse_values = []
n_times = flow_map.time.size
for t in range(n_times):
    this_sim = sim_res.isel(time=t)
    observed_deficit = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0).isel(wt=0)
    pred = (this_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_sim.WS
    diff = observed_deficit.T - pred
    rmse = np.sqrt(np.mean(diff**2))
    rmse_values.append(rmse)

    # Plot flow field error using the new function (with average and p90 error).
    plot_flow_fields(sim_res, flow_map, flow_roi, target_x, target_y, t, out_folder='figs')

overall_rmse = np.mean(rmse_values)
print(f"RMSE values per time step: {rmse_values}")
print(f"Overall RMSE: {overall_rmse}")

# Plot final bar chart for parameters.
keys = list(best_params.keys())
best_vals = [best_params[k] for k in keys]
default_vals = [defaults[k] for k in keys]
plt.figure(figsize=(10,6))
plt.bar(keys, best_vals, label='Optimized')
plt.bar(keys, default_vals, edgecolor='black', linewidth=2,
        color='none', capsize=5, label='Default')
plt.title('Optimal RMSE: %.4f' % best_rmse)
plt.tight_layout()
plt.savefig('bar_LB_%i_UP_%i.png' % (X_LB, X_UB))
plt.clf()
