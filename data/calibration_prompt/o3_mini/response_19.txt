Below is one way to “rewrite” your code into a more modular structure. In this version the “wake‐farm model” instantiation, simulation and plotting (both the convergence animation and flow field error plots) are separated into functions. (All deficit‐setting logic, MODEL/UPSTREAM branches and default/bounds parameters are preserved.) You can adapt further as needed.

Below is the complete example:

------------------------------------------------
#!/usr/bin/env python
"""
A modular version to perform parameter optimization for wake modeling.
This version splits the code into functions:
 • load_data_and_setup – loads the wind‐farm/site data, creates the target grid.
 • instantiate_wfm – builds the wake farm model (wfm) based on input parameters,
     the MODEL flag and UPSTREAM/DOWNSTREAM choices.
 • run_simulation – runs the simulation and creates a flow map.
 • evaluate_rmse – a wrapper that (with a given set of parameters) runs a simulation,
     computes prediction error versus reference (average error and p90 value).
 • update_convergence_plot – used by the animation to report convergence.
 • plot_flow_field_errors – creates plots of error fields (average and 90th percentile).
The same default parameters, pbounds and deficit settings are kept.
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr

# PyWake imports
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.wind_farm_models import All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from py_wake.deficit_models.utils import ct2a_mom1d

from bayes_opt import BayesianOptimization

# Global flags/parameters (feel free to group these in a config object)
DOWNWIND = True
MODEL = 2  # can be 1 or 2
if MODEL not in {1, 2}:
    raise Exception("Bad MODEL Number")

def load_data_and_setup():
    """Load reference dataset and create a target horizontal grid."""
    # Load the DTU10MW dataset
    dat = xr.load_dataset('./DTU10MW.nc')
    turbine = DTU10MW()
    D = turbine.diameter()
    dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
    
    # Set region of interest
    if DOWNWIND:
        X_LB, X_UB = 2, 10
    else:
        X_LB, X_UB = -2, -1
    roi_x = slice(X_LB * D, X_UB * D)
    roi_y = slice(-2 * D, 2 * D)
    flow_roi = dat.sel(x=roi_x, y=roi_y)
    target_x = flow_roi.x
    target_y = flow_roi.y

    # define WS/TI combinations for simulation
    TIs = np.arange(0.05, 0.45, 0.05)
    WSs = np.arange(4, 11)
    full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
    full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
    assert full_ws.size == full_ti.size, "WS, TI mismatch!"

    # Site
    site = Hornsrev1Site()
    return turbine, D, flow_roi, target_x, target_y, full_ws, full_ti, site, (X_LB, X_UB)

def instantiate_wfm(params, site, turbine, full_ws, full_ti):
    """
    Instantiate and return a wake farm model (wfm) with the input parameters.
    The upstream/downstream logic and deficit settings are maintained.
    """
    # Create the wake deficit model, turbulence arguments and blockage (if needed)
    if DOWNWIND:
        if MODEL == 1:
            def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
            blockage_args = {}
        else:
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = TurboGaussianDeficit(A=params['A'],
                                                     cTI=[params['cti1'], params['cti2']],
                                                     ctlim=params['ctlim'],
                                                     ceps=params['ceps'],
                                                     ct2a=ct2a_mom1d,
                                                     groundModel=Mirror(),
                                                     rotorAvgModel=GaussianOverlapAvgModel())
            # Note: alternate key for specific WS scaling
            wake_deficitModel.WS_key = 'WS_jlk'
            blockage_args = {}
    else:
        # For upstream example: note that in this branch def_args and turb_args remain empty
        def_args = {}
        turb_args = {}
        wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
        blockage_args = {'ss_alpha': params['ss_alpha'],
                         'ss_beta': params['ss_beta'],
                         'r12p': np.array([params['rp1'], params['rp2']]),
                         'ngp': np.array([params['ng1'], params['ng2'], params['ng3'], params['ng4']])}
        if MODEL == 2:
            blockage_args['groundModel'] = Mirror()

    # Build the wake farm model (here using All2AllIterative)
    wfm = All2AllIterative(site, turbine,
                           wake_deficitModel=wake_deficitModel,
                           superpositionModel=LinearSum(),
                           deflectionModel=None,
                           turbulenceModel=CrespoHernandez(**turb_args),
                           blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
                           )
    return wfm

def run_simulation(wfm, target_x, target_y, full_ws, full_ti):
    """Runs the simulation with the given wake farm model and returns the flow map and simulation result"""
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    # Gather the flow map along the horizontal grid:
    # here we extract the effective wind speed, note that WS_eff gives velocity.
    flow_map = None
    # Loop over time steps to build a complete flow map along dimension 'time'
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        if flow_map is None:
            flow_map = fm
        else:
            flow_map = xr.concat([flow_map, fm], dim='time')
    return sim_res, flow_map

def compute_rmse(params, site, turbine, target_x, target_y, full_ws, full_ti, flow_roi):
    """
    Instantiates a wake farm model with the current params, runs the simulation, and computes error metrics.
    Returns negative RMSE as the objective for Bayesian optimization.
    """
    # Instantiate model
    wfm = instantiate_wfm(params, site, turbine, full_ws, full_ti)
    sim_res, flow_map = run_simulation(wfm, target_x, target_y, full_ws, full_ti)
    # Use simulation WS and flow_map to compute relative velocity deficit error:
    # Here, note that the simulation WS is the free-stream wind speed and WS_eff is the velocity field
    # then: prediction = (WS - WS_eff)/WS
    # In the reference, the flow_roi.deficits has velocity deficits.
    pred_all = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
    # Create reference from the flow_roi, based on interpolation (and picking wt=0)
    obs_values = []
    # Loop over time to pick matching reference deficits
    for t in range(sim_res.time.size):
        this_sim = sim_res.isel(time=t, wt=0)
        obs_deficit = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0)
        obs_values.append(obs_deficit.T)
    all_obs = xr.concat(obs_values, dim='time')
    # Compute pixelwise error
    diff = all_obs - pred_all
    # Here we compute RMSE averaged over time and grid
    rmse = float(np.sqrt(((diff) ** 2).mean(['x', 'y'])).mean('time'))
    # One may check also the average error and 90th percentile error.
    # If any nan appears, we return a very low score.
    if np.isnan(rmse):
        return -0.5
    return -rmse

# Now wrap the evaluate_rmse to be used with bayes_opt
def evaluate_rmse(**kwargs):
    # In this evaluation function, we repeat once the same reference simulation.
    # The outer code loaded the flow_roi and other info.
    global site, turbine, target_x, target_y, full_ws, full_ti, flow_roi
    return compute_rmse(kwargs, site, turbine, target_x, target_y, full_ws, full_ti, flow_roi)

# Next, update the convergence plot for the animation.
def update_convergence_plot(frame, optimizer, defaults, ax1, ax2):
    """Update the convergence plot: top left shows convergence and bottom shows
       bar plot comparison of best-parameters vs default-values."""
    ax1.clear()
    ax2.clear()
    best_so_far_params = {}
    best_so_far_rmse = float('inf')
    best_so_far_rmses = []
    # Loop over iterations up to the current frame
    for i in range(frame + 1):
        # Our target values are stored as negative RMSE, so we recover absolute values
        curr_rmse = -optimizer.space.target[i]
        if curr_rmse <= best_so_far_rmse:
            best_so_far_rmse = curr_rmse
            best_so_far_params = optimizer.res[i]['params']
        best_so_far_rmses.append(best_so_far_rmse)
    ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5, label='All Points')
    ax1.plot(np.array(best_so_far_rmses), color='black', label='Best so far')
    ax1.set_title('Optimization Convergence')
    ax1.set_xlabel('Iteration')
    ax1.set_ylabel('RMSE')
    ax1.grid(True)
    ax1.legend()

    # Bar plot of best vs default values:
    keys = list(best_so_far_params.keys())
    best_vals = [best_so_far_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    ax2.bar(keys, best_vals, label='Optimized', alpha=0.7)
    ax2.bar(keys, default_vals, edgecolor='black', linewidth=2,
            fill=False, label='Default')
    ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
    ax2.tick_params(axis='x', rotation=45)
    ax2.legend()
    plt.tight_layout()
    return ax1, ax2

def plot_flow_field_errors(sim_res, flow_map, flow_roi, target_x, target_y):
    """
    For each time step, compute and plot the difference between observed and predicted deficits.
    Also report statistics like average error and 90th percentile.
    """
    rmse_values = []
    avg_errors = []
    p90_errors = []
    nt = sim_res.time.size
    for t in range(nt):
        this_sim = sim_res.isel(time=t)
        observed = flow_roi.deficits.interp(ct=this_sim.CT, ti=this_sim.TI, z=0).isel(wt=0)
        pred = (this_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_sim.WS
        diff = observed.T - pred

        # Record error metrics from the diff field
        rmse = np.sqrt(np.mean(diff.data**2))
        avg_err = np.mean(np.abs(diff.data))
        p90_err = np.percentile(np.abs(diff.data), 90)
        rmse_values.append(rmse)
        avg_errors.append(avg_err)
        p90_errors.append(p90_err)

        # Plot the observed, prediction and diff field
        fig, axs = plt.subplots(3, 1, figsize=(5, 15))
        im0 = axs[0].contourf(target_x, target_y, observed.T)
        fig.colorbar(im0, ax=axs[0])
        axs[0].set_title(f'Observed, time {t}')
        im1 = axs[1].contourf(target_x, target_y, pred)
        fig.colorbar(im1, ax=axs[1])
        axs[1].set_title('Prediction')
        im2 = axs[2].contourf(target_x, target_y, diff)
        fig.colorbar(im2, ax=axs[2])
        axs[2].set_title(f'Difference (Avg: {avg_err:.4f} p90: {p90_err:.4f})')
        plt.tight_layout()
        plt.savefig(f'figs/downsream_err_{t}.png')
        plt.close(fig)
    overall_rmse = np.mean(rmse_values)
    print("RMSE per time step:", rmse_values)
    print("Overall RMSE:", overall_rmse)
    print("Average absolute errors:", avg_errors)
    print("P90 of absolute errors:", p90_errors)
    return overall_rmse, avg_errors, p90_errors

# =============================================================================
# Main code execution:
# =============================================================================
if __name__ == '__main__':
    # Load data and create target grid
    turbine, D, flow_roi, target_x, target_y, full_ws, full_ti, site, (X_LB, X_UB) = load_data_and_setup()
    
    # First simulation: run reference simulation used in the optimization target function
    # We simulate once using the default settings for plotting & later baseline simulation.
    if DOWNWIND:
        if MODEL == 1:
            defaults = {'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2,
                        'b_f': -0.68, 'c_f': 2.41,

            pbounds = {'a_s': (0.001, 0.5), 'b_s': (0.001, 0.01),
                       'c_s': (0.001, 0.5), 'b_f': (-2, 1),
                       'c_f': (0.1, 5),
                       'ch1': (-1, 2), 'ch2': (-1, 2), 'ch3': (-1, 2), 'ch4': (-1, 2)}

            pbounds = {'A': (0.001, 0.5),
                       'cti1': (0.01, 5), 'cti2': (0.01, 5),
                       'ceps': (0.01, 3),
                       'ctlim': (0.01, 1),
                       'ch1': (-1, 2), 'ch2': (-1, 2),
                       'ch3': (-1, 2), 'ch4': (-1, 2)}
    else:
        pbounds = {'ss_alpha': (0.05, 3), 'ss_beta': (0.05, 3),
                   'rp1': (-2, 2), 'rp2': (-2, 2),
                   'ng1': (-3, 3), 'ng2': (-3, 3),
                   'ng3': (-3, 3), 'ng4': (-3, 3),
                   'fg1': (-2, 2), 'fg2': (-2, 2),

        defaults = {'ss_alpha': 0.8888888888888888,
                    'ss_beta': 1.4142135623730951,
                    'rp1': -0.672, 'rp2': 0.4897,
                    'ng1': -1.381, 'ng2': 2.627,
                    'ng3': -1.524, 'ng4': 1.336,
                    'fg1': -0.06489, 'fg2': 0.4911,
                    'fg3': 1.116, 'fg4': -0.1577}
    
    # Setup Bayesian optimization
    optimizer = BayesianOptimization(f=evaluate_rmse, pbounds=pbounds, random_state=1)
    # Probe the defaults first (lazy evaluation)
    optimizer.probe(params=defaults, lazy=True)
    optimizer.maximize(init_points=50, n_iter=200)
    
    best_params = optimizer.max['params']
    best_rmse = -optimizer.max['target']
    
    print("Best parameters found:", best_params)
    print("Best RMSE:", best_rmse)
    
    # Create convergence animation:
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    ani = animation.FuncAnimation(fig, lambda f: update_convergence_plot(f, optimizer, defaults, ax1, ax2),
                                  frames=len(optimizer.space.target), repeat=False)
    writer = animation.FFMpegWriter(fps=15)
    ani.save(f'optimization_animation_{X_LB}_{X_UB}.mp4', writer=writer)
    plt.close('all')
    
    # Run final simulation using best parameters:
    wfm = instantiate_wfm(best_params, site, turbine, full_ws, full_ti)
    sim_res, flow_map = run_simulation(wfm, target_x, target_y, full_ws, full_ti)
    
    overall_rmse, avg_errors, p90_errors = plot_flow_field_errors(sim_res, flow_map, flow_roi, target_x, target_y)
    
    # Final bar plot comparing best and default parameter values:
    keys = list(best_params.keys())
    best_vals = [best_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    plt.figure(figsize=(10, 6))
    plt.bar(keys, best_vals, label='Optimized')
    plt.bar(keys, default_vals, edgecolor='black', linewidth=2, fill=False, label='Default')
    plt.title('Final Optimal RMSE: %.4f' % best_rmse)
    plt.tight_layout()
    plt.savefig(f'bar_LB_{X_LB}_UP_{X_UB}.png')
    plt.close()

------------------------------------------------

Explanation of key changes:
1. The code is reorganized into functions. The wake farm model instantiation is now handled by instantiate_wfm() so you can easily call it with any parameter set.
2. A new function run_simulation() packages the simulation over the horizontal grid.
3. The compute_rmse() function builds the objective (in this case negative RMSE) as before, and the evaluation function for bayes_opt simply wraps it.
4. The flow-field plotting now produces not only the contour plots of observed, predicted and difference fields but also prints (and displays on the third subplot) the average error and the p90 (90th percentile) of the absolute error.
5. The convergence animation is now generated with update_convergence_plot() which shows both the full history and best RMSE so far together with a bar plot of best parameters vs default values.
6. All default and pbounds definitions for each branch (DOWNWIND vs upstream and MODEL 1 vs 2) are kept.

This structure should make future changes easier (for example if you want to change how the flow field errors are computed or plotted, or add additional models). 

Feel free to further refactor or add logging etc. as needed.
