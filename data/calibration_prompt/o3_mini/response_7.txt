#!/usr/bin/env python
"""
A modular re-write for wake farm model optimization.
This version adds helper functions to:
  • load and preprocess the reference dataset
  • set the region-of-interest (ROI)
  • create the wake-farm model (wfm) given the MODEL/DOWNWIND logic and deficit settings
  • evaluate the RMSE used for BayesianOptimization
  • animate the optimizer convergence & plot results
"""

def load_reference_data(data_file, turbine):
    """
    Load the reference dataset and scale the coordinate system based on turbine diameter.
    """
    dat = xr.load_dataset(data_file)
    D = turbine.diameter()
    dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
    return dat, D

def get_roi(dat, D, downwind=True):
    """
    Define region-of-interest based on downstream (DOWNWIND) or upstream logic.
    Returns the ROI of the dataset and target coordinates.
    """
    if downwind:
        X_LB, X_UB = 2, 10
    else:
        X_LB, X_UB = -2, -1
        
    roi_x = slice(X_LB * D, X_UB * D)
    roi_y = slice(-2 * D, 2 * D)
    flow_roi = dat.sel(x=roi_x, y=roi_y)
    
    target_x = flow_roi.x
    target_y = flow_roi.y
    return flow_roi, target_x, target_y, X_LB, X_UB

def generate_conditions():
    """
    Generate the range(s) of turbulence intensities (TIs) and wind speeds (WS)
    """
    TIs = np.arange(0.05, 0.45, 0.05)
    WSs = np.arange(4, 11)
    full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
    full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
    
    assert full_ws.size == full_ti.size
    return full_ws, full_ti

def get_wfm(site, turbine, MODEL, DOWNWIND, params, defaults):
    """
    Instantiate and return a wake-farm model (wfm) based on the MODEL and DOWNWIND flags.
    The deficit and turbulence settings are managed by the dictionary "params".
    
    For DOWNWIND True:
      - MODEL==1 uses BlondelSuperGaussianDeficit2020
      - MODEL==2 uses TurboGaussianDeficit
    Else (upstream scenario) the blockage settings are applied.
    """
    # Initialize dictionaries for deficit, turbulence, and blockage arguments
    def_args = {}
    turb_args = {}
    blockage_args = {}
    
    if DOWNWIND:
        if MODEL == 1:
            def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            # No blockage model changes here
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
        else:
            turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
            wake_deficitModel = TurboGaussianDeficit(A=params['A'],
                                                    cTI=[params['cti1'], params['cti2']],
                                                    ctlim=params['ctlim'],
                                                    ceps=params['ceps'],
                                                    ct2a=ct2a_mom1d,
                                                    groundModel=Mirror(),
                                                    rotorAvgModel=GaussianOverlapAvgModel())
            wake_deficitModel.WS_key = 'WS_jlk'
    else:  # upstream scenario
        # For upstream case we may need to adjust blockage parameters
        blockage_args = {'ss_alpha': params['ss_alpha'],
                         'ss_beta': params['ss_beta'],
                         'r12p': np.array([params['rp1'], params['rp2']]),
                         'ngp': np.array([params['ng1'], params['ng2'], params['ng3'], params['ng4']])}
        if MODEL == 2:
            blockage_args['groundModel'] = Mirror()
        wake_deficitModel = BlondelSuperGaussianDeficit2020()  # Default deficit model for upstream
    
    wfm = All2AllIterative(site, turbine,
                           wake_deficitModel=wake_deficitModel,
                           superpositionModel=LinearSum(),
                           deflectionModel=None,
                           turbulenceModel=CrespoHernandez(**turb_args),
                           blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args))
    return wfm

def compute_error_stats(observed, predicted):
    """
    Compute error statistics such as average RMSE and p90 error.
    (Note: p90 indicates the error value below which 90% of errors fall.)
    """
    diff = observed - predicted
    rmse = np.sqrt(np.mean(diff**2))
    p90 = np.percentile(np.abs(diff), 90)
    return rmse, p90

def evaluate_rmse(**kwargs):
    """
    Function to be optimized using BayesianOptimization.
    Given the dictionary of parameter values (kwargs), build the wfm, simulate,
    compute predictions, and return the negative RMSE.
    
    (Negative RMSE is returned since BayesianOptimization maximizes the objective.)
    """
    # These global variables below should be defined in the main scope:
    # - DOWNWIND, MODEL
    # - site, turbine, full_ws, full_ti, target_x, target_y, flow_roi, all_obs
    global DOWNWIND, MODEL, site, turbine, full_ws, full_ti, target_x, target_y, flow_roi, all_obs
    
    # Create a wake-farm model based on parameters passed by the optimizer.
    wfm = get_wfm(site, turbine, MODEL, DOWNWIND, kwargs, defaults={})  # defaults not used in this call

    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    
    # Compute the flow maps for every simulation time step
    flow_map = None
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        if flow_map is None:
            flow_map = fm
        else:
            flow_map = xr.concat([flow_map, fm], dim='time')
    
    pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
    rmse = float(np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
    
    if np.isnan(rmse):
        return -0.5
    return -rmse

def animate_optimization(optimizer, defaults, save_filename, fps=15):
    """
    Create a matplotlib animation that shows the convergence of the optimization
    and how the parameter values evolve relative to their defaults.
    """
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    def update_plot(frame):
        ax1.clear()
        ax2.clear()
        best_so_far_params = {}
        best_so_far_rmse = float('inf')
        best_so_far_rmses = []
        for i in range(frame + 1):
            current_rmse = -optimizer.space.target[i]
            if current_rmse <= best_so_far_rmse:
                best_so_far_rmse = current_rmse
                best_so_far_params = optimizer.res[i]['params']
            best_so_far_rmses.append(best_so_far_rmse)
        
        # Left: RMSE convergence history
        ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5, label='History')
        ax1.plot(np.array(best_so_far_rmses), color='black', label='Best So Far')
        ax1.set_title('Optimization Convergence')
        ax1.set_xlabel('Iteration')
        ax1.set_ylabel('RMSE')
        ax1.grid(True)
        ax1.legend()
        
        # Right: Bar plot of parameter values vs defaults
        keys = list(best_so_far_params.keys()) if best_so_far_params else list(defaults.keys())
        best_vals = [best_so_far_params.get(k, 0) for k in keys]
        default_vals = [defaults[k] for k in keys]
        ax2.bar(keys, best_vals, label='Optimized')
        ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, color='none', capsize=5, label='Default')
        ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
        ax2.tick_params(axis='x', rotation=45)
        ax2.legend()
        plt.tight_layout()
        return ax1, ax2

    ani = animation.FuncAnimation(fig, update_plot, frames=len(optimizer.space.target), repeat=False)
    
    writer = animation.FFMpegWriter(fps=fps)
    ani.save(save_filename, writer=writer)
    plt.close('all')

Explanation of the Modularization

1. The function load_reference_data() loads and scales the dataset.
2. get_roi() defines the ROI and sets target_x, target_y.
3. generate_conditions() builds the arrays full_ws and full_ti.
4. The key function get_wfm() contains the logic that switches between the different
   deficit models (Blondel vs Turbo) for downstream and upstream cases. All
   associated settings (deficit, turbine, and blockage arguments) are passed in.
5. evaluate_rmse() now simply builds the wfm using get_wfm() and computes the RMSE.
6. The animate_optimization() function takes care of the optimization animation.
7. Finally, the main block runs the optimization, computes error statistics (with a hook
   for additional average / p90 error metrics) and creates plotting output.

This structure should make it easier to modify parts of the flow field plotting (for example,
reporting averaged errors and p90 values) or to change how the wake farm model is instantiated,
while keeping the MODEL and UPSTREAM logic centralized.
