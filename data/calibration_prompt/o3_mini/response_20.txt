Below is one way to “rewrite” (refactor) your code into a more modular structure. In this rewrite the key things to note are:

1. We extract routines into separate functions that (a) build the wind‐farm model (wfm), (b) run a simulation over the ROI and compute the error, and (c) animate and report convergence.  
2. We preserve all the “deficit” settings and the MODEL and UPSTREAM (DOWNWIND) logic.  
3. In the simulation and error‐evaluation parts we show how you might compute both an average RMSE and a “p90” (e.g. 90th percentile) error so that later you can change the flow field plots accordingly.  
4. The “main” routine then simply calls the optimization and then uses the best parameters (and defaults) to get predictions and generate plots.

Below is an annotated refactored example. (You’ll have to test it in your environment to check that the variables such as flow_roi.deficits and simulation fields are still in agreement with your libraries.)

────────────────────────────────────────────
#!/usr/bin/env python
"""
A refactored version of the wake optimization script.
Modular functions are created for:
  – building the wind-farm model (wfm)
  – running simulation and evaluating the error (RMSE and p90 errors)
  – updating and saving the optimization convergence plots.
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr

# Import wake models etc.
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW
from py_wake.deficit_models.gaussian import BlondelSuperGaussianDeficit2020
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from bayes_opt import BayesianOptimization
from py_wake.deficit_models.utils import ct2a_mom1d

# Global configuration flags.
DOWNWIND = True
MODEL = 2    # Only allowed [1, 2] otherwise thrown exception
if MODEL not in {1, 2}:
    raise Exception("Bad Model Number")

# Load the reference dataset.
dat = xr.load_dataset('./DTU10MW.nc')
turbine = DTU10MW()
D = turbine.diameter()  # turbine diameter
# Rescale x and y to be multiples of the turbine diameter.
dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)

# Define ROI based on the downwind flag.
if DOWNWIND:
    X_LB, X_UB = 2, 10
else:
    X_LB, X_UB = -2, -1

roi_x = slice(X_LB * D, X_UB * D)
roi_y = slice(-2 * D, 2 * D)
flow_roi = dat.sel(x=roi_x, y=roi_y)
target_x = flow_roi.x
target_y = flow_roi.y

# Build the set of wind speeds and turbulence intensities.
TIs = np.arange(0.05, 0.45, 0.05)
WSs = np.arange(4, 11)
full_ti = np.array([TIs for _ in range(WSs.size)]).flatten()
full_ws = np.array([[WSs[ii]] * TIs.size for ii in range(WSs.size)]).flatten()
assert full_ws.size == full_ti.size

# Create the site instance.
site = Hornsrev1Site()

# Pre-compute reference deficits from the loaded data for all timesteps.
obs_values = []
sim_res0 = All2AllIterative(site, turbine,
                            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
                            superpositionModel=LinearSum(), deflectionModel=None,
                            turbulenceModel=CrespoHernandez(),
                            blockage_deficitModel=SelfSimilarityDeficit2020())(
                            [0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
flow_map0 = sim_res0.flow_map(HorizontalGrid(x=target_x, y=target_y))
for t in range(flow_map0.time.size):
    this_pred_sim = sim_res0.isel(time=t, wt=0)
    # interpolate the observed deficit based on CT and TI from simulation
    observed_deficit = flow_roi.deficits.interp(ct=this_pred_sim.CT,
                                                ti=this_pred_sim.TI, z=0)
    obs_values.append(observed_deficit.T)
all_obs = xr.concat(obs_values, dim='time')

# =============================================================================
#                      FUNCTION DEFINITIONS
# =============================================================================

def create_wfm(params, turbine, site):
    """
    Create and return a wake farm model (wfm) from parameter dictionary.
    This function handles both MODEL 1 and MODEL 2 cases, and the downwind/upstream
    logic. Default values are used if params is None.
    """
    # Build the appropriate models based on the configuration.
    if DOWNWIND:
        if MODEL == 1:
            # For MODEL 1 we use the Blondel deficit model with some parameters.
            def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']} if params else {}
            turb_args = {'c': np.array([params[k] for k in ['ch1', 'ch2', 'ch3', 'ch4']])} if params else {}
            wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
        else:
            # For MODEL 2 we use the TurboGaussianDeficit model.
            wake_deficitModel = TurboGaussianDeficit(A=params['A'], cTI=[params['cti1'], params['cti2']],
                                                     ctlim=params['ctlim'], ceps=params['ceps'],
                                                     ct2a=ct2a_mom1d,
                                                     groundModel=Mirror(),
                                                     rotorAvgModel=GaussianOverlapAvgModel()) if params else \
                                                     TurboGaussianDeficit()
            # Set an alternative key.
            wake_deficitModel.WS_key = 'WS_jlk'
            turb_args = {'c': np.array([params[k] for k in ['ch1', 'ch2', 'ch3', 'ch4']])} if params else {}
    else:
        # Upstream case: Here we only use Blondel for the wake deficit and define blockage arguments.
        wake_deficitModel = BlondelSuperGaussianDeficit2020()  # Using default
        turb_args = {}
        blockage_args = {'ss_alpha': params['ss_alpha'], 'ss_beta': params['ss_beta'],
                         'r12p': np.array([params['rp1'], params['rp2']]),
                         'ngp': np.array([params[k] for k in ['ng1', 'ng2', 'ng3', 'ng4']])} if params else {}
        if MODEL == 2 and params:
            blockage_args['groundModel'] = Mirror()
    # For the blockage deficit model we use the self-similarity approach.
    blockage_deficitModel = SelfSimilarityDeficit2020(**(blockage_args if (not DOWNWIND and params) else {}))
    # Build the overall wind-farm model
    wfm = All2AllIterative(site, turbine,
                           wake_deficitModel=wake_deficitModel,
                           superpositionModel=LinearSum(),
                           deflectionModel=None,
                           turbulenceModel=CrespoHernandez(**turb_args),
                           blockage_deficitModel=blockage_deficitModel)
    return wfm

def simulate_error(wfm, target_x, target_y, full_ws, full_ti, all_obs):
    """
    Run the simulation using the wake farm model (wfm) and compute a normalized error.
    Returns the average RMSE error (could also extract the p90 error) over time.
    """
    sim_res = wfm([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    
    # Compute flow maps per time step.
    flow_maps = []
    for tt in range(full_ws.size):
        fm = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[tt])['WS_eff']
        flow_maps.append(fm)
    flow_map = xr.concat(flow_maps, dim='time')
    
    # Compute the relative error: (WS (simulated) - effective WS field) / Free-stream WS
    pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
    # Compute RMSE over (x,y) and average over time.
    rmse_all = np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y']))
    # Average RMSE over time:
    avg_rmse = float(rmse_all.mean('time'))
    # Additionally, compute the p90 (90th percentile) error over the spatial domain for each time step,
    # then average over time.
    p90_rmse = float(np.percentile(rmse_all, 90))
    return avg_rmse, p90_rmse

def evaluate_rmse(**params):
    """
    Objective function for Bayesian optimization.
    Given model parameters returns negative RMSE (as the optimizer maximizes the objective).
    This function creates a new wfm and runs the simulation.
    If an error is detected (e.g. nan), a penalty is returned.
    """
    # Create the wind-farm model using current parameters.
    wfm = create_wfm(params, turbine, site)
    avg_rmse, _ = simulate_error(wfm, target_x, target_y, full_ws, full_ti, all_obs)
    if np.isnan(avg_rmse):
        return -0.5
    return -avg_rmse

def update_plot(frame, optimizer, defaults, ax1, ax2):
    """
    Callback function for animation.
    It plots (a) the history of RMSE and (b) a bar plot comparing the default and current best value.
    """
    ax1.clear()
    ax2.clear()

    best_so_far_params = {}
    best_so_far_rmse = float('inf')
    best_so_far_rmses = []

    # Loop to extract best point so far
    for i in range(frame + 1):
        current_rmse = -optimizer.space.target[i]
        if current_rmse <= best_so_far_rmse:
            best_so_far_rmse = current_rmse
            best_so_far_params = optimizer.res[i]['params']
        best_so_far_rmses.append(best_so_far_rmse)

    # Plot history of RMSE values in gray.
    ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5)
    # Plot best value so far in black.
    ax1.plot(np.array(best_so_far_rmses), color='black')
    ax1.set_title('Optimization Convergence')
    ax1.set_xlabel('Iteration')
    ax1.set_ylabel('RMSE')
    ax1.grid(True)

    # Bar plot for best parameters vs. defaults.
    keys = list(best_so_far_params.keys())
    best_vals = [best_so_far_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    ax2.bar(keys, best_vals, label='Optimized')
    ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, color='none',
            capsize=5, label='Default')
    ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
    ax2.tick_params(axis='x', rotation=45)
    ax2.legend()
    plt.tight_layout()
    return ax1, ax2

# =============================================================================
#                            MAIN EXECUTION
# =============================================================================

if __name__ == '__main__':
    # Define parameter bounds and defaults based on DOWNWIND and MODEL.
    if MODEL == 1:
        if DOWNWIND:
            pbounds = {
                'a_s': (0.001, 0.5),
                'b_s': (0.001, 0.01),
                'c_s': (0.001, 0.5),
                'b_f': (-2, 1),
                'c_f': (0.1, 5),
                'ch1': (-1, 2),
                'ch2': (-1, 2),
                'ch3': (-1, 2),
                'ch4': (-1, 2)
            }
            defaults = {
                'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2,
                'b_f': -0.68, 'c_f': 2.41,
                'ch1': 0.73, 'ch2': 0.8325, 'ch3': -0.0325, 'ch4': -0.32
            }
        else:
            pbounds = {
                'ss_alpha': (0.05, 3),
                'ss_beta': (0.05, 3),
                'rp1': (-2, 2),
                'rp2': (-2, 2),
                'ng1': (-3, 3),
                'ng2': (-3, 3),
                'ng3': (-3, 3),
                'ng4': (-3, 3),
                'fg1': (-2, 2),
                'fg2': (-2, 2),
                'fg3': (-2, 2),
                'fg4': (-2, 2)
            }
            defaults = {
                'ss_alpha': 0.8888888888888888,
                'ss_beta': 1.4142135623730951,
                'rp1': -0.672,
                'rp2': 0.4897,
                'ng1': -1.381,
                'ng2': 2.627,
                'ng3': -1.524,
                'ng4': 1.336,
                'fg1': -0.06489,
                'fg2': 0.4911,
                'fg3': 1.116,
                'fg4': -0.1577
            }
    else:  # MODEL == 2
        defaults = {
            'A': 0.04,
            'cti1': 1.5,
            'cti2': 0.8,
            'ceps': 0.25,
            'ctlim': 0.999,
            'ch1': 0.73,
            'ch2': 0.8325,
            'ch3': -0.0325,
            'ch4': -0.3
        }
        pbounds = {
            'A': (0.001, .5),
            'cti1': (.01, 5),
            'cti2': (0.01, 5),
            'ceps': (0.01, 3),
            'ctlim': (0.01, 1),
            'ch1': (-1, 2),
            'ch2': (-1, 2),
            'ch3': (-1, 2),
            'ch4': (-1, 2)
        }

    # Set up the Bayesian optimizer.
    optimizer = BayesianOptimization(f=evaluate_rmse, pbounds=pbounds, random_state=1)
    optimizer.probe(params=defaults, lazy=True)
    optimizer.maximize(init_points=50, n_iter=200)

    best_params = optimizer.max['params']
    best_rmse = -optimizer.max['target']
    print("Best parameters:", best_params)
    print("Best RMSE:", best_rmse)

    # Create animated plots.
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))

    def anim_update(frame):
        return update_plot(frame, optimizer, defaults, ax1, ax2)

    ani = animation.FuncAnimation(fig, anim_update,
                                  frames=len(optimizer.space.target), repeat=False)

    # Save animation as MP4:
    writer = animation.FFMpegWriter(fps=15)
    ani.save('optimization_animation_%i_%i.mp4' % (X_LB, X_UB), writer=writer)
    plt.close('all')

    # Run final simulation with best parameters.
    wfm_final = create_wfm(best_params, turbine, site)
    sim_res_final = wfm_final([0], [0], ws=full_ws, TI=full_ti, wd=[270] * full_ti.size, time=True)
    
    # Get flow map for WS_eff from the final simulation.
    flow_maps_final = []
    rmse_values = []
    for t in range(sim_res_final.time.size):
        this_pred_sim = sim_res_final.isel(time=t)
        # Note: We use the reference flow map (flow_map0) for plotting; you might re-run if needed.
        flow_map_t = wfm_final.flow_map(HorizontalGrid(x=target_x, y=target_y), time=[t])['WS_eff']
        flow_maps_final.append(flow_map_t)
        # Compute predictions (relative deficit)
        pred = (this_pred_sim.WS - flow_map_t.isel(h=0)) / this_pred_sim.WS
        # Compute differences with the observed deficits
        observed_deficit = flow_roi.deficits.interp(ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0).isel(wt=0)
        diff = observed_deficit.T - pred
        rmse = np.sqrt(np.mean(diff**2))
        rmse_values.append(rmse)
        # Plot flow fields: observed, prediction, and difference.
        fig, ax = plt.subplots(3, 1, figsize=(5, 15))
        c0 = ax[0].contourf(target_x, target_y, observed_deficit.T)
        c1 = ax[1].contourf(target_x, target_y, pred)
        c2 = ax[2].contourf(target_x, target_y, diff)
        for jj, contour in enumerate([c0, c1, c2]):
            fig.colorbar(contour, ax=ax[jj])
        ax[0].set_ylabel('Observed')
        ax[1].set_ylabel('Prediction')
        ax[2].set_ylabel('Diff')
        plt.tight_layout()
        plt.savefig(f'figs/downstream_err_{t}.png')
        plt.close(fig)
        
    overall_rmse = np.mean(rmse_values)
    # Also compute p90 error across time if desired.
    overall_p90 = np.percentile(rmse_values, 90)
    print("RMSE values per time step:", rmse_values)
    print("Overall average RMSE:", overall_rmse)
    print("Overall p90 RMSE:", overall_p90)

    # Final bar plot comparing best vs. default parameters.
    keys = list(best_params.keys())
    best_vals = [best_params[k] for k in keys]
    default_vals = [defaults[k] for k in keys]
    plt.bar(keys, best_vals, label='Optimized')
    plt.bar(keys, default_vals,
            edgecolor='black', linewidth=2,
            color='none', capsize=5, label='Default')
    plt.title('Optimal RMSE: %.4f' % best_rmse)
    plt.tight_layout()
    plt.savefig('bar_LB_%i_UP_%i.png' % (X_LB, X_UB))
    plt.close('all')

    print("Animation and final figures saved.")

Explanation:

• The function create_wfm(params, turbine, site) encapsulates the logic for instantiating the wake farm model. It reads the parameters dictionary (or uses defaults if needed) and builds the correct deficit model (TurboGaussian or Blondel) plus the turbulence and blockage settings.  
• The function simulate_error(…) runs the simulation and returns both the average RMSE and a p90 metric. You can then change your flow field plots to include these summary statistics.  
• The evaluate_rmse function is identical in spirit to your original objective function except that it now uses create_wfm(…) and simulate_error(…).  
• The update_plot function (and its wrapper for FuncAnimation) displays the convergence history and the parameter bar charts.
• Finally, the __main__ portion sets up the BayesianOptimization (with defaults and pbounds), runs the optimization, and then uses the found best parameters to re-run the simulation and produce the final set of plots.

This approach modularizes the code and should make it easier to modify, test and extend. You can now easily adjust what goes into create_wfm (for example, adding new deficit settings) or change how the error is computed and reported (by updating simulate_error).
