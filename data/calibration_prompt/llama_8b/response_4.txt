    dat = xr.load_dataset('./DTU10MW.nc')

    D = turbine.diameter()
    dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)

def get_roi(dat, X_LB, X_UB):
    roi_x = slice(X_LB * D, X_UB * D)
    roi_y = slice(-2 * D, 2 * D)
    flow_roi = dat.sel(x=roi_x, y=roi_y)
    target_x = flow_roi.x
    target_y = flow_roi.y
    return flow_roi, target_x, target_y

    wfm = All2AllIterative(site, turbine,

                            superpositionModel=superpositionModel, deflectionModel=deflectionModel,
                            turbulenceModel=turbulenceModel,
                            blockage_deficitModel=blockage_deficitModel)
    return wfm

    flow_map = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y))
