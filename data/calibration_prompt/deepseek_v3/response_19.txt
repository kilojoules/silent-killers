Looking at your code, I can see several areas for improvement. The main issues are:

Here's a refactored version that addresses these issues:

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

class WakeModelOptimizer:
    """Class to handle wake model optimization and evaluation"""
    
    def __init__(self, model_type=1, downstream=True):
        """
        Initialize the optimizer
        
        Parameters
        ----------
        model_type : int
            1 for BlondelSuperGaussianDeficit2020, 2 for TurboGaussianDeficit
        downstream : bool
            True for downstream wake, False for upstream blockage
        """
        self.model_type = model_type
        self.downstream = downstream
        
        # Load and prepare data
        self.load_data()
        self.setup_site_and_turbine()
        self.setup_parameter_bounds()
        
    def load_data(self):
        """Load and prepare the reference data"""
        dat = xr.load_dataset('./DTU10MW.nc')
        turbine = DTU10MW()
        D = turbine.diameter()
        
        # Scale coordinates by diameter
        dat = dat.assign_coords(x=dat.x * D, y=dat.y * D)
        
        # Define region of interest
        if self.downstream:
            X_LB, X_UB = (2, 10)
        else:
            X_LB, X_UB = (-2, -1)
            
        roi_x = slice(X_LB * D, X_UB * D)
        roi_y = slice(-2 * D, 2 * D)
        
        self.flow_roi = dat.sel(x=roi_x, y=roi_y)
        self.target_x = self.flow_roi.x
        self.target_y = self.flow_roi.y
        self.X_LB = X_LB
        self.X_UB = X_UB
        
    def setup_site_and_turbine(self):
        """Setup site and turbine configuration"""
        self.site = Hornsrev1Site()
        self.turbine = DTU10MW()
        
        # Setup wind speed and turbulence intensity combinations
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.tile(TIs, WSs.size)
        full_ws = np.repeat(WSs, TIs.size)
        
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.full_wd = np.full_like(full_ws, 270)
        
        # Pre-calculate observed deficits
        self.precalculate_observed_deficits()
        
    def precalculate_observed_deficits(self):
        """Pre-calculate observed deficits for all conditions"""
        # Create a baseline simulation to get CT values
        baseline_wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        baseline_sim = baseline_wfm(
            [0], [0], 
            ws=self.full_ws, 
            TI=self.full_ti, 
            wd=self.full_wd, 
            time=True
        )
        
        # Extract observed deficits for all conditions
        obs_values = []
        for t in range(baseline_sim.time.size):
            this_sim = baseline_sim.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_sim.CT, 
                ti=this_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
        
        self.all_obs = xr.concat(obs_values, dim='time')
        
    def setup_parameter_bounds(self):
        """Setup parameter bounds based on model type"""
        if self.model_type == 1:
            if self.downstream:
                self.pbounds = {
                    'a_s': (0.001, 0.5),
                    'b_s': (0.001, 0.01),
                    'c_s': (0.001, 0.5),
                    'b_f': (-2, 1),
                    'c_f': (0.1, 5),
                    'ch1': (-1, 2),
                    'ch2': (-1, 2),
                    'ch3': (-1, 2),
                    'ch4': (-1, 2),
                }
                self.defaults = {
                    'a_s': 0.17, 'b_s': 0.005, 'c_s': 0.2, 
                    'b_f': -0.68, 'c_f': 2.41,
                    'ch1': 0.73, 'ch2': 0.8325, 
                    'ch3': -0.0325, 'ch4': -0.32
                }
            else:
                self.pbounds = {
                    'ss_alpha': (0.05, 3),
                    'ss_beta': (0.05, 3),
                    'rp1': (-2, 2),
                    'rp2': (-2, 2),
                    'ng1': (-3, 3),
                    'ng2': (-3, 3),
                    'ng3': (-3, 3),
                    'ng4': (-3, 3),
                    'fg1': (-2, 2),
                    'fg2': (-2, 2),
                    'fg3': (-2, 2),
                    'fg4': (-2, 2)
                }
                self.defaults = {
                    'ss_alpha': 0.8888888888888888,
                    'ss_beta': 1.4142135623730951,
                    'rp1': -0.672,
                    'rp2': 0.4897,
                    'ng1': -1.381,
                    'ng2': 2.627,
                    'ng3': -1.524,
                    'ng4': 1.336,
                    'fg1': -0.06489,
                    'fg2': 0.4911,
                    'fg3': 1.116,
                    'fg4': -0.1577
                }
        else:  # model_type == 2
            self.pbounds = {
                'A': (0.001, 0.5),
                'cti1': (0.01, 5),
                'cti2': (0.01, 5),
                'ceps': (0.01, 3),
                'ctlim': (0.01, 1),
                'ch1': (-1, 2),
                'ch2': (-1, 2),
                'ch3': (-1, 2),
                'ch4': (-1, 2),
            }
            self.defaults = {
                'A': 0.04,
                'cti1': 1.5,
                'cti2': 0.8,
                'ceps': 0.25,
                'ctlim': 0.999,
                'ch1': 0.73,
                'ch2': 0.8325,
                'ch3': -0.0325,
                'ch4': -0.3
            }
    
    def create_wake_model(self, params):
        """Create a wake model with given parameters"""
        
        if self.downstream:
            if self.model_type == 1:
                # BlondelSuperGaussianDeficit2020 for downstream
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
                blockage_args = {}
            else:
                # TurboGaussianDeficit for downstream
                wake_deficit_model = TurboGaussianDeficit(
                    A=params['A'],
                    cTI=[params['cti1'], params['cti2']],
                    ctlim=params['ctlim'],
                    ceps=params['ceps'],
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficit_model.WS_key = 'WS_jlk'
                blockage_args = {}
            
            turb_args = {'c': np.array([params['ch1'], params['ch2'], 
                                       params['ch3'], params['ch4']])}
            
        else:
            # Upstream blockage
            def_args = {}
            turb_args = {}
            blockage_args = {
                'ss_alpha': params['ss_alpha'],
                'ss_beta': params['ss_beta'],
                'r12p': np.array([params['rp1'], params['rp2']]),
                'ngp': np.array([params['ng1'], params['ng2'], 
                                params['ng3'], params['ng4']])
            }
            
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
            
            wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
        
        # Create the wind farm model
        wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=wake_deficit_model,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm
    
    def calculate_deficits(self, sim_res):
        """Calculate velocity deficits from simulation results"""
        # Create flow map for all time steps at once
        flow_map = sim_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        # Calculate velocity deficits: (U_inf - U_wake) / U_inf
        # Note: flow_map.WS_eff is the wake velocity
        deficits = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
        
        return deficits
    
    def evaluate_rmse(self, **params):
        """Evaluate RMSE for given parameters"""
        try:
            # Create wake model with current parameters
            wfm = self.create_wake_model(params)
            
            # Run simulation
            sim_res = wfm(
                [0], [0],
                ws=self.full_ws,
                TI=self.full_ti,
                wd=self.full_wd,
                time=True
            )
            
            # Calculate predicted deficits
            pred_deficits = self.calculate_deficits(sim_res)
            
            # Calculate RMSE
            rmse = float(np.sqrt(((self.all_obs - pred_deficits) ** 2)
                                .mean(['x', 'y']))
                        .mean('time'))
            
            if np.isnan(rmse):
                return -0.5
                
            return -rmse  # Negative for maximization
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def run_optimization(self, init_points=50, n_iter=200):
        """Run Bayesian optimization"""
        self.optimizer = BayesianOptimization(
            f=self.evaluate_rmse,
            pbounds=self.pbounds,
            random_state=1
        )
        
        # Probe with default values
        self.optimizer.probe(params=self.defaults, lazy=True)
        
        # Run optimization
        self.optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        # Extract best parameters
        self.best_params = self.optimizer.max['params']
        self.best_rmse = -self.optimizer.max['target']
        
        return self.best_params, self.best_rmse
    
    def create_optimization_animation(self, filename):
        """Create animation of optimization progress"""
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Track best RMSE so far
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                current_rmse = -self.optimizer.space.target[i]
                if current_rmse <= best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(self.optimizer.space.target[:frame+1]), 
                    color='gray', alpha=0.5, label='All evaluations')
            ax1.plot(best_so_far_rmses, color='black', linewidth=2, 
                    label='Best so far')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            ax1.legend()
            
            # Get best parameters so far
            best_idx = np.argmin(-np.array(self.optimizer.space.target[:frame+1]))
            best_params_so_far = self.optimizer.res[best_idx]['params']
            
            # Plot parameter comparison
            keys = list(best_params_so_far.keys())
            best_vals = [best_params_so_far[k] for k in keys]
            default_vals = [self.defaults.get(k, 0) for k in keys]
            
            x_pos = np.arange(len(keys))
            ax2.bar(x_pos - 0.2, best_vals, 0.4, label='Optimized', alpha=0.8)
            ax2.bar(x_pos + 0.2, default_vals, 0.4, label='Default', alpha=0.8)
            ax2.set_xticks(x_pos)
            ax2.set_xticklabels(keys, rotation=45, ha='right')
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(self.optimizer.space.target),
            repeat=False
        )
        
        # Save animation
        writer = animation.FFMpegWriter(fps=15)
        ani.save(filename, writer=writer)
        plt.close('all')
    
    def evaluate_best_model(self):
        """Evaluate the best model and create visualizations"""
        # Create model with best parameters
        best_wfm = self.create_wake_model(self.best_params)
        
        # Run simulation
        sim_res = best_wfm(
            [0], [0],
            ws=self.full_ws,

        )
        
        # Calculate deficits
        flow_map = sim_res.flow_map(

        # Calculate RMSE for each time step
        rmse_values = []

        for t in range(sim_res.time.size):
            obs = self.all_obs.isel(time=t)

            error = obs - pred
            
            # Statistics
            rmse = float(np.sqrt(np.mean(error**2)))
            mean_error = float(np.mean(error))
            p90_error = float(np.percentile(np.abs(error), 90))
            
            rmse_values.append(rmse)
            error_stats.append({

                'rmse': rmse,
                'mean_error': mean_error,

            )
        
        # Calculate overall statistics
        overall_rmse = np.mean(rmse_values)
        error_array = (self.all_obs - pred_deficits).values.flatten()
        overall_mean_error = np.mean(error_array)
        overall_p90_error = np.percentile(np.abs(error_array), 90)
        
        print(f"Overall RMSE: {overall_rmse:.4f}")
        print(f"Overall mean error: {overall_mean_error:.4f}")
        print(f"Overall P90 error: {overall_p90_error:.4f}")
        
        # Create summary plots
        self.create_summary_plots(
            rmse_values, error_stats,
            overall_rmse, overall_mean_error, overall_p90_error
        )
        
        return rmse_values, error_stats
