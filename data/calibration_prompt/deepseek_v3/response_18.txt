import xarray as xr
import numpy as np
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW


class DataLoader:
    """Handles loading and preprocessing of reference data"""
    
    def __init__(self, data_path='./DTU10MW.nc'):
        self.data_path = data_path
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        
    def load_data(self, downstream=True, x_lb=2, x_ub=10):
        """Load and preprocess data for specified region"""
        
        # Load data
        dat = xr.load_dataset(self.data_path)
        
        # Scale coordinates by diameter
        dat = dat.assign_coords(x=dat.x * self.D, y=dat.y * self.D)
        
        # Define region of interest
        if downstream:
            roi_x = slice(x_lb * self.D, x_ub * self.D)
        else:
            roi_x = slice(x_ub * self.D, x_lb * self.D)  # Note: x_ub < x_lb for upstream
            
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        # Extract region
        flow_roi = dat.sel(x=roi_x, y=roi_y)
        
        return flow_roi, dat
    
    def get_operating_conditions(self):
        """Generate operating conditions for evaluation"""
        TIs = np.arange(0.05, 0.45, 0.05)
        WSs = np.arange(4, 11)
        
        full_ti = np.tile(TIs, WSs.size)
        full_ws = np.repeat(WSs, TIs.size)
        
        return full_ws, full_ti

# evaluator.py
import numpy as np
import xarray as xr
from py_wake import HorizontalGrid


class ModelEvaluator:
    """Evaluates model performance against reference data"""
    
    def __init__(self, flow_roi, target_x, target_y):
        self.flow_roi = flow_roi
        self.target_x = target_x
        self.target_y = target_y
        
    def prepare_observations(self, sim_res, full_ws, full_ti):
        """Prepare observed deficits from reference data"""
        obs_values = []
        
        for t in range(full_ws.size):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, 
                ti=this_pred_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
        
        return xr.concat(obs_values, dim='time')
    
    def calculate_errors(self, sim_res, flow_map, all_obs):
        """Calculate various error metrics"""
        
        # Calculate predicted deficits
        pred = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
        
        # Calculate RMSE
        rmse = float(np.sqrt(((all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
        
        # Calculate additional error metrics
        abs_error = np.abs(all_obs - pred)
        
        error_metrics = {
            'rmse': rmse,
            'mae': float(abs_error.mean(['x', 'y', 'time'])),
            'p90_error': float(abs_error.quantile(0.9, ['x', 'y', 'time'])),
            'max_error': float(abs_error.max(['x', 'y', 'time'])),
            'bias': float((all_obs - pred).mean(['x', 'y', 'time']))
        }
        
        return error_metrics, pred
    
    def evaluate_flow_field(self, sim_res, wfm, full_ws, full_ti):
        """Evaluate flow field predictions"""
        
        # Generate flow field predictions
        flow_map = None
        for t in range(full_ws.size):
            fm = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y), 
                time=[t]
            )['WS_eff']
            
            if flow_map is None:
                flow_map = fm
            else:
                flow_map = xr.concat([flow_map, fm], dim='time')
        
        # Prepare observations
        all_obs = self.prepare_observations(sim_res, full_ws, full_ti)
        
        # Calculate errors
        error_metrics, pred = self.calculate_errors(sim_res, flow_map, all_obs)
        
        return error_metrics, pred, all_obs, flow_map

# optimizer.py
from bayes_opt import BayesianOptimization
import numpy as np


class ModelOptimizer:
    """Handles Bayesian optimization of model parameters"""
    
    def __init__(self, evaluator, data_loader, model_factory, site, turbine):
        self.evaluator = evaluator
        self.data_loader = data_loader
        self.model_factory = model_factory
        self.site = site
        self.turbine = turbine
        
    def create_objective_function(self, downstream=True, model=1):
        """Create objective function for optimization"""
        
        full_ws, full_ti = self.data_loader.get_operating_conditions()
        
        def objective(**kwargs):
            # Create WFM with current parameters
            wfm = self.model_factory.create_wfm(
                self.site, self.turbine, 
                downstream=downstream, model=model, **kwargs
            )
            
            # Run simulation
            sim_res = wfm(
                [0], [0], 
                ws=full_ws, 
                TI=full_ti, 
                wd=[270] * full_ti.size, 
                time=True
            )
            
            # Evaluate
            error_metrics, _, _, _ = self.evaluator.evaluate_flow_field(
                sim_res, wfm, full_ws, full_ti
            )
            
            # Return negative RMSE for minimization
            rmse = error_metrics['rmse']
            if np.isnan(rmse):
                return -0.5
            return -rmse
        
        return objective
    
    def optimize(self, pbounds, defaults, downstream=True, model=1, 
                 init_points=50, n_iter=200):
        """Run Bayesian optimization"""
        
        objective = self.create_objective_function(downstream, model)
        
        optimizer = BayesianOptimization(
            f=objective,
            pbounds=pbounds,
            random_state=1
        )
        
        optimizer.probe(params=defaults, lazy=True)
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        return optimizer
