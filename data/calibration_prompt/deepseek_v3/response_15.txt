class WakeModelFactory:
    """Factory class for creating wind farm models with different configurations"""
    
    def __init__(self, model_type, downstream=True):
        self.model_type = model_type
        self.downstream = downstream
        self.site = Hornsrev1Site()
        self.turbine = DTU10MW()
        
    def create_wfm(self, params):
        """Create wind farm model based on parameters"""
        
        if self.downstream:
            if self.model_type == 1:
                # BlondelSuperGaussianDeficit2020 for downstream
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficitModel = BlondelSuperGaussianDeficit2020(**def_args)
                turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
                blockage_args = {}
                
            elif self.model_type == 2:
                # TurboGaussianDeficit for downstream
                wake_deficitModel = TurboGaussianDeficit(
                    A=params['A'],
                    cTI=[params['cti1'], params['cti2']],
                    ctlim=params['ctlim'],
                    ceps=params['ceps'],
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficitModel.WS_key = 'WS_jlk'
                turb_args = {'c': np.array([params['ch1'], params['ch2'], params['ch3'], params['ch4']])}
                blockage_args = {}
                
        else:
            # Upstream/blockage models
            def_args = {}
            turb_args = {}
            blockage_args = {
                'ss_alpha': params['ss_alpha'],
                'ss_beta': params['ss_beta'],
                'r12p': np.array([params['rp1'], params['rp2']]),
                'ngp': np.array([params['ng1'], params['ng2'], params['ng3'], params['ng4']])
            }
            
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
        
        # Create wind farm model
        wfm = All2AllIterative(
            site=self.site,
            windTurbines=self.turbine,
            wake_deficitModel=wake_deficitModel,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm


class DataLoader:

class WakeOptimizer:
    """Main optimization class"""
    
    def __init__(self, model_factory, data_loader, flow_roi, full_ws, full_ti):
        self.model_factory = model_factory
        self.data_loader = data_loader
        self.flow_roi = flow_roi
        self.full_ws = full_ws
        self.full_ti = full_ti
        
        # Pre-load observed deficits
        self.observed_deficits = self._load_observed_deficits()
        
    def _load_observed_deficits(self):
        """Load all observed deficit data"""
        obs_values = []
        
        # Create a dummy simulation to get CT/TI values
        site = Hornsrev1Site()
        turbine = DTU10MW()
        wfm = All2AllIterative(
            site, turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                      wd=[270] * self.full_ti.size, time=True)
        
        target_x = self.flow_roi.x
        target_y = self.flow_roi.y
        flow_map = sim_res.flow_map(HorizontalGrid(x=target_x, y=target_y))
        
        for t in range(flow_map.time.size):
            this_pred_sim = sim_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            obs_values.append(observed_deficit.T)
        
        return xr.concat(obs_values, dim='time')
    
    def evaluate_rmse(self, **kwargs):
        """Evaluate RMSE for given parameters"""
        try:
            # Create wind farm model
            wfm = self.model_factory.create_wfm(kwargs)
            
            # Run simulation
            sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti, 
                          wd=[270] * self.full_ti.size, time=True)
            
            # Calculate flow map for all time steps
            target_x = self.flow_roi.x
            target_y = self.flow_roi.y
            
            flow_maps = []
            for tt in range(self.full_ws.size):
                fm = sim_res.flow_map(
                    HorizontalGrid(x=target_x, y=target_y), 
                    time=[tt]
                )['WS_eff']
                flow_maps.append(fm)
            
            flow_map = xr.concat(flow_maps, dim='time')
            
            # Calculate velocity deficit (CORRECTED)
            # WS_eff is the effective wind speed at each point
            # sim_res.WS is the freestream wind speed
            # Velocity deficit = (WS_freestream - WS_effective) / WS_freestream
            predicted_deficit = (sim_res.WS - flow_map.isel(h=0)) / sim_res.WS
            
            # Calculate RMSE
            rmse = float(np.sqrt(((self.observed_deficits - predicted_deficit) ** 2)
                                .mean(['x', 'y'])).mean('time'))
            
            if np.isnan(rmse):
                return -0.5
            return -rmse
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def run_optimization(self, pbounds, defaults, init_points=50, n_iter=200):
        """Run Bayesian optimization"""
        optimizer = BayesianOptimization(
            f=self.evaluate_rmse,
            pbounds=pbounds,
            random_state=1
        )
        
        optimizer.probe(params=defaults, lazy=True)
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        return optimizer


class FlowFieldAnalyzer:
    """Analyzes flow field results with improved statistics"""
    
    def __init__(self, flow_roi, target_x, target_y):
        self.flow_roi = flow_roi
        self.target_x = target_x
        self.target_y = target_y
        
    def analyze_flow_field(self, sim_res, flow_map, best_params, model_type, downstream):
        """Analyze flow field with comprehensive statistics"""
        
        rmse_values = []
        error_maps = []
        
        for t in range(flow_map.time.size):
            this_pred_sim = sim_res.isel(time=t)
            
            # Get observed deficit
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            ).isel(wt=0)
            
            # Calculate predicted deficit
            predicted_deficit = (this_pred_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_pred_sim.WS
            
            # Calculate error
            error = observed_deficit.T - predicted_deficit
            error_maps.append(error)
            
            # Calculate RMSE for this time step
            rmse = np.sqrt(np.mean(error**2))
            rmse_values.append(rmse)
            
            # Create detailed plot for this time step
            self._create_detailed_plot(
                t, observed_deficit.T, predicted_deficit, error,
                this_pred_sim.WS.values, this_pred_sim.TI.values
            )
        
        # Calculate comprehensive statistics
        all_errors = xr.concat(error_maps, dim='time')
        
        stats = {
            'rmse_per_step': rmse_values,
            'overall_rmse': np.mean(rmse_values),
            'mean_error': float(all_errors.mean()),
            'std_error': float(all_errors.std()),
            'p90_error': float(all_errors.quantile(0.9)),
            'p95_error': float(all_errors.quantile(0.95)),
            'max_error': float(all_errors.max()),
            'min_error': float(all_errors.min())
        }
        
        # Create summary plots
        self._create_summary_plots(all_errors, stats, best_params, model_type, downstream)
        
        return stats

class AnimationCreator:
    """Creates optimization animation"""
    
    @staticmethod
    def create_animation(optimizer, defaults, X_LB, X_UB):
        """Create optimization convergence animation"""
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Get best parameters up to current frame
            best_so_far_params = {}
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                if -optimizer.space.target[i] <= best_so_far_rmse:
                    best_so_far_rmse = -optimizer.space.target[i]
                    best_so_far_params = optimizer.res[i]['params']
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5)
            ax1.plot(np.array(best_so_far_rmses), color='black')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Plot parameters
            keys = list(best_so_far_params.keys())
            best_vals = [best_so_far_params[k] for k in keys]
            default_vals = [defaults.get(k, 0) for k in keys]
            
            x = np.arange(len(keys))
            width = 0.35
            
            ax2.bar(x - width/2, best_vals, width, label='Optimized', alpha=0.8)
            ax2.bar(x + width/2, default_vals, width, label='Default', alpha=0.8)
            
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.set_xticks(x)
            ax2.set_xticklabels(keys, rotation=45, ha='right')
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(optimizer.space.target), 
            repeat=False
        )
        
        # Save animation
        writer = animation.FFMpegWriter(fps=15)
        ani.save(f'optimization_animation_{X_LB}_{X_UB}.mp4', writer=writer)
        plt.close('all')
