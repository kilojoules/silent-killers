class ModelEvaluator:
    """Handles model evaluation and RMSE calculation."""
    
    def __init__(self, reference_data, model_factory):
        self.reference_data = reference_data
        self.model_factory = model_factory
        
        # Create test conditions
        self.TIs = np.arange(0.05, 0.45, 0.05)
        self.WSs = np.arange(4, 11)
        self.full_ti = np.tile(self.TIs, len(self.WSs))
        self.full_ws = np.repeat(self.WSs, len(self.TIs))
        
        # Pre-compute reference deficits for all conditions
        self._precompute_reference_deficits()
    
    def _precompute_reference_deficits(self):
        """Pre-compute reference deficits for all test conditions."""
        self.reference_deficits = []
        
        # We'll compute these on-demand in evaluate() to match model CT/TI values
        pass
    
    def calculate_deficit(self, sim_res, flow_map):
        """Calculate velocity deficit from simulation results."""
        # IMPORTANT: flow_map.WS_eff is the effective wind speed
        # Deficit = 1 - (WS_eff / WS_freestream)
        ws_freestream = sim_res.WS  # Freestream wind speed
        ws_eff = flow_map.WS_eff  # Effective wind speed in wake
        
        # Calculate deficit: 1 - (velocity in wake / freestream velocity)
        deficit = 1 - (ws_eff / ws_freestream)
        return deficit
    
    def evaluate(self, **kwargs):
        """Evaluate model with given parameters."""
        try:
            # Create model
            wfm = self.model_factory.create_wake_model(**kwargs)
            
            # Run simulation for all conditions
            sim_res = wfm(
                [0], [0],  # Single turbine at origin
                ws=self.full_ws,
                TI=self.full_ti,
                wd=[270] * len(self.full_ws),
                time=True
            )
            
            # Create grid for flow map
            grid = HorizontalGrid(
                x=self.reference_data.x,
                y=self.reference_data.y
            )
            
            # Calculate flow map for all time steps at once
            flow_map = sim_res.flow_map(grid=grid)
            
            # Calculate predicted deficits
            pred_deficits = self.calculate_deficit(sim_res, flow_map)
            
            # Get reference deficits for each condition
            errors = []
            for t in range(len(self.full_ws)):
                # Get CT and TI for this condition
                ct = float(sim_res.CT.isel(time=t, wt=0))
                ti = float(sim_res.TI.isel(time=t, wt=0))
                
                # Interpolate reference deficit
                ref_deficit = self.reference_data.deficits.interp(
                    ct=ct, ti=ti, z=0
                ).T
                
                # Get predicted deficit for this time
                pred_deficit = pred_deficits.isel(time=t, h=0)
                
                # Calculate error
                error = ref_deficit - pred_deficit
                errors.append(error)
            
            # Calculate overall RMSE
            all_errors = xr.concat(errors, dim='time')
            rmse = float(np.sqrt((all_errors ** 2).mean(['x', 'y'])).mean('time'))
            
            return -rmse if not np.isnan(rmse) else -0.5
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5

# Visualization functions
def create_optimization_animation(optimizer, defaults, output_path):
    """Create animation of optimization progress."""
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    def update_plot(frame):
        ax1.clear()
        ax2.clear()
        
        # Track best RMSE so far
        best_so_far_rmse = float('inf')
        best_so_far_rmses = []
        
        for i in range(frame + 1):
            current_rmse = -optimizer.space.target[i]
            if current_rmse <= best_so_far_rmse:
                best_so_far_rmse = current_rmse
            best_so_far_rmses.append(best_so_far_rmse)
        
        # Plot convergence
        ax1.plot(-np.array(optimizer.space.target[:frame+1]), 
                color='gray', alpha=0.5, label='All evaluations')
        ax1.plot(best_so_far_rmses, color='black', linewidth=2, 
                label='Best so far')
        ax1.set_title('Optimization Convergence')
        ax1.set_xlabel('Iteration')
        ax1.set_ylabel('RMSE')
        ax1.legend()
        ax1.grid(True)
        
        # Plot parameters for best iteration so far
        best_idx = np.argmin(-np.array(optimizer.space.target[:frame+1]))
        best_params = optimizer.res[best_idx]['params']
        
        keys = list(best_params.keys())
        best_vals = [best_params[k] for k in keys]
        default_vals = [defaults.get(k, 0) for k in keys]
        
        x_pos = np.arange(len(keys))
        ax2.bar(x_pos - 0.2, best_vals, width=0.4, label='Optimized')
        ax2.bar(x_pos + 0.2, default_vals, width=0.4, label='Default')
        ax2.set_xticks(x_pos)
        ax2.set_xticklabels(keys, rotation=45)
        ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
        ax2.legend()
        
        plt.tight_layout()
        return ax1, ax2
    
    ani = animation.FuncAnimation(
        fig, update_plot, 
        frames=len(optimizer.space.target),
        repeat=False
    )
    
    writer = animation.FFMpegWriter(fps=15)
    ani.save(output_path, writer=writer)
    plt.close(fig)

    all_errors = []
    all_ref_deficits = []
    all_pred_deficits = []
    
    for t in range(len(test_conditions['ws'])):
        ct = float(sim_res.CT.isel(time=t, wt=0))
        ti = float(sim_res.TI.isel(time=t, wt=0))

        pred_deficit = pred_deficits.isel(time=t, h=0)
