Looking at your code, I can see several areas for improvement. The main issues are:

1. **Monolithic structure** - Everything is in one big script
2. **Repetitive code** - Similar logic appears in multiple places
3. **Hard to maintain** - Changing model configurations requires modifying many parts
4. **Inefficient flow map computation** - You're computing flow maps in a loop

Here's a modular rewrite that addresses these issues while keeping all your deficit settings and MODEL/UPSTREAM logic:

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import xarray as xr
from bayes_opt import BayesianOptimization

# PyWake imports
from py_wake.rotor_avg_models.gaussian_overlap_model import GaussianOverlapAvgModel
from py_wake.deficit_models.gaussian import TurboGaussianDeficit, BlondelSuperGaussianDeficit2020
from py_wake.deficit_models.utils import ct2a_mom1d
from py_wake.deficit_models import SelfSimilarityDeficit2020
from py_wake import HorizontalGrid
from py_wake.deflection_models import JimenezWakeDeflection
from py_wake.turbulence_models import CrespoHernandez
from py_wake.rotor_avg_models import RotorCenter
from py_wake.wind_farm_models import PropagateDownwind, All2AllIterative
from py_wake.superposition_models import LinearSum
from py_wake.ground_models import Mirror
from py_wake.examples.data.hornsrev1 import Hornsrev1Site
from py_wake.examples.data.dtu10mw._dtu10mw import DTU10MW


class WakeModelFactory:
    """Factory class to create wind farm models with different configurations"""
    
    def __init__(self, model_type, downstream=True):
        self.model_type = model_type
        self.downstream = downstream
        self.site = Hornsrev1Site()
        self.turbine = DTU10MW()
        
    def create_wfm(self, params):
        """Create a wind farm model with given parameters"""
        
        if self.downstream:
            if self.model_type == 1:
                # BlondelSuperGaussianDeficit2020 for downstream
                def_args = {k: params[k] for k in ['a_s', 'b_s', 'c_s', 'b_f', 'c_f']}
                wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
                turb_args = {'c': np.array([params['ch1'], params['ch2'], 
                                           params['ch3'], params['ch4']])}
                blockage_args = {}
                
            else:  # model_type == 2
                # TurboGaussianDeficit for downstream
                wake_deficit_model = TurboGaussianDeficit(
                    A=params['A'],
                    cTI=[params['cti1'], params['cti2']],
                    ctlim=params['ctlim'],
                    ceps=params['ceps'],
                    ct2a=ct2a_mom1d,
                    groundModel=Mirror(),
                    rotorAvgModel=GaussianOverlapAvgModel()
                )
                wake_deficit_model.WS_key = 'WS_jlk'
                turb_args = {'c': np.array([params['ch1'], params['ch2'],
                                           params['ch3'], params['ch4']])}
                blockage_args = {}
                
        else:  # Upstream
            def_args = {}
            turb_args = {}
            blockage_args = {
                'ss_alpha': params['ss_alpha'],
                'ss_beta': params['ss_beta'],
                'r12p': np.array([params['rp1'], params['rp2']]),
                'ngp': np.array([params['ng1'], params['ng2'],
                                params['ng3'], params['ng4']])
            }
            
            if self.model_type == 2:
                blockage_args['groundModel'] = Mirror()
                
            wake_deficit_model = BlondelSuperGaussianDeficit2020(**def_args)
        
        # Create the wind farm model
        wfm = All2AllIterative(
            site=self.site,
            windTurbines=self.turbine,
            wake_deficitModel=wake_deficit_model,
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(**turb_args),
            blockage_deficitModel=SelfSimilarityDeficit2020(**blockage_args)
        )
        
        return wfm


class DataLoader:
    """Handles loading and preprocessing of reference data"""
    
    def __init__(self, data_path='./DTU10MW.nc'):
        self.data_path = data_path
        self.turbine = DTU10MW()
        self.D = self.turbine.diameter()
        
    def load_data(self, downstream=True):
        """Load and preprocess the reference data"""
        dat = xr.load_dataset(self.data_path)
        dat = dat.assign_coords(x=dat.x * self.D, y=dat.y * self.D)
        
        # Define region of interest
        if downstream:
            X_LB, X_UB = 2, 10
        else:
            X_LB, X_UB = -2, -1
            
        roi_x = slice(X_LB * self.D, X_UB * self.D)
        roi_y = slice(-2 * self.D, 2 * self.D)
        
        flow_roi = dat.sel(x=roi_x, y=roi_y)
        return flow_roi, X_LB, X_UB


class WakeOptimizer:
    """Main optimization class"""
    
    def __init__(self, model_factory, data_loader, flow_roi, full_ws, full_ti):
        self.model_factory = model_factory
        self.data_loader = data_loader
        self.flow_roi = flow_roi
        self.full_ws = full_ws
        self.full_ti = full_ti
        self.target_x = flow_roi.x
        self.target_y = flow_roi.y
        
        # Precompute observed deficits
        self._precompute_observed_deficits()
        
    def _precompute_observed_deficits(self):
        """Precompute observed deficits for all conditions"""
        self.obs_values = []
        
        # Create a dummy simulation to get CT values
        dummy_wfm = self.model_factory.create_wfm({})
        dummy_sim = dummy_wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                             wd=[270] * self.full_ti.size, time=True)
        
        for t in range(self.full_ws.size):
            this_pred_sim = dummy_sim.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            )
            self.obs_values.append(observed_deficit.T)
        
        self.all_obs = xr.concat(self.obs_values, dim='time')
    
    def evaluate_rmse(self, **kwargs):
        """Objective function for Bayesian optimization"""
        try:
            # Create wind farm model with current parameters
            wfm = self.model_factory.create_wfm(kwargs)
            
            # Run simulation
            sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                         wd=[270] * self.full_ti.size, time=True)
            
            # Compute flow map for all times at once (more efficient)
            flow_map = sim_res.flow_map(
                HorizontalGrid(x=self.target_x, y=self.target_y)
            )
            
            # Compute predicted deficits
            pred = (sim_res.WS - flow_map.WS_eff.isel(h=0)) / sim_res.WS
            
            # Calculate RMSE
            rmse = float(np.sqrt(((self.all_obs - pred) ** 2).mean(['x', 'y'])).mean('time'))
            
            if np.isnan(rmse):
                return -0.5
            return -rmse
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5
    
    def optimize(self, pbounds, defaults, init_points=50, n_iter=200):
        """Run Bayesian optimization"""
        optimizer = BayesianOptimization(
            f=self.evaluate_rmse,
            pbounds=pbounds,
            random_state=1
        )
        
        optimizer.probe(params=defaults, lazy=True)
        optimizer.maximize(init_points=init_points, n_iter=n_iter)
        
        self.optimizer = optimizer
        self.best_params = optimizer.max['params']
        self.best_rmse = -optimizer.max['target']
        
        return optimizer
    
    def create_animation(self, output_path):
        """Create optimization convergence animation"""
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Get best parameters up to current frame
            best_so_far_params = {}
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                if -self.optimizer.space.target[i] <= best_so_far_rmse:
                    best_so_far_rmse = -self.optimizer.space.target[i]
                    best_so_far_params = self.optimizer.res[i]['params']
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(self.optimizer.space.target), color='gray', alpha=0.5)
            ax1.plot(np.array(best_so_far_rmses), color='black')
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Plot parameter comparison
            keys = list(best_so_far_params.keys())
            best_vals = [best_so_far_params[k] for k in keys]
            default_vals = [self.defaults.get(k, 0) for k in keys]
            
            ax2.bar(keys, best_vals, label='Optimized')
            ax2.bar(keys, default_vals, edgecolor='black', linewidth=2, 
                   color='none', capstyle='butt', label='Default')
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.tick_params(axis='x', rotation=45)
            ax2.legend()
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(self.optimizer.space.target), 
            repeat=False
        )
        
        writer = animation.FFMpegWriter(fps=15)
        ani.save(output_path, writer=writer)
        plt.close('all')
    
    def analyze_results(self, output_dir='figs'):
        """Analyze and visualize optimization results"""
        import os
        os.makedirs(output_dir, exist_ok=True)
        
        # Create final model with best parameters
        wfm = self.model_factory.create_wfm(self.best_params)
        sim_res = wfm([0], [0], ws=self.full_ws, TI=self.full_ti,
                     wd=[270] * self.full_ti.size, time=True)
        
        flow_map = sim_res.flow_map(
            HorizontalGrid(x=self.target_x, y=self.target_y)
        )
        
        # Calculate RMSE for each time step
        rmse_values = []
        error_stats = []
        
        for t in range(self.full_ws.size):
            this_pred_sim = sim_res.isel(time=t)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_pred_sim.CT, ti=this_pred_sim.TI, z=0
            ).isel(wt=0)
            
            pred = (this_pred_sim.WS - flow_map.WS_eff.isel(h=0, time=t)) / this_pred_sim.WS
            diff = observed_deficit.T - pred
            
            # Calculate statistics
            rmse = np.sqrt(np.mean(diff**2))
            rmse_values.append(rmse)
            
            abs_errors = np.abs(diff.values.flatten())
            error_stats.append({
                'mean_error': np.mean(diff.values),
                'std_error': np.std(diff.values),
                'p90_error': np.percentile(abs_errors, 90),
                'max_error': np.max(abs_errors)
            })
            
            # Create detailed plots
            fig, axes = plt.subplots(3, 2, figsize=(12, 15))
            
            # Velocity deficit plots
            im0 = axes[0, 0].contourf(self.target_x, self.target_y, observed_deficit.T)
            axes[0, 0].set_title('Observed Velocity Deficit')
            axes[0, 0].set_ylabel('y/D')
            plt.colorbar(im0, ax=axes[0, 0])
            
            im1 = axes[1, 0].contourf(self.target_x, self.target_y, pred)
            axes[1, 0].set_title('Predicted Velocity Deficit')
            axes[1, 0].set_ylabel('y/D')
            plt.colorbar(im1, ax=axes[1, 0])
            
            im2 = axes[2, 0].contourf(self.target_x, self.target_y, diff)
            axes[2, 0].set_title('Error (Obs - Pred)')
            axes[2, 0].set_xlabel('x/D')
            axes[2, 0].set_ylabel('y/D')
            plt.colorbar(im2, ax=axes[2, 0])
            
            # Error distribution plot
            axes[0, 1].hist(diff.values.flatten(), bins=50, edgecolor='black')
            axes[0, 1].set_title('Error Distribution')
            axes[0, 1].set_xlabel('Error')
            axes[0, 1].set_ylabel('Frequency')
            
            # Statistics table
            stats_text = f"""Statistics for time step {t}:
            WS = {this_pred_sim.WS.values:.1f} m/s
            TI = {this_pred_sim.TI.values:.2f}
            CT = {this_pred_sim.CT.values:.3f}
            RMSE = {rmse:.4f}
            Mean Error = {error_stats[-1]['mean_error']:.4f}
            Std Error = {error_stats[-1]['std_error']:.4f}
            P90 Error = {error_stats[-1]['p90_error']:.4f}
            Max Error = {error_stats[-1]['max_error']:.4f}"""
            
            axes[1, 1].text(0.1, 0.5, stats_text, transform=axes[1, 1].transAxes,
                           verticalalignment='center', fontsize=10)
            axes[1, 1].axis('off')
            
            # Error along centerline
            centerline_idx = np.argmin(np.abs(self.target_y.values))
            axes[2, 1].plot(self.target_x, diff.isel(y=centerline_idx), 'b-', linewidth=2)
            axes[2, 1].axhline(y=0, color='r', linestyle='--', alpha=0.5)
            axes[2, 1].set_title('Error along Centerline (y=0)')
            axes[2, 1].set_xlabel('x/D')
            axes[2, 1].set_ylabel('Error')
            axes[2, 1].grid(True, alpha=0.3)
            
            plt.tight_layout()
            plt.savefig(f'{output_dir}/detailed_analysis_{t:03d}.png', dpi=150)
            plt.close()
        
        # Create summary statistics
        error_df = pd.DataFrame(error_stats)
        overall_stats = {
            'overall_rmse': np.mean(rmse_values),
            'mean_rmse': np.mean(rmse_values),
            'std_rmse': np.std(rmse_values),
            'mean_p90_error': np.mean(error_df['p90_error']),
            'mean_max_error': np.mean(error_df['max_error'])
        }
        
        print(f"RMSE values per time step: {rmse_values}")
        print(f"Overall RMSE: {overall_stats['overall_rmse']:.4f}")
        print(f"Average P90 Error: {overall_stats['mean_p90_error']:.4f}")
        print(f"Average Max Error: {overall_stats['mean_max_error']:.4f}")
        
        # Plot parameter comparison
        self._plot_parameter_comparison()
        
        return overall_stats, error_df
    
    def _plot_parameter_comparison(self):
        """Plot comparison between optimized and default parameters"""
        keys = list(self.best_params.keys())
        best_vals = [self.best_params[k] for k in keys]
        default_vals = [self.defaults.get(k, 0) for k in keys]
        
        plt.figure(figsize=(10, 6))
        x_pos = np.arange(len(keys))
        width = 0.35
        
        plt.bar(x_pos - width/2, best_vals, width, label='Optimized', alpha=0.8)
        plt.bar(x_pos + width/2, default_vals, width, label='Default', alpha=0.8)
        
        plt.xlabel('Parameters')
        plt.ylabel('Value')
        plt.title(f'Parameter Comparison (Best RMSE: {self.best_rmse:.4f})')
        plt.xticks(x_pos, keys, rotation=45)
        plt.legend()
        plt.grid(True, alpha=0.3)
        plt.tight_layout()
        plt.savefig('parameter_comparison.png', dpi=150)
        plt.close()
```
