Looking at your code, I can see several areas for improvement. The main issues are:

1. **Code organization**: The logic for creating different wake models is scattered and duplicated
2. **Flow field calculation**: You're calculating deficits incorrectly (using `WS_eff` instead of velocity deficits)
3. **Performance**: You're creating flow maps in a loop which is inefficient
4. **Readability**: The code mixes configuration, optimization, and visualization

Here's a modular rewrite that addresses these issues:

class WakeModelEvaluator:
    """Evaluator for wake model performance"""
    
    def __init__(self, config, flow_roi, turbine, site):
        self.config = config
        self.flow_roi = flow_roi
        self.turbine = turbine
        self.site = site
        
        # Prepare wind conditions
        self.full_ws, self.full_ti = self._prepare_wind_conditions()
        
        # Prepare reference data
        self.reference_data = self._prepare_reference_data()
    
    def _prepare_wind_conditions(self):
        """Prepare wind speed and TI arrays"""
        full_ti = np.tile(self.config.TIs, len(self.config.WSs))
        full_ws = np.repeat(self.config.WSs, len(self.config.TIs))
        return full_ws, full_ti
    
    def _prepare_reference_data(self):
        """Extract reference deficit data"""
        obs_values = []
        
        # Create a dummy simulation to get CT and TI values
        dummy_wfm = All2AllIterative(
            self.site, self.turbine,
            wake_deficitModel=BlondelSuperGaussianDeficit2020(),
            superpositionModel=LinearSum(),
            deflectionModel=None,
            turbulenceModel=CrespoHernandez(),
            blockage_deficitModel=SelfSimilarityDeficit2020()
        )
        
        dummy_res = dummy_wfm(
            [0], [0], 
            ws=self.full_ws, 
            TI=self.full_ti, 
            wd=[270] * len(self.full_ws),
            time=True
        )
        
        # Extract reference deficits for each condition
        for t in range(len(self.full_ws)):
            this_sim = dummy_res.isel(time=t, wt=0)
            observed_deficit = self.flow_roi.deficits.interp(
                ct=this_sim.CT, 
                ti=this_sim.TI, 
                z=0
            )
            obs_values.append(observed_deficit.T)
        
        return xr.concat(obs_values, dim='time')
    
    def calculate_deficit(self, flow_map, sim_res):
        """Calculate velocity deficit from flow map"""
        # IMPORTANT: flow_map.WS_eff is the effective wind speed
        # Velocity deficit = (U_inf - U_wake) / U_inf
        # Where U_inf = sim_res.WS (free stream) and U_wake = flow_map.WS_eff
        return (sim_res.WS - flow_map) / sim_res.WS
    
    def evaluate(self, params):
        """Evaluate model with given parameters"""
        try:
            # Create model configuration
            model_config = WakeModelFactory.create_model(self.config, params)
            
            # Create wind farm model
            wfm = All2AllIterative(
                self.site,
                self.turbine,
                **model_config
            )
            
            # Run simulation
            sim_res = wfm(
                [0], [0],
                ws=self.full_ws,
                TI=self.full_ti,
                wd=[270] * len(self.full_ws),
                time=True
            )
            
            # Create flow map for all time steps at once (more efficient)
            flow_map = sim_res.flow_map(
                HorizontalGrid(x=self.flow_roi.x, y=self.flow_roi.y)
            )['WS_eff']
            
            # Calculate predicted deficits
            pred_deficits = self.calculate_deficit(flow_map, sim_res)
            
            # Calculate RMSE
            rmse = float(np.sqrt(
                ((self.reference_data - pred_deficits) ** 2).mean(['x', 'y'])
            ).mean('time'))
            
            if np.isnan(rmse):
                return -0.5
            return -rmse
            
        except Exception as e:
            print(f"Error in evaluation: {e}")
            return -0.5

class Visualization:
    """Visualization utilities"""
    
    @staticmethod
    def create_optimization_animation(optimizer, defaults, config):
        """Create animation of optimization progress"""
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        def update_plot(frame):
            ax1.clear()
            ax2.clear()
            
            # Track best RMSE so far
            best_so_far_rmse = float('inf')
            best_so_far_rmses = []
            
            for i in range(frame + 1):
                current_rmse = -optimizer.space.target[i]
                if current_rmse <= best_so_far_rmse:
                    best_so_far_rmse = current_rmse
                best_so_far_rmses.append(best_so_far_rmse)
            
            # Plot convergence
            ax1.plot(-np.array(optimizer.space.target), color='gray', alpha=0.5)
            ax1.plot(best_so_far_rmses, color='black', linewidth=2)
            ax1.set_title('Optimization Convergence')
            ax1.set_xlabel('Iteration')
            ax1.set_ylabel('RMSE')
            ax1.grid(True)
            
            # Get best parameters so far
            best_idx = np.argmin(-optimizer.space.target[:frame+1])
            best_params = optimizer.res[best_idx]['params']
            
            # Plot parameter comparison
            keys = list(best_params.keys())
            best_vals = [best_params[k] for k in keys]
            default_vals = [defaults.get(k, 0) for k in keys]
            
            x_pos = np.arange(len(keys))
            ax2.bar(x_pos - 0.2, best_vals, width=0.4, label='Optimized')
            ax2.bar(x_pos + 0.2, default_vals, width=0.4, label='Default', 
                   edgecolor='black', linewidth=2, facecolor='none')
            
            ax2.set_xticks(x_pos)
            ax2.set_xticklabels(keys, rotation=45)
            ax2.set_title(f'Best RMSE: {best_so_far_rmse:.4f}')
            ax2.legend()
            ax2.grid(True, alpha=0.3)
            
            plt.tight_layout()
            return ax1, ax2
        
        ani = animation.FuncAnimation(
            fig, update_plot, 
            frames=len(optimizer.space.target), 
            repeat=False
        )
        
        # Save animation
        writer = animation.FFMpegWriter(fps=15)
        ani.save(f'optimization_animation_{config.X_LB}_{config.X_UB}.mp4', writer=writer)
        plt.close(fig)
    
    @staticmethod

        axes[1].set_title('Predicted Deficit')
        axes[1].set_xlabel('x/D')
        plt.colorbar(im2, ax=axes[1])
        
        # Plot difference
        im3 = axes[2].contourf(target_x, target_y, diff, levels=20, cmap='RdBu_r')
        axes[2].set_title(f'Difference (RMSE: {rmse:.4f})')

        plt.colorbar(im3, ax=axes[2])
        
        plt.suptitle(f'Time Step {t}')

def main():
    # Initialize configuration
    config = Config()
    
    # Validate model number
    if config.MODEL not in {1, 2}:
        raise ValueError("MODEL must be 1 or 2")
    
    # Load data
    flow_roi, turbine, D = load_and_prepare_data(config)
    site = Hornsrev1Site()
    
    # Create evaluator
    evaluator = WakeModelEvaluator(config, flow_roi, turbine, site)
    
    # Get parameter bounds and defaults
    param_manager = ParameterManager()
    pbounds = param_manager.get_parameter_bounds(config)
    defaults = param_manager.get_defaults(config)
    
    # Setup optimization
    optimizer = BayesianOptimization(
        f=evaluator.evaluate,
        pbounds=pbounds,
        random_state=1,
        verbose=2
    )
    
    # Add default point
    optimizer.probe(params=defaults, lazy=True)
    
    # Run optimization
    print(f"Starting optimization with {config.INIT_POINTS} initial points and {config.N_ITER} iterations")
    optimizer.maximize(
        init_points=config.INIT_POINTS,
        n_iter=config.N_ITER
    )
    
    # Get best results
    best_params = optimizer.max['params']
    best_rmse = -optimizer.max['target']
    print(f"\nBest RMSE: {best_rmse:.4f}")
    print("Best parameters:", best_params)
    
    # Create visualization of optimization
    Visualization.create_optimization_animation(optimizer, defaults, config)
    
    # Create final model with best parameters
    model_config = WakeModelFactory.create_model(config, best_params)
    wfm = All2AllIterative(site, turbine, **model_config)
    
    # Run final simulation
    sim_res = wfm(
        [0], [0],
        ws=evaluator.full_ws,
        TI=evaluator.full_ti,
        wd=[270] * len(evaluator.full_ws),
        time=True
    )
    
    # Calculate flow map
    flow_map = sim_res.flow_map(
        HorizontalGrid(x=flow_roi.x, y=flow_roi.y)
    )['WS_eff']
    
    # Calculate and analyze errors
    rmse_values = []
    
    for t in range(len(evaluator.full_ws)):
        # Get observed and predicted deficits
        observed = evaluator.reference_data.isel(time=t).values
        predicted = evaluator.calculate_deficit(
            flow_map.isel(time=t), 
            sim_res.isel(time=t)
        ).values
        
        # Calculate RMSE for this condition
        diff = observed - predicted
        rmse = np.sqrt(np.mean(diff**2))
        rmse_values.append(rmse)
        
        # Plot comparison
        Visualization.plot_flow_field_comparison(
            observed, predicted, diff,
            flow_roi.x.values, flow_roi.y.values,
            t, rmse
        )
    
    # Plot error statistics
    Visualization.plot_error_statistics(rmse_values, config)
    
    # Print summary
    print(f"\nError Statistics:")
    print(f"  Average RMSE: {np.mean(rmse_values):.4f}")
    print(f"  P90 RMSE: {np.percentile(rmse_values, 90):.4f}")
    print(f"  Min RMSE: {np.min(rmse_values):.4f}")
    print(f"  Max RMSE: {np.max(rmse_values):.4f}")
    
    # Plot final parameter comparison
    fig, ax = plt.subplots(figsize=(10, 6))
    keys = list(best_params.keys())
    best_vals = [best_params[k] for k in keys]
    default_vals = [defaults.get(k, 0) for k in keys]
    
    x_pos = np.arange(len(keys))
    ax.bar(x_pos - 0.2, best_vals, width=0.4, label='Optimized')
    ax.bar(x_pos + 0.2, default_vals, width=0.4, label='Default',
           edgecolor='black', linewidth=2, facecolor='none')
    
    ax.set_xticks(x_pos)
    ax.set_xticklabels(keys, rotation=45)
    ax.set_title(f'Parameter Comparison (Best RMSE: {best_rmse:.4f})')
    ax.legend()
    ax.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.savefig(f'parameter_comparison_{config.X_LB}_{config.X_UB}.png', dpi=150)
    plt.close()


if __name__ == "__main__":
    main()
