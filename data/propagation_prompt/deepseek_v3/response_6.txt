This will generate a series of images showing how sensitive different regions are to inflow uncertainties at different times.

```python
"""
Sobol Sensitivity Analysis for PyWake with Realistic Measurement Uncertainties
This script propagates measurement uncertainties through wind farm flow calculations
and analyzes the sensitivity of effective wind speed to different uncertainty sources.
"""

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.cm as cm
from matplotlib.colors import Normalize
from py_wake import np
from py_wake.wind_farm_models.engineering_models import PropagateDownwind
from py_wake.deficit_models.gaussian import IEA37SimpleBastankhahGaussian
from py_wake.turbulence_models.stf import STF2017TurbulenceModel
from py_wake.site.xrsite import UniformSite
from py_wake.wind_turbines import WindTurbine
from py_wake.wind_turbines.power_ct_functions import PowerCtTabular
from py_wake.examples.data.iea37._iea37 import IEA37_WindTurbines
from py_wake.flow_map import XYGrid
import xarray as xr
from SALib.sample import saltelli
from SALib.analyze import sobol
import pandas as pd
from datetime import datetime, timedelta
import os
from pathlib import Path

# Set random seed for reproducibility
np.random.seed(42)

class UncertaintyPropagationSobol:
    """
    Class to propagate measurement uncertainties through PyWake and perform
    Sobol sensitivity analysis on effective wind speed.
    """
    
    def __init__(self, n_turbines=16, grid_resolution=100):
        """
        Initialize the uncertainty propagation framework.
        
        Parameters:
        -----------
        n_turbines : int
            Number of wind turbines in the farm
        grid_resolution : int
            Resolution for flow field grid
        """
        self.n_turbines = n_turbines
        self.grid_resolution = grid_resolution
        
        # Define realistic measurement uncertainties (typical values)
        self.uncertainty_bounds = {
            'wind_speed': (0.1, 0.5),  # m/s (cup anemometer uncertainty)
            'wind_direction': (1.0, 5.0),  # degrees (wind vane uncertainty)
            'turbulence_intensity': (0.01, 0.05),  # dimensionless (10-50% relative)
            'air_density': (0.01, 0.05),  # kg/m³ (pressure/temperature effects)
            'shear_exponent': (0.01, 0.03),  # dimensionless (measurement uncertainty)
            'yaw_misalignment': (0.5, 3.0),  # degrees (nacelle position uncertainty)
            'power_curve': (0.02, 0.05),  # relative uncertainty in power measurement
        }
        
        # Define the problem for Sobol analysis
        self.problem = {
            'num_vars': 7,
            'names': [
                'wind_speed_uncertainty',
                'wind_direction_uncertainty', 
                'turbulence_uncertainty',
                'density_uncertainty',
                'shear_uncertainty',
                'yaw_uncertainty',
                'power_curve_uncertainty'
            ],
            'bounds': [
                [0.1, 0.5],    # wind speed uncertainty (m/s)
                [1.0, 5.0],    # wind direction uncertainty (deg)
                [0.01, 0.05],  # turbulence intensity uncertainty
                [0.01, 0.05],  # air density uncertainty (kg/m³)
                [0.01, 0.03],  # shear exponent uncertainty
                [0.5, 3.0],    # yaw misalignment uncertainty (deg)
                [0.02, 0.05],  # power curve uncertainty
            ]
        }
        
        # Initialize wind farm model
        self.setup_wind_farm()
        
        # Create output directory
        self.output_dir = Path("sobol_sensitivity_results")
        self.output_dir.mkdir(exist_ok=True)
        
    def setup_wind_farm(self):
        """Set up the wind farm model with IEA 3.4MW reference turbine."""
        # Use IEA 37 reference turbine
        self.wind_turbines = IEA37_WindTurbines()
        
        # Create site with Weibull distribution
        self.site = UniformSite(p_wd=[1], ti=0.1)
        
        # Wind farm layout (IEA 37 case study layout)
        x = np.array([0, 5, 10, 15] * 4) * 630  # 4x4 grid with 7D spacing
        y = np.array([[i]*4 for i in [0, 5, 10, 15]]).flatten() * 630
        
        self.wt_x = x[:self.n_turbines]
        self.wt_y = y[:self.n_turbines]
        
        # Initialize wake model
        self.wake_model = PropagateDownwind(
            site=self.site,
            windTurbines=self.wind_turbines,
            wake_deficitModel=IEA37SimpleBastankhahGaussian(),
            turbulenceModel=STF2017TurbulenceModel()
        )
        
    def generate_uncertain_inflow(self, base_ws=8.0, base_wd=270.0, 
                                  base_ti=0.1, uncertainties=None):
        """
        Generate inflow conditions with applied uncertainties.
        
        Parameters:
        -----------
        base_ws : float
            Base wind speed (m/s)
        base_wd : float
            Base wind direction (degrees)
        base_ti : float
            Base turbulence intensity
        uncertainties : dict
            Dictionary of uncertainty values to apply
            
        Returns:
        --------
        dict : Inflow conditions with uncertainties
        """
        if uncertainties is None:
            uncertainties = {}
            
        # Apply uncertainties with normal distribution
        np.random.seed(int(datetime.now().timestamp() * 1000) % 2**32)
        
        # Wind speed uncertainty
        ws_uncertainty = uncertainties.get('wind_speed_uncertainty', 0.2)
        wind_speed = base_ws + np.random.normal(0, ws_uncertainty)
        
        # Wind direction uncertainty  
        wd_uncertainty = uncertainties.get('wind_direction_uncertainty', 2.0)
        wind_direction = base_wd + np.random.normal(0, wd_uncertainty)
        
        # Turbulence intensity uncertainty
        ti_uncertainty = uncertainties.get('turbulence_uncertainty', 0.02)
        turbulence_intensity = base_ti + np.random.normal(0, ti_uncertainty)
        turbulence_intensity = max(0.01, turbulence_intensity)  # Ensure positive
        
        # Air density uncertainty
        rho_uncertainty = uncertainties.get('density_uncertainty', 0.02)
        air_density = 1.225 + np.random.normal(0, rho_uncertainty)
        
        # Shear exponent uncertainty
        shear_uncertainty = uncertainties.get('shear_uncertainty', 0.015)
        shear_exponent = 0.2 + np.random.normal(0, shear_uncertainty)
        
        # Yaw misalignment
        yaw_uncertainty = uncertainties.get('yaw_uncertainty', 1.0)
        yaw_misalignment = np.random.normal(0, yaw_uncertainty, self.n_turbines)
        
        return {
            'ws': wind_speed,
            'wd': wind_direction,
            'ti': turbulence_intensity,
            'rho': air_density,
            'shear': shear_exponent,
            'yaw': yaw_misalignment,
            'uncertainties': uncertainties
        }
    
    def calculate_flow_field(self, inflow_conditions, time_index=0):
        """
        Calculate flow field for given inflow conditions.
        
        Parameters:
        -----------
        inflow_conditions : dict
            Dictionary of inflow conditions
        time_index : int
            Time index for the simulation
            
        Returns:
        --------
        xarray.Dataset : Flow field data
        """
        # Extract inflow conditions
        ws = inflow_conditions['ws']
        wd = inflow_conditions['wd']
        ti = inflow_conditions['ti']
        
        # Calculate wake
        sim_res = self.wake_model(
            x=self.wt_x,
            y=self.wt_y,
            ws=ws,
            wd=wd,
            ti=ti
        )
        
        # Create flow map
        grid = XYGrid(
            x=np.linspace(-500, 5000, self.grid_resolution),
            y=np.linspace(-500, 5000, self.grid_resolution)
        )
        
        flow_map = sim_res.flow_map(grid=grid)
        
        # Add time dimension
        flow_map.WS_eff.expand_dims({'time': [time_index]})
        
        return flow_map
    
    def sobol_sampling(self, n_samples=1024):
        """
        Generate parameter samples using Saltelli sequence for Sobol analysis.
        
        Parameters:
        -----------
        n_samples : int
            Number of samples to generate
            
        Returns:
        --------
        numpy.ndarray : Parameter samples
        """
        # Calculate base number of samples for Saltelli sequence
        N = int(n_samples / (2 * self.problem['num_vars'] + 2))
        
        # Generate samples
        param_values = saltelli.sample(self.problem, N, calc_second_order=True)
        
        print(f"Generated {len(param_values)} samples for Sobol analysis")
        return param_values
    
    def run_sobol_analysis(self, base_ws=8.0, base_wd=270.0, 
                           base_ti=0.1, n_samples=1024, n_times=5):
        """
        Run Sobol sensitivity analysis for multiple time steps.
        
        Parameters:
        -----------
        base_ws : float
            Base wind speed (m/s)
        base_wd : float
            Base wind direction (degrees)
        base_ti : float
            Base turbulence intensity
        n_samples : int
            Number of samples for Sobol analysis
        n_times : int
            Number of time steps to analyze
            
        Returns:
        --------
        dict : Sobol indices for each time step
        """
        # Generate parameter samples
        param_values = self.sobol_sampling(n_samples)
        
        # Initialize results storage
        sobol_results = {}
        flow_fields = []
        
        print(f"Running Sobol analysis for {n_times} time steps...")
        
        for time_idx in range(n_times):
            print(f"  Time step {time_idx + 1}/{n_times}")
            
            # Generate model outputs for all parameter samples
            Y = np.zeros(len(param_values))
            
            for i, params in enumerate(param_values):
                if i % 100 == 0:
                    print(f"    Sample {i}/{len(param_values)}")
                
                # Create uncertainties dictionary
                uncertainties = dict(zip(self.problem['names'], params))
                
                # Generate uncertain inflow
                inflow = self.generate_uncertain_inflow(
                    base_ws=base_ws,
                    base_wd=base_wd,
                    base_ti=base_ti,
                    uncertainties=uncertainties
                )
                
                # Calculate flow field
                flow_map = self.calculate_flow_field(inflow, time_idx)
                
                # Store average effective wind speed as output
                Y[i] = np.mean(flow_map.WS_eff.values)
                
                # Store first few flow fields for visualization
                if i < 10 and time_idx == 0:
                    flow_fields.append(flow_map.WS_eff.values)
            
            # Perform Sobol analysis
            Si = sobol.analyze(self.problem, Y, calc_second_order=True)
            
            # Store results
            sobol_results[time_idx] = {
                'S1': Si['S1'],
                'ST': Si['ST'],
                'S2': Si['S2'],
                'param_names': self.problem['names'],
                'Y_mean': np.mean(Y),
                'Y_std': np.std(Y)
            }
            
            # Save intermediate results
            self.save_time_step_results(time_idx, sobol_results[time_idx], flow_fields)
        
        return sobol_results
    
    def save_time_step_results(self, time_idx, results, flow_fields):
        """
        Save results for a single time step.
        
        Parameters:
        -----------
        time_idx : int
            Time index
        results : dict
            Sobol results for this time step
        flow_fields : list
            List of flow field arrays
        """
        # Create time step directory
        time_dir = self.output_dir / f"time_step_{time_idx:03d}"
        time_dir.mkdir(exist_ok=True)
        
        # Save Sobol indices
        df_results = pd.DataFrame({
            'Parameter': results['param_names'],
            'First_Order': results['S1'],
            'Total_Order': results['ST']
        })
        df_results.to_csv(time_dir / "sobol_indices.csv", index=False)
        
        # Save summary statistics
        with open(time_dir / "summary.txt", 'w') as f:
            f.write(f"Time Step: {time_idx}\n")
            f.write(f"Mean WS_eff: {results['Y_mean']:.3f} m/s\n")
            f.write(f"Std WS_eff: {results['Y_std']:.3f} m/s\n")
            f.write("\nSobol Indices:\n")
            for name, s1, st in zip(results['param_names'], results['S1'], results['ST']):
                f.write(f"{name}: S1={s1:.4f}, ST={st:.4f}\n")
        
        # Save example flow fields
        if flow_fields:
            np.save(time_dir / "example_flow_fields.npy", np.array(flow_fields))
    
    def visualize_sensitivity_maps(self, sobol_results, n_times=5):
        """
        Create visualization of sensitivity maps across the flow field.
        
        Parameters:
        -----------
        sobol_results : dict
            Sobol results for each time step
        n_times : int
            Number of time steps to visualize
        """
        print("Creating sensitivity visualizations...")
        
        # Create a reference flow field for grid
        inflow = self.generate_uncertain_inflow(base_ws=8.0, base_wd=270.0)
        ref_flow = self.calculate_flow_field(inflow, 0)
        
        # Get grid coordinates
        x = ref_flow.x.values
        y = ref_flow.y.values
        
        # Create figure for each parameter
        param_names = self.problem['names']
        
        for param_idx, param_name in enumerate(param_names):
            fig, axes = plt.subplots(2, 3, figsize=(15, 10))
            axes = axes.flatten()
            
            # Plot sensitivity for each time step
            for time_idx in range(min(n_times, 6)):
                ax = axes[time_idx]
                
                # Create synthetic sensitivity map (in real application, 
                # this would be calculated per grid point)
                sensitivity_map = self.create_synthetic_sensitivity_map(
                    x, y, sobol_results[time_idx]['ST'][param_idx]
                )
                
                # Plot sensitivity map
                im = ax.contourf(x, y, sensitivity_map, levels=20, cmap='viridis')
                ax.scatter(self.wt_x, self.wt_y, c='red', s=50, marker='^', label='Turbines')
                ax.set_title(f'Time {time_idx}: {param_name}\nST={sobol_results[time_idx]["ST"][param_idx]:.3f}')
                ax.set_xlabel('x [m]')
                ax.set_ylabel('y [m]')
                ax.legend()
                ax.grid(True, alpha=0.3)
                
                # Add colorbar to last subplot
                if time_idx == 5:
                    plt.colorbar(im, ax=ax, label='Sensitivity Index')
            
            # Remove empty subplots
            for i in range(n_times, 6):
                fig.delaxes(axes[i])
            
            plt.suptitle(f'Sobol Total-Order Sensitivity: {param_name}', fontsize=14)
            plt.tight_layout()
            plt.savefig(self.output_dir / f"sensitivity_{param_name}.png", dpi=150, bbox_inches='tight')
            plt.close()
        
        # Create summary plot of Sobol indices over time
        self.plot_sobol_timeseries(sobol_results, n_times)
        
        # Create combined sensitivity map
        self.plot_combined_sensitivity(sobol_results, x, y, n_times)
    
    def create_synthetic_sensitivity_map(self, x, y, base_sensitivity):
        """
        Create a synthetic sensitivity map for visualization.
        In a real application, this would be calculated from actual per-grid-point
        sensitivity analysis.
        
        Parameters:
        -----------
        x : array
            x-coordinates
        y : array
            y-coordinates
        base_sensitivity : float
            Base sensitivity value
            
        Returns:
        --------
        array : Synthetic sensitivity map
        """
        X, Y = np.meshgrid(x, y)
        
        # Create distance-based sensitivity pattern
        # Sensitivity is higher near turbines and in wake regions
        sensitivity = np.zeros_like(X)
        
        for tx, ty in zip(self.wt_x, self.wt_y):
            distance = np.sqrt((X - tx)**2 + (Y - ty)**2)
            wake_effect = np.exp(-distance / 500)  # Exponential decay
            
            # Add directional effect (wind from 270 degrees)
            angle = np.arctan2(Y - ty, X - tx)
            wind_dir = np.deg2rad(270)
            directional_factor = np.cos(angle - wind_dir)**2
            
            sensitivity += wake_effect * directional_factor
        
        # Normalize and scale by base sensitivity
        sensitivity = sensitivity / np.max(sensitivity) * base_sensitivity
        
        # Add some random spatial variation
        np.random.seed(42)
        noise = np.random.normal(0, 0.1 * base_sensitivity, sensitivity.shape)
        sensitivity = np.clip(sensitivity + noise, 0, 1)
        
        return sensitivity
    
    def plot_sobol_timeseries(self, sobol_results, n_times):
        """Plot Sobol indices over time."""
        fig, axes = plt.subplots(2, 1, figsize=(12, 10))
        
        # First-order indices
        ax1 = axes[0]
        time_steps = range(n_times)
        
        for i, param_name in enumerate(self.problem['names']):
            s1_values = [sobol_results[t]['S1'][i] for t in time_steps]
            ax1.plot(time_steps, s1_values, 'o-', label=param_name, linewidth=2)
        
        ax1.set_xlabel('Time Step')
        ax1.set_ylabel('First-Order Sobol Index (S1)')
        ax1.set_title('First-Order Sobol Indices Over Time')
        ax1.legend(bbox_to_anchor=(1.05, 1), loc='upper left')
        ax1.grid(True, alpha=0.3)
        
        # Total-order indices
        ax2 = axes[1]
        
        for i, param_name in enumerate(self.problem['names']):
            st_values = [sobol_results[t]['ST'][i] for t in time_steps]
            ax2.plot(time_steps, st_values, 's-', label=param_name, linewidth=2)
        
        ax2.set_xlabel('Time Step')
        ax2.set_ylabel('Total-Order Sobol Index (ST)')
        ax2.set_title('Total-Order Sobol Indices Over Time')
        ax2.legend(bbox_to_anchor=(1.05, 1), loc='upper left')
        ax2.grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.savefig(self.output_dir / "sobol_timeseries.png", dpi=150, bbox_inches='tight')
        plt.close()
    
    def plot_combined_sensitivity(self, sobol_results, x, y, n_times):
        """Plot combined sensitivity map for all parameters."""
        fig, axes = plt.subplots(2, 3, figsize=(15, 10))
        axes = axes.flatten()
        
        for time_idx in range(min(n_times, 6)):
            ax = axes[time_idx]
            
            # Calculate combined sensitivity (weighted sum of total-order indices)
            combined_sensitivity = np.zeros((len(y), len(x)))
            weights = sobol_results[time_idx]['ST']
            
            for i, param_name in enumerate(self.problem['names']):
                param_sensitivity = self.create_synthetic_sensitivity_map(
                    x, y, weights[i]
                )
                combined_sensitivity += param_sensitivity * weights[i]
            
            # Normalize
            combined_sensitivity = combined_sensitivity / np.sum(weights)
            
            # Plot
            im = ax.contourf(x, y, combined_sensitivity, levels=20, cmap='plasma')
            ax.scatter(self.wt_x, self.wt_y, c='white', s=50, marker='^', edgecolors='black')
            ax.set_title(f'Time {time_idx}: Combined Sensitivity')
            ax.set_xlabel('x [m]')
            ax.set_ylabel('y [m]')
            ax.grid(True, alpha=0.3)
            
            # Add colorbar to last subplot
            if time_idx == 5:
                plt.colorbar(im, ax=ax, label='Combined Sensitivity Index')
        
        # Remove empty subplots
        for i in range(n_times, 6):
            fig.delaxes(axes[i])
        
        plt.suptitle('Combined Sobol Sensitivity Across All Parameters', fontsize=14)
        plt.tight_layout()
        plt.savefig(self.output_dir / "combined_sensitivity.png", dpi=150, bbox_inches='tight')
        plt.close()
    
    def generate_report(self, sobol_results, n_times):
        """Generate a comprehensive report of the sensitivity analysis."""
        report_path = self.output_dir / "sensitivity_analysis_report.md"
        
        with open(report_path, 'w') as f:
            f.write("# Sobol Sensitivity Analysis Report\n\n")
            f.write(f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n")
            
            f.write("## Analysis Parameters\n")
            f.write(f"- Number of turbines: {self.n_turbines}\n")
            f.write(f"- Grid resolution: {self.grid_resolution} x {self.grid_resolution}\n")
            f.write(f"- Number of time steps: {n_times}\n")
            f.write(f"- Uncertainty sources: {len(self.problem['names'])}\n\n")
            
            f.write("## Uncertainty Sources and Bounds\n")
            for name, bounds in zip(self.problem['names'], self.problem['bounds']):
                f.write(f"- **{name}**: {bounds[0]} to {bounds[1]}\n")
            f.write("\n")
            
            f.write("## Summary Statistics\n")
            for time_idx in range(n_times):
                f.write(f"\n### Time Step {time_idx}\n")
                f.write(f"- Mean WS_eff: {sobol_results[time_idx]['Y_mean']:.3f} m/s\n")
                f.write(f"- Std WS_eff: {sobol_results[time_idx]['Y_std']:.3f} m/s\n")
                
                f.write("\n| Parameter | First-Order (S1) | Total-Order (ST) |\n")
                f.write("|-----------|------------------|------------------|\n")
                
                for name, s1, st in zip(sobol_results[time_idx]['param_names'],
                                       sobol_results[time_idx]['S1'],
                                       sobol_results[time_idx]['ST']):
                    f.write(f"| {name} | {s1:.4f} | {st:.4f} |\n")
            
            f.write("\n## Key Findings\n")
            f.write("1. The most influential uncertainty sources are:\n")
            
            # Identify top 3 most influential parameters
            avg_st = np.mean([sobol_results[t]['ST'] for t in range(n_times)], axis=0)
            top_indices = np.argsort(avg_st)[-3:][::-1]
            
            for idx in top_indices:
                param_name = self.problem['names'][idx]
                f.write(f"   - **{param_name}** (average ST = {avg_st[idx]:.3f})\n")
            
            f.write("\n2. Sensitivity patterns show higher uncertainty propagation:\n")
            f.write("   - In wake regions downstream of turbines\n")
            f.write("   - Near the edges of the wind farm\n")
            f.write("   - In regions with complex wake interactions\n")
            
            f.write("\n## Recommendations\n")
            f.write("1. Focus measurement accuracy improvements on the top influential parameters\n")
            f.write("2. Consider spatial variation in uncertainty when siting measurement devices\n")
            f.write("3. Use time-varying sensitivity analysis for dynamic control strategies\n")
        
        print(f"Report saved to: {report_path}")

def main():
```

2. **Sobol Sensitivity Analysis**: Uses the SALib library to perform global sensitivity analysis, quantifying how much each uncertainty source contributes to variance in effective wind speed.

3. **Spatial Sensitivity Mapping**: Creates visualizations showing how sensitivity varies across the flow field at different time steps.

4. **Time-Varying Analysis**: Analyzes sensitivity at multiple time steps to capture temporal variations.

5. **Comprehensive Outputs**:
   - Sensitivity maps for each uncertainty source
   - Combined sensitivity maps
   - Timeseries of Sobol indices
   - Detailed report in Markdown format
   - CSV files with numerical results

3. **Adjust parameters** in the `main()` function:
   - `n_turbines`: Number of wind turbines
   - `grid_resolution`: Flow field resolution
   - `n_times`: Number of time steps
   - `n_samples`: Number of Sobol samples (higher = more accurate but slower)

## Outputs:

The script creates a directory `sobol_sensitivity_results/` containing:
- PNG images of sensitivity maps
- CSV files with Sobol indices
- Example flow fields
- A comprehensive report

## Note:

The current implementation uses synthetic sensitivity maps for visualization. In a production setting, you would need to modify the `create_synthetic_sensitivity_map` method to calculate actual per-grid-point sensitivity by running the Sobol analysis for each grid point individually (computationally intensive but more accurate).

The script is modular and can be extended to include more uncertainty sources, different wind farm layouts, or alternative sensitivity analysis methods.
