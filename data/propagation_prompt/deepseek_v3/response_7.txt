I'll create a script that propagates realistic measurement uncertainties through PyWake and performs Sobol sensitivity analysis on the effective wind speed (WS_eff) across the flow field. This will generate a series of images showing how sensitive different regions of the flow field are to inflow uncertainties at different times.

```python
"""
Sobol Sensitivity Analysis of Wind Farm Flow Field to Inflow Uncertainties
This script propagates realistic measurement uncertainties through PyWake and
analyzes the sensitivity of effective wind speed (WS_eff) across the flow field.
"""

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.cm as cm
from matplotlib.colors import Normalize
from pathlib import Path
import time
from datetime import datetime
import json
import warnings
warnings.filterwarnings('ignore')

# PyWake imports
from py_wake import np
from py_wake.wind_turbines import WindTurbine
from py_wake.wind_turbines.power_ct_functions import PowerCtTabular
from py_wake.site import UniformSite
from py_wake import BastankhahGaussian
from py_wake.examples.data import hornsrev1
from py_wake.flow_map import HorizontalGrid

# Sensitivity analysis imports
from SALib.sample import saltelli
from SALib.analyze import sobol

# Set random seed for reproducibility
np.random.seed(42)

class UncertaintyPropagationSensitivity:
    """
    Class to propagate measurement uncertainties through PyWake and perform
    Sobol sensitivity analysis on the flow field.
    """
    
    def __init__(self, output_dir="sensitivity_results"):
        """
        Initialize the sensitivity analysis framework.
        
        Parameters:
        -----------
        output_dir : str
            Directory to save results and images
        """
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(exist_ok=True)
        
        # Create subdirectories
        self.flowfield_dir = self.output_dir / "flow_fields"
        self.sensitivity_dir = self.output_dir / "sensitivity_maps"
        self.flowfield_dir.mkdir(exist_ok=True)
        self.sensitivity_dir.mkdir(exist_ok=True)
        
        # Initialize wind farm model
        self.setup_wind_farm()
        
        # Define uncertainty parameters with realistic measurement uncertainties
        self.define_uncertainty_parameters()
        
        # Setup flow field grid
        self.setup_flow_field_grid()
        
        print(f"Initialized sensitivity analysis framework")
        print(f"Results will be saved to: {self.output_dir}")
    
    def setup_wind_farm(self):
        """Setup a realistic wind farm scenario."""
        print("Setting up wind farm model...")
        
        # Define wind turbine characteristics (based on a generic 2MW turbine)
        power_curve = np.array([[3, 0], [4, 75], [5, 150], [6, 300], 
                                [7, 500], [8, 800], [9, 1200], [10, 1600],
                                [11, 1900], [12, 2000], [13, 2000], 
                                [14, 2000], [15, 2000], [20, 2000], [25, 0]])
        
        ct_curve = np.array([[3, 0.85], [4, 0.85], [5, 0.85], [6, 0.85],
                             [7, 0.84], [8, 0.83], [9, 0.81], [10, 0.78],
                             [11, 0.74], [12, 0.70], [13, 0.65], 
                             [14, 0.60], [15, 0.55], [20, 0.30], [25, 0.0]])
        
        # Create wind turbine
        self.wt = WindTurbine(
            name='Generic2MW',
            diameter=80,
            hub_height=70,
            powerCtFunction=PowerCtTabular(power_curve[:, 0], 
                                          power_curve[:, 1], 
                                          'kW', ct_curve[:, 1])
        )
        
        # Setup site with Weibull distribution
        self.site = UniformSite(p_wd=[1], ti=0.1)
        
        # Setup wind farm layout (3x3 grid)
        self.x = np.array([0, 200, 400, 0, 200, 400, 0, 200, 400])
        self.y = np.array([0, 0, 0, 200, 200, 200, 400, 400, 400])
        
        # Initialize wake model
        self.wake_model = BastankhahGaussian(self.site, self.wt)
        
        print(f"Wind farm setup complete: {len(self.x)} turbines")
    
    def define_uncertainty_parameters(self):
        """Define realistic measurement uncertainties for inflow parameters."""
        print("Defining uncertainty parameters...")
        
        # Define problem for Sobol analysis
        self.problem = {
            'num_vars': 5,
            'names': [
                'wind_speed',      # Mean wind speed at hub height [m/s]
                'wind_direction',  # Wind direction [deg]
                'turbulence_intensity',  # Turbulence intensity [-]
                'shear_exponent',  # Wind shear exponent [-]
                'air_density'     # Air density [kg/m³]
            ],
            'bounds': [
                [6.0, 10.0],      # Wind speed: ±2 m/s uncertainty
                [260.0, 280.0],   # Wind direction: ±10° uncertainty
                [0.08, 0.12],     # TI: ±0.02 uncertainty
                [0.10, 0.20],     # Shear: ±0.05 uncertainty
                [1.18, 1.22]      # Density: ±0.02 kg/m³ uncertainty
            ]
        }
        
        # Reference values (nominal conditions)
        self.nominal_values = {
            'wind_speed': 8.0,
            'wind_direction': 270.0,
            'turbulence_intensity': 0.10,
            'shear_exponent': 0.15,
            'air_density': 1.20
        }
        
        print(f"Uncertainty parameters defined: {self.problem['names']}")
    
    def setup_flow_field_grid(self):
        """Setup grid for flow field analysis."""
        # Define grid covering the wind farm with some margin
        x_min, x_max = min(self.x) - 200, max(self.x) + 400
        y_min, y_max = min(self.y) - 200, max(self.y) + 400
        
        self.grid_resolution = 5  # meters
        self.x_grid = np.arange(x_min, x_max, self.grid_resolution)
        self.y_grid = np.arange(y_min, y_max, self.grid_resolution)
        
        print(f"Flow field grid: {len(self.x_grid)} x {len(self.y_grid)} points")
    
    def generate_parameter_samples(self, n_samples=1024):
        """
        Generate parameter samples using Saltelli's sequence.
        
        Parameters:
        -----------
        n_samples : int
            Base number of samples (will generate N = n_samples * (2D + 2) samples)
            
        Returns:
        --------
        samples : ndarray
            Parameter samples
        """
        print(f"Generating {n_samples} base samples using Saltelli sequence...")
        
        # Generate samples
        param_values = saltelli.sample(self.problem, n_samples, calc_second_order=True)
        
        # Calculate total number of samples
        N = len(param_values)
        print(f"Total samples generated: {N}")
        
        return param_values
    
    def simulate_flow_field(self, params):
        """
        Simulate flow field for given parameters.
        
        Parameters:
        -----------
        params : dict or array
            Parameter values
            
        Returns:
        --------
        flow_field : ndarray
            2D array of WS_eff values
        """
        if isinstance(params, np.ndarray):
            # Convert array to dict
            param_dict = {
                'wind_speed': params[0],
                'wind_direction': params[1],
                'turbulence_intensity': params[2],
                'shear_exponent': params[3],
                'air_density': params[4]
            }
        else:
            param_dict = params
        
        try:
            # Update site with current parameters
            self.site.default_wd = param_dict['wind_direction']
            self.site.default_ws = param_dict['wind_speed']
            
            # Simulate wind farm
            sim_result = self.wake_model(
                self.x, self.y,
                ws=param_dict['wind_speed'],
                wd=param_dict['wind_direction'],
                TI=param_dict['turbulence_intensity']
            )
            
            # Calculate flow field
            flow_map = sim_result.flow_map(
                HorizontalGrid(x=self.x_grid, y=self.y_grid, h=70)
            )
            
            # Extract WS_eff
            ws_field = flow_map.WS_eff.values
            
            return ws_field
            
        except Exception as e:
            print(f"Error in simulation: {e}")
            # Return NaN array of appropriate shape
            return np.full((len(self.y_grid), len(self.x_grid)), np.nan)
    
    def perform_sensitivity_analysis(self, n_samples=256, n_timesteps=5):
        """
        Perform Sobol sensitivity analysis across the flow field.
        
        Parameters:
        -----------
        n_samples : int
            Base number of samples for Sobol analysis
        n_timesteps : int
            Number of time steps to analyze
        """
        print("\n" + "="*60)
        print("Starting Sobol Sensitivity Analysis")
        print("="*60)
        
        # Generate parameter samples
        param_samples = self.generate_parameter_samples(n_samples)
        N = len(param_samples)
        
        # Initialize storage for flow fields
        flow_fields = np.zeros((N, len(self.y_grid), len(self.x_grid)))
        
        # Run simulations
        print(f"\nRunning {N} simulations...")
        start_time = time.time()
        
        for i in range(N):
            if i % 50 == 0:
                elapsed = time.time() - start_time
                print(f"  Completed {i}/{N} simulations ({i/N*100:.1f}%) - "
                      f"Elapsed: {elapsed:.1f}s")
            
            flow_fields[i] = self.simulate_flow_field(param_samples[i])
        
        total_time = time.time() - start_time
        print(f"All simulations completed in {total_time:.1f} seconds")
        print(f"Average time per simulation: {total_time/N:.3f} seconds")
        
        # Perform sensitivity analysis at each grid point
        print("\nPerforming Sobol analysis at each grid point...")
        
        # Initialize sensitivity arrays
        n_params = self.problem['num_vars']
        sensitivity_maps = {
            'S1': np.zeros((n_params, len(self.y_grid), len(self.x_grid))),
            'ST': np.zeros((n_params, len(self.y_grid), len(self.x_grid))),
            'S2': np.zeros((n_params, n_params, len(self.y_grid), len(self.x_grid)))
        }
        
        # Calculate Sobol indices for each grid point
        for i in range(len(self.y_grid)):
            for j in range(len(self.x_grid)):
                # Extract WS_eff values at this grid point
                Y = flow_fields[:, i, j]
                
                # Skip if all NaN
                if np.all(np.isnan(Y)):
                    continue
                
                # Perform Sobol analysis
                try:
                    Si = sobol.analyze(self.problem, Y, 
                                      calc_second_order=True,
                                      print_to_console=False)
                    
                    # Store first order indices
                    sensitivity_maps['S1'][:, i, j] = Si['S1']
                    
                    # Store total order indices
                    sensitivity_maps['ST'][:, i, j] = Si['ST']
                    
                    # Store second order indices
                    if Si['S2'] is not None:
                        sensitivity_maps['S2'][:, :, i, j] = Si['S2']
                        
                except Exception as e:
                    # If analysis fails, leave as zeros
                    pass
            
            if i % 20 == 0:
                print(f"  Processed {i}/{len(self.y_grid)} rows")
        
        # Save sensitivity results
        self.save_sensitivity_results(sensitivity_maps, param_samples, flow_fields)
        
        # Generate visualization
        self.visualize_sensitivity(sensitivity_maps, n_timesteps)
        
        return sensitivity_maps
    
    def save_sensitivity_results(self, sensitivity_maps, param_samples, flow_fields):
        """Save sensitivity analysis results."""
        print("\nSaving results...")
        
        # Save metadata
        metadata = {
            'timestamp': datetime.now().isoformat(),
            'problem_definition': self.problem,
            'nominal_values': self.nominal_values,
            'grid_shape': flow_fields.shape[1:],
            'grid_resolution': self.grid_resolution,
            'x_grid_range': [float(self.x_grid[0]), float(self.x_grid[-1])],
            'y_grid_range': [float(self.y_grid[0]), float(self.y_grid[-1])],
            'turbine_positions': {
                'x': self.x.tolist(),
                'y': self.y.tolist()
            }
        }
        
        with open(self.output_dir / 'metadata.json', 'w') as f:
            json.dump(metadata, f, indent=2)
        
        # Save sensitivity maps as numpy arrays
        np.save(self.output_dir / 'sensitivity_S1.npy', sensitivity_maps['S1'])
        np.save(self.output_dir / 'sensitivity_ST.npy', sensitivity_maps['ST'])
        np.save(self.output_dir / 'parameter_samples.npy', param_samples)
        np.save(self.output_dir / 'flow_fields.npy', flow_fields)
        
        print(f"Results saved to {self.output_dir}")
    
    def visualize_sensitivity(self, sensitivity_maps, n_timesteps=5):
        """
        Generate visualization of sensitivity analysis.
        
        Parameters:
        -----------
        sensitivity_maps : dict
            Dictionary containing sensitivity indices
        n_timesteps : int
            Number of time steps to visualize
        """
        print("\nGenerating visualizations...")
        
        # Create colormap for sensitivity
        sens_cmap = cm.viridis
        
        # Plot turbine positions
        turbine_marker = dict(marker='^', s=200, c='red', edgecolor='black', 
                             linewidth=1, zorder=10)
        
        # 1. Plot first order sensitivity for each parameter
        param_names = self.problem['names']
        
        fig, axes = plt.subplots(2, 3, figsize=(15, 10))
        axes = axes.flatten()
        
        for idx, (ax, param_name) in enumerate(zip(axes, param_names)):
            if idx < len(param_names):
                sens_map = sensitivity_maps['S1'][idx]
                
                # Create mask for valid values
                mask = ~np.isnan(sens_map)
                sens_map_masked = np.where(mask, sens_map, 0)
                
                im = ax.imshow(sens_map_masked, 
                              extent=[self.x_grid[0], self.x_grid[-1],
                                      self.y_grid[0], self.y_grid[-1]],
                              origin='lower',
                              cmap=sens_cmap,
                              vmin=0, vmax=1)
                
                ax.scatter(self.x, self.y, **turbine_marker)
                ax.set_title(f'S1: {param_name}', fontsize=12, fontweight='bold')
                ax.set_xlabel('X [m]')
                ax.set_ylabel('Y [m]')
                ax.grid(True, alpha=0.3)
                
                # Add colorbar
                plt.colorbar(im, ax=ax, label='First-order Sobol index')
            else:
                ax.axis('off')
        
        plt.suptitle('First-Order Sobol Sensitivity Indices for WS_eff', 
                    fontsize=14, fontweight='bold')
        plt.tight_layout()
        plt.savefig(self.sensitivity_dir / 'first_order_sensitivity.png', 
                   dpi=300, bbox_inches='tight')
        plt.close()
        
        # 2. Plot total order sensitivity for each parameter
        fig, axes = plt.subplots(2, 3, figsize=(15, 10))
        axes = axes.flatten()
        
        for idx, (ax, param_name) in enumerate(zip(axes, param_names)):
            if idx < len(param_names):
                sens_map = sensitivity_maps['ST'][idx]
                
                # Create mask for valid values
                mask = ~np.isnan(sens_map)
                sens_map_masked = np.where(mask, sens_map, 0)
                
                im = ax.imshow(sens_map_masked, 
                              extent=[self.x_grid[0], self.x_grid[-1],
                                      self.y_grid[0], self.y_grid[-1]],
                              origin='lower',
                              cmap=sens_cmap,
                              vmin=0, vmax=1)
                
                ax.scatter(self.x, self.y, **turbine_marker)
                ax.set_title(f'ST: {param_name}', fontsize=12, fontweight='bold')
                ax.set_xlabel('X [m]')
                ax.set_ylabel('Y [m]')
                ax.grid(True, alpha=0.3)
                
                # Add colorbar
                plt.colorbar(im, ax=ax, label='Total-order Sobol index')
            else:
                ax.axis('off')
        
        plt.suptitle('Total-Order Sobol Sensitivity Indices for WS_eff', 
                    fontsize=14, fontweight='bold')
        plt.tight_layout()
        plt.savefig(self.sensitivity_dir / 'total_order_sensitivity.png', 
                   dpi=300, bbox_inches='tight')
        plt.close()
        
        # 3. Create time series of sensitivity (simulated by varying parameters)
        print("Generating time series of flow field sensitivity...")
        
        # Create parameter variations over time
        time_steps = np.linspace(0, 2*np.pi, n_timesteps)
        
        for t_idx, t in enumerate(time_steps):
            # Vary parameters sinusoidally over time
            varied_params = self.nominal_values.copy()
            
            # Add time-varying perturbations
            perturbation = 0.5 * np.sin(t)
            
            varied_params['wind_speed'] += perturbation * 2.0
            varied_params['wind_direction'] += perturbation * 10.0
            varied_params['turbulence_intensity'] += perturbation * 0.02
            
            # Simulate flow field with varied parameters
            ws_field = self.simulate_flow_field(varied_params)
            
            # Calculate sensitivity-weighted flow field
            # Weight by total sensitivity to wind speed (most important parameter)
            sensitivity_weight = sensitivity_maps['ST'][0]  # Wind speed sensitivity
            
            # Normalize sensitivity weight
            if np.max(sensitivity_weight) > 0:
                sensitivity_weight = sensitivity_weight / np.max(sensitivity_weight)
            
            # Create sensitivity-highlighted flow field
            highlighted_field = ws_field * (1 + 0.5 * sensitivity_weight)
            
            # Plot
            fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 6))
            
            # Plot flow field
            im1 = ax1.imshow(ws_field,
                           extent=[self.x_grid[0], self.x_grid[-1],
                                   self.y_grid[0], self.y_grid[-1]],
                           origin='lower',
                           cmap=cm.jet,
                           vmin=4, vmax=12)
            
            ax1.scatter(self.x, self.y, **turbine_marker)
            ax1.set_title(f'Flow Field at Time Step {t_idx+1}', fontsize=12, fontweight='bold')
            ax1.set_xlabel('X [m]')
            ax1.set_ylabel('Y [m]')
            ax1.grid(True, alpha=0.3)
            plt.colorbar(im1, ax=ax1, label='WS_eff [m/s]')
            
            # Plot sensitivity-highlighted field
            im2 = ax2.imshow(highlighted_field,
                           extent=[self.x_grid[0], self.x_grid[-1],
                                   self.y_grid[0], self.y_grid[-1]],
                           origin='lower',
                           cmap=cm.plasma,
                           vmin=4, vmax=14)
            
            ax2.scatter(self.x, self.y, **turbine_marker)
            ax2.set_title(f'Sensitivity-Weighted Flow Field', fontsize=12, fontweight='bold')
            ax2.set_xlabel('X [m]')
            ax2.set_ylabel('Y [m]')
            ax2.grid(True, alpha=0.3)
            plt.colorbar(im2, ax=ax2, label='Sensitivity-Weighted WS_eff [m/s]')
            
            plt.suptitle(f'Time Step {t_idx+1}/{n_timesteps}: '
                        f'WS={varied_params["wind_speed"]:.1f}m/s, '
                        f'WD={varied_params["wind_direction"]:.0f}°', 
                        fontsize=14, fontweight='bold')
            plt.tight_layout()
            
            # Save figure
            fig_path = self.flowfield_dir / f'flow_field_sensitivity_t{t_idx+1:02d}.png'
            plt.savefig(fig_path, dpi=300, bbox_inches='tight')
            plt.close()
            
            print(f"  Saved time step {t_idx+1}/{n_timesteps}")
        
        # 4. Create summary plot of most sensitive regions
        print("Generating summary plot...")
        
        # Calculate overall sensitivity (sum of total indices)
        overall_sensitivity = np.sum(sensitivity_maps['ST'], axis=0)
        
        fig, ax = plt.subplots(figsize=(10, 8))
        
        im = ax.imshow(overall_sensitivity,
                      extent=[self.x_grid[0], self.x_grid[-1],
                              self.y_grid[0], self.y_grid[-1]],
                      origin='lower',
                      cmap=cm.hot,
                      alpha=0.8)
        
        # Overlay flow field contours
        nominal_field = self.simulate_flow_field(self.nominal_values)
        contours = ax.contour(self.x_grid, self.y_grid, nominal_field,
                             levels=10, colors='white', alpha=0.5, linewidths=0.5)
        ax.clabel(contours, inline=True, fontsize=8, fmt='%1.1f m/s')
        
        ax.scatter(self.x, self.y, **turbine_marker)
        ax.set_title('Overall Sensitivity of WS_eff to Inflow Uncertainties', 
                    fontsize=14, fontweight='bold')
        ax.set_xlabel('X [m]')
        ax.set_ylabel('Y [m]')
        ax.grid(True, alpha=0.3)
        
        plt.colorbar(im, ax=ax, label='Sum of Total-Order Sobol Indices')
        
        plt.tight_layout()
        plt.savefig(self.sensitivity_dir / 'overall_sensitivity_summary.png',
                   dpi=300, bbox_inches='tight')
        plt.close()
        
        print(f"Visualizations saved to {self.sensitivity_dir} and {self.flowfield_dir}")
    
    def generate_report(self):
        """Generate a summary report of the sensitivity analysis."""
        print("\n" + "="*60)
        print("Generating Analysis Report")
        print("="*60)
        
        report = f"""
        SOBOL SENSITIVITY ANALYSIS REPORT
        =================================
        
        Analysis performed: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}
        
        WIND FARM CONFIGURATION:
        ------------------------
        - Number of turbines: {len(self.x)}
        - Turbine type: {self.wt.name}
        - Rotor diameter: {self.wt.diameter} m
        - Hub height: {self.wt.hub_height} m
        
        UNCERTAINTY PARAMETERS:
        -----------------------
        """
        
        for i, name in enumerate(self.problem['names']):
            bounds = self.problem['bounds'][i]
            nominal = self.nominal_values[name]
            report += f"  - {name}: {bounds[0]:.2f} to {bounds[1]:.2f} (nominal: {nominal:.2f})\n"
        
        report += f"""
        FLOW FIELD ANALYSIS:
        --------------------
        - Grid resolution: {self.grid_resolution} m
        - Grid dimensions: {len(self.x_grid)} x {len(self.y_grid)} points
        - Analysis area: {self.x_grid[0]:.0f} to {self.x_grid[-1]:.0f} m (X)
                         {self.y_grid[0]:.0f} to {self.y_grid[-1]:.0f} m (Y)
        
        RESULTS:
        --------
        Results have been saved to: {self.output_dir}
        
        Files generated:
        - metadata.json: Analysis metadata and parameters
        - sensitivity_S1.npy: First-order Sobol indices
        - sensitivity_ST.npy: Total-order Sobol indices
        - parameter_samples.npy: Parameter samples used
        - flow_fields.npy: Simulated flow fields
        
        Visualizations:
        - first_order_sensitivity.png: First-order sensitivity maps
        - total_order_sensitivity.png: Total-order sensitivity maps
        - overall_sensitivity_summary.png: Summary of sensitive regions
        - flow_field_sensitivity_t*.png: Time series of sensitivity
        
        INTERPRETATION:
        ---------------
        1. First-order indices (S1) show the direct effect of each parameter.
        2. Total-order indices (ST) show the total effect including interactions.
        3. Higher values indicate greater sensitivity.
        4. Regions with high sensitivity require more accurate measurements.
        5. Wake regions typically show higher sensitivity due to complex interactions.
        
        RECOMMENDATIONS:
        ----------------
        1. Focus measurement accuracy on parameters with high sensitivity.
        2. Consider adaptive measurement strategies for sensitive regions.
        3. Use results to optimize sensor placement.
        4. Incorporate sensitivity information into control strategies.
        """
        
        # Save report
        report_path = self.output_dir / 'sensitivity_analysis_report.txt'
        with open(report_path, 'w') as f:
            f.write(report)
        
        print(report)
        print(f"\nReport saved to: {report_path}")
```

This script provides a comprehensive framework for propagating measurement uncertainties through PyWake and performing Sobol sensitivity analysis on the flow field. Here's what it does:

## Key Features:

1. **Realistic Uncertainty Modeling**: 
   - Wind speed (±2 m/s)
   - Wind direction (±10°)
