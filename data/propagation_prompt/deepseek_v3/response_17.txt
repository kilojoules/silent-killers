        return flow_field.WS_eff.squeeze().values

def perform_sobol_analysis(wind_farm, time_steps=6, n_samples=512):
    """
    Perform Sobol sensitivity analysis across multiple time steps
    
    Parameters:
    -----------
    wind_farm : WindFarmWithUncertainties
        Wind farm model
    time_steps : int
        Number of time steps to analyze
    n_samples : int
        Number of samples for Sobol analysis
    """
    
    print(f"Starting Sobol sensitivity analysis with {n_samples} samples...")
    print(f"Analyzing {time_steps} time steps")
    
    # Generate parameter samples using Saltelli sequence
    param_values = saltelli.sample(wind_farm.problem, n_samples, calc_second_order=True)
    
    # Initialize results storage
    sensitivity_results = []
    flow_field_samples = []
    
    # Track progress
    total_simulations = len(param_values) * time_steps
    current_sim = 0
    
    # For each time step
    for t in range(time_steps):
        print(f"\nTime step {t+1}/{time_steps}")
        
        # Store flow fields for this time step
        flow_fields_t = []
        
        # Run simulations for all parameter sets
        for i, params in enumerate(param_values):
            # Simulate flow field
            flow_field = wind_farm.simulate_flow_field(params, time_step=t)
            flow_fields_t.append(flow_field)
            
            # Progress tracking
            current_sim += 1
            if i % 50 == 0:
                print(f"  Completed {current_sim}/{total_simulations} simulations")
        
        # Convert to numpy array
        flow_fields_t = np.array(flow_fields_t)
        flow_field_samples.append(flow_fields_t)
        
        # Reshape for Sobol analysis (each grid point separately)
        n_grid_points = flow_fields_t.shape[1] * flow_fields_t.shape[2]
        flow_fields_reshaped = flow_fields_t.reshape(len(param_values), n_grid_points)
        
        # Perform Sobol analysis for each grid point
        print(f"  Performing Sobol analysis for {n_grid_points} grid points...")
        
        # We'll analyze a subset of grid points for efficiency
        # or use parallel processing for full analysis
        sample_indices = np.linspace(0, n_grid_points-1, min(1000, n_grid_points), dtype=int)
        
        S1_all = []
        ST_all = []
        
        for idx in sample_indices:
            Y = flow_fields_reshaped[:, idx]
            
            # Sobol analysis
            Si = sobol.analyze(
                wind_farm.problem,
                Y,
                calc_second_order=False,
                print_to_console=False
            )
            
            S1_all.append(Si['S1'])
            ST_all.append(Si['ST'])
        
        # Store results
        sensitivity_results.append({
            'time_step': t,
            'S1': np.array(S1_all),
            'ST': np.array(ST_all),
            'sample_indices': sample_indices,
            'grid_shape': flow_fields_t[0].shape
        })
        
        # Save intermediate results
        save_sensitivity_results(sensitivity_results[-1], t)
    
    return sensitivity_results, flow_field_samples

def save_sensitivity_results(results, time_step):
    """Save sensitivity results for a time step"""
    
    # Create figure
    fig, axes = plt.subplots(2, 4, figsize=(20, 10))
    axes = axes.flatten()
    
    # Plot first order indices for each parameter
    param_names = ['WS', 'WD', 'TI', 'Rho', 'ka', 'Ct', 'Power']
    
    for i, (ax, param_name) in enumerate(zip(axes[:7], param_names)):
        # Reshape sensitivity indices to grid
        S1_grid = np.full(results['grid_shape'], np.nan)
        indices_flat = results['sample_indices']
        S1_values = results['S1'][:, i]
        
        # Map flat indices to 2D grid
        ny, nx = results['grid_shape']
        for idx, val in zip(indices_flat, S1_values):
            y_idx = idx // nx
            x_idx = idx % nx
            S1_grid[y_idx, x_idx] = val
        
        # Plot
        im = ax.imshow(S1_grid, cmap='viridis', origin='lower',
                      extent=[-200, 3500, -200, 2500])
        ax.set_title(f'S1: {param_name}')
        ax.set_xlabel('x [m]')
        ax.set_ylabel('y [m]')
        plt.colorbar(im, ax=ax)
    
    # Plot total sensitivity for wind speed (most important parameter)
    ax = axes[7]
    ST_grid = np.full(results['grid_shape'], np.nan)
    ST_values = results['ST'][:, 0]  # Wind speed total sensitivity
    
    for idx, val in zip(results['sample_indices'], ST_values):
        y_idx = idx // nx
        x_idx = idx % nx
        ST_grid[y_idx, x_idx] = val
    
    im = ax.imshow(ST_grid, cmap='plasma', origin='lower',
                  extent=[-200, 3500, -200, 2500])
    ax.set_title('ST: Wind Speed')
    ax.set_xlabel('x [m]')
    ax.set_ylabel('y [m]')
    plt.colorbar(im, ax=ax)
    
    plt.suptitle(f'Sobol Sensitivity Indices - Time Step {time_step}', fontsize=16)
    plt.tight_layout()
    
    # Save figure
    filename = output_dir / f'sensitivity_time_{time_step:02d}.png'
    plt.savefig(filename, dpi=150, bbox_inches='tight')
    plt.close()
    
    print(f"  Saved sensitivity plot: {filename}")

def create_sensitivity_movie(sensitivity_results):
    """Create an animation of sensitivity evolution over time"""
    
    # Create summary figure
    fig, axes = plt.subplots(2, 3, figsize=(18, 12))
    
    # Parameters to track
    track_params = ['WS', 'WD', 'TI', 'ka', 'Ct', 'Power']
    
    # Extract time series of average sensitivity
    time_steps = len(sensitivity_results)
    avg_S1 = {param: [] for param in track_params}
    max_S1 = {param: [] for param in track_params}
    
    for t in range(time_steps):
        results = sensitivity_results[t]
        for i, param in enumerate(track_params):
            avg_S1[param].append(np.nanmean(results['S1'][:, i]))
            max_S1[param].append(np.nanmax(results['S1'][:, i]))
    
    # Plot time series
    time_points = np.arange(time_steps)
    
    for idx, param in enumerate(track_params):
        ax = axes.flatten()[idx]
        
        ax.plot(time_points, avg_S1[param], 'b-', linewidth=2, label='Mean S1')
        ax.plot(time_points, max_S1[param], 'r--', linewidth=2, label='Max S1')
        
        ax.set_title(f'{param} Sensitivity Evolution')
        ax.set_xlabel('Time Step')
        ax.set_ylabel('Sobol Index')
        ax.legend()
        ax.grid(True, alpha=0.3)
    
    plt.suptitle('Sensitivity Evolution Over Time', fontsize=16)
    plt.tight_layout()
    
    # Save summary
    filename = output_dir / 'sensitivity_evolution.png'
    plt.savefig(filename, dpi=150, bbox_inches='tight')
    plt.close()
    
    print(f"Saved sensitivity evolution plot: {filename}")

This script provides a comprehensive Sobol sensitivity analysis for wind farm flow fields with realistic measurement uncertainties. Here's what it does:

## Key Features:

1. **Realistic Uncertainty Modeling**:
   - Wind speed measurement uncertainty (0.1-0.5 m/s)
   - Wind direction uncertainty (1-5 degrees)
   - Turbulence intensity uncertainty
   - Air density uncertainty
   - Wake model parameter uncertainties
   - Turbine performance uncertainties

2. **Sobol Sensitivity Analysis**:
   - Uses Saltelli sampling for efficient exploration
   - Calculates first-order (S1) and total (ST) sensitivity indices
   - Analyzes sensitivity across the entire flow field

3. **Time-Varying Analysis**:
   - Simulates diurnal wind variations
   - Analyzes sensitivity at multiple time steps
   - Shows how sensitivity evolves over time

4. **Visualization Output**:
   - Sensitivity maps for each parameter
   - Uncertainty propagation visualizations
   - Time evolution plots
   - Flow field statistics (mean, std, min, max)

## Output Files:

The script generates:
- `sensitivity_time_XX.png`: Sensitivity maps for each time step
- `uncertainty_time_XX.png`: Uncertainty propagation visualizations
- `sensitivity_evolution.png`: Time evolution of sensitivity
- `sensitivity_results.npy`: Raw results for further analysis
- `analysis_summary.txt`: Text summary of the analysis
