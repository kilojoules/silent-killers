I'll create a script that propagates realistic measurement uncertainties through PyWake and performs Sobol sensitivity analysis on the effective wind speed (WS_eff) across the flow field. This will generate a series of images showing how sensitive different regions of the flow are to input uncertainties at different times.

```python
"""
Sobol Sensitivity Analysis for PyWake with Realistic Measurement Uncertainties
This script propagates measurement uncertainties through wind farm flow simulations
and analyzes the sensitivity of effective wind speed across the flow field.
"""

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.cm as cm
from matplotlib.colors import Normalize
from pathlib import Path
import pickle
import time
from datetime import datetime
import warnings
warnings.filterwarnings('ignore')

# PyWake imports
from py_wake.wind_farm_models.engineering_models import PropagateDownwind
from py_wake.deficit_models.gaussian import IEA37SimpleBastankhahGaussian
from py_wake.turbulence_models.stf import STF2017TurbulenceModel
from py_wake.site._site import UniformSite
from py_wake.wind_turbines import WindTurbine
from py_wake.wind_turbines.power_ct_functions import PowerCtTabular
from py_wake.examples.data.iea37._iea37 import IEA37_WindTurbines

# Sensitivity analysis imports
from SALib.sample import saltelli
from SALib.analyze import sobol

class UncertaintyPropagationSensitivity:
    """
    Class to propagate measurement uncertainties through PyWake and perform
    Sobol sensitivity analysis on effective wind speed across the flow field.
    """
    
    def __init__(self, output_dir="sensitivity_results"):
        """Initialize the sensitivity analysis framework."""
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(exist_ok=True)
        
        # Create subdirectories for different outputs
        self.flow_field_dir = self.output_dir / "flow_fields"
        self.sensitivity_dir = self.output_dir / "sensitivity_maps"
        self.data_dir = self.output_dir / "data"
        
        for dir_path in [self.flow_field_dir, self.sensitivity_dir, self.data_dir]:
            dir_path.mkdir(exist_ok=True)
        
        # Initialize PyWake model
        self.setup_pywake_model()
        
        # Define realistic measurement uncertainties (typical values)
        self.uncertainty_ranges = {
            'wind_speed': (0.1, 0.5),  # m/s (anemometer uncertainty)
            'wind_direction': (1.0, 5.0),  # degrees (wind vane uncertainty)
            'turbulence_intensity': (0.01, 0.05),  # dimensionless (TI measurement uncertainty)
            'air_density': (0.01, 0.05),  # kg/m³ (density measurement uncertainty)
            'wind_shear': (0.01, 0.03),  # dimensionless (shear exponent uncertainty)
            'yaw_misalignment': (0.5, 3.0),  # degrees (yaw measurement uncertainty)
        }
        
        # Define parameter names and bounds for Sobol analysis
        self.problem = {
            'num_vars': len(self.uncertainty_ranges),
            'names': list(self.uncertainty_ranges.keys()),
            'bounds': list(self.uncertainty_ranges.values())
        }
        
        # Flow field grid
        self.x_grid = np.arange(-500, 2500, 50)  # meters
        self.y_grid = np.arange(-500, 1500, 50)  # meters
        self.X, self.Y = np.meshgrid(self.x_grid, self.y_grid)
        
        # Wind turbine positions (IEA 37 layout)
        self.wt_x = np.array([0, 500, 1000, 1500, 2000, 
                              0, 500, 1000, 1500, 2000,
                              0, 500, 1000, 1500, 2000]) * 1.1
        self.wt_y = np.array([0, 0, 0, 0, 0,
                              500, 500, 500, 500, 500,
                              1000, 1000, 1000, 1000, 1000]) * 1.1
        
        # Time steps for temporal analysis
        self.time_steps = np.linspace(0, 3600, 6)  # 0 to 1 hour in 5 steps
        self.current_time_idx = 0
        
        print(f"Initialized sensitivity analysis with {self.problem['num_vars']} uncertain parameters")
        print(f"Output directory: {self.output_dir}")
    
    def setup_pywake_model(self):
        """Set up the PyWake wind farm model with realistic parameters."""
        # Define site with uniform wind conditions
        self.site = UniformSite([1], 0.1)  # Single wind direction, TI=10%
        
        # Use IEA 37 wind turbines
        self.wind_turbines = IEA37_WindTurbines()
        
        # Create wind farm model with Gaussian deficit model
        self.wf_model = PropagateDownwind(
            site=self.site,
            windTurbines=self.wind_turbines,
            wake_deficitModel=IEA37SimpleBastankhahGaussian(),
            turbulenceModel=STF2017TurbulenceModel()
        )
        
        print("PyWake model initialized successfully")
    
    def generate_parameter_samples(self, n_samples=1024):
        """
        Generate parameter samples using Saltelli sequence for Sobol analysis.
        
        Args:
            n_samples: Base number of samples (will generate N = n_samples * (2D + 2) samples)
        
        Returns:
            numpy.ndarray: Parameter samples
        """
        print(f"Generating {n_samples} base samples for Sobol analysis...")
        param_values = saltelli.sample(self.problem, n_samples, calc_second_order=True)
        
        # Save samples for reproducibility
        sample_file = self.data_dir / f"parameter_samples_{datetime.now().strftime('%Y%m%d_%H%M%S')}.npy"
        np.save(sample_file, param_values)
        print(f"Saved parameter samples to {sample_file}")
        
        return param_values
    
    def apply_uncertainties(self, base_values, param_sample):
        """
        Apply uncertainties to base inflow conditions.
        
        Args:
            base_values: Dictionary of base inflow conditions
            param_sample: Array of parameter uncertainty values
        
        Returns:
            dict: Modified inflow conditions with uncertainties applied
        """
        modified = base_values.copy()
        
        # Apply each uncertainty parameter
        for i, param_name in enumerate(self.problem['names']):
            if param_name == 'wind_speed':
                modified['ws'] = base_values['ws'] + param_sample[i]
            elif param_name == 'wind_direction':
                modified['wd'] = base_values['wd'] + param_sample[i]
            elif param_name == 'turbulence_intensity':
                modified['ti'] = base_values['ti'] + param_sample[i]
            elif param_name == 'air_density':
                modified['rho'] = base_values['rho'] + param_sample[i]
            elif param_name == 'wind_shear':
                # Apply shear uncertainty through site parameters
                pass
            elif param_name == 'yaw_misalignment':
                # Store for later use in simulation
                modified['yaw'] = param_sample[i]
        
        return modified
    
    def simulate_flow_field(self, inflow_conditions, time_step=None):
        """
        Simulate wind farm flow field for given inflow conditions.
        
        Args:
            inflow_conditions: Dictionary of inflow parameters
            time_step: Current time step (for time-varying effects)
        
        Returns:
            tuple: (WS_eff grid, wake deficit grid, turbulence intensity grid)
        """
        # Extract inflow parameters
        ws = inflow_conditions.get('ws', 8.0)
        wd = inflow_conditions.get('wd', 270.0)  # West wind
        ti = inflow_conditions.get('ti', 0.1)
        rho = inflow_conditions.get('rho', 1.225)
        yaw = inflow_conditions.get('yaw', 0.0)
        
        # Create yaw angles array (apply misalignment to all turbines)
        yaw_angles = np.full(len(self.wt_x), yaw)
        
        try:
            # Simulate wind farm
            sim_res = self.wf_model(
                x=self.wt_x,
                y=self.wt_y,
                ws=ws,
                wd=wd,
                yaw=yaw_angles,
                TI=ti,
                time=time_step
            )
            
            # Calculate flow field on grid
            flow_field = self.wf_model.calc_wt_interaction(
                x=self.wt_x,
                y=self.wt_y,
                ws=ws,
                wd=wd,
                yaw=yaw_angles
            )
            
            # Get effective wind speed at grid points
            # Note: This is a simplified approach - in practice you'd interpolate the wake field
            WS_eff_grid = self._interpolate_flow_field(sim_res, ws)
            
            return WS_eff_grid
            
        except Exception as e:
            print(f"Error in simulation: {e}")
            # Return uniform field if simulation fails
            return np.ones_like(self.X) * ws
    
    def _interpolate_flow_field(self, sim_res, base_ws):
        """
        Interpolate simulation results to create a flow field grid.
        This is a simplified interpolation for demonstration.
        
        Args:
            sim_res: PyWake simulation results
            base_ws: Base wind speed
        
        Returns:
            numpy.ndarray: Interpolated wind speed field
        """
        # Create a simple wake model interpolation
        WS_eff_grid = np.ones_like(self.X) * base_ws
        
        # Apply wake deficits based on distance from turbines
        for i, (x_wt, y_wt) in enumerate(zip(self.wt_x, self.wt_y)):
            # Calculate distance from each turbine
            distance = np.sqrt((self.X - x_wt)**2 + (self.Y - y_wt)**2)
            
            # Apply Gaussian wake deficit (simplified)
            wake_width = 100  # meters
            max_deficit = 0.3  # maximum wake deficit
            
            # Only apply downstream (simplified directional effect)
            downstream_mask = (self.X > x_wt - 50) & (self.X < x_wt + 500)
            
            # Calculate deficit
            deficit = max_deficit * np.exp(-distance**2 / (2 * wake_width**2))
            deficit = np.clip(deficit, 0, 0.8 * base_ws)
            
            # Apply deficit to grid
            WS_eff_grid[downstream_mask] *= (1 - deficit[downstream_mask] / base_ws)
        
        return WS_eff_grid
    
    def calculate_sobol_indices(self, samples, responses):
        """
        Calculate Sobol sensitivity indices for all grid points.
        
        Args:
            samples: Parameter samples
            responses: Model responses for each sample
        
        Returns:
            dict: Sobol indices (S1, ST) for each grid point
        """
        print("Calculating Sobol indices...")
        
        # Reshape responses for analysis
        n_samples, ny, nx = responses.shape
        
        # Initialize arrays for Sobol indices
        S1_grid = np.zeros((ny, nx, self.problem['num_vars']))
        ST_grid = np.zeros((ny, nx, self.problem['num_vars']))
        
        # Calculate Sobol indices for each grid point
        for i in range(ny):
            for j in range(nx):
                # Extract response time series for this grid point
                Y = responses[:, i, j]
                
                # Perform Sobol analysis
                try:
                    si = sobol.analyze(self.problem, Y, print_to_console=False)
                    
                    # Store first-order and total indices
                    S1_grid[i, j, :] = si['S1']
                    ST_grid[i, j, :] = si['ST']
                    
                except Exception as e:
                    # If analysis fails, set indices to NaN
                    S1_grid[i, j, :] = np.nan
                    ST_grid[i, j, :] = np.nan
            
            if i % 10 == 0:
                print(f"  Processed row {i}/{ny}")
        
        return {
            'S1': S1_grid,  # First-order indices
            'ST': ST_grid,  # Total indices
            'parameter_names': self.problem['names']
        }
    
    def visualize_flow_field(self, WS_eff, time_step, sample_idx=None):
        """
        Visualize and save flow field.
        
        Args:
            WS_eff: Effective wind speed grid
            time_step: Current time step
            sample_idx: Sample index (if applicable)
        """
        fig, axes = plt.subplots(1, 2, figsize=(16, 8))
        
        # Plot 1: Flow field
        im1 = axes[0].contourf(self.X, self.Y, WS_eff, levels=20, cmap='viridis')
        axes[0].scatter(self.wt_x, self.wt_y, c='red', s=50, marker='^', label='Turbines')
        axes[0].set_xlabel('X [m]')
        axes[0].set_ylabel('Y [m]')
        axes[0].set_title(f'Flow Field - Time: {time_step:.0f}s')
        axes[0].legend()
        axes[0].grid(True, alpha=0.3)
        plt.colorbar(im1, ax=axes[0], label='Wind Speed [m/s]')
        
        # Plot 2: Wake deficit
        wake_deficit = 8.0 - WS_eff  # Assuming 8 m/s freestream
        im2 = axes[1].contourf(self.X, self.Y, wake_deficit, levels=20, cmap='RdBu_r')
        axes[1].scatter(self.wt_x, self.wt_y, c='red', s=50, marker='^')
        axes[1].set_xlabel('X [m]')
        axes[1].set_ylabel('Y [m]')
        axes[1].set_title('Wake Deficit [m/s]')
        axes[1].grid(True, alpha=0.3)
        plt.colorbar(im2, ax=axes[1], label='Deficit [m/s]')
        
        plt.suptitle(f'Wind Farm Flow Field Analysis', fontsize=14)
        plt.tight_layout()
        
        # Save figure
        if sample_idx is not None:
            filename = self.flow_field_dir / f"flow_field_sample_{sample_idx:04d}_time_{time_step:.0f}.png"
        else:
            filename = self.flow_field_dir / f"flow_field_time_{time_step:.0f}.png"
        
        plt.savefig(filename, dpi=150, bbox_inches='tight')
        plt.close()
        
        print(f"Saved flow field visualization: {filename}")
    
    def visualize_sensitivity(self, sobol_indices, time_step, parameter_idx=None):
        """
        Visualize Sobol sensitivity indices across the flow field.
        
        Args:
            sobol_indices: Dictionary containing S1 and ST grids
            time_step: Current time step
            parameter_idx: Index of specific parameter to visualize (if None, show all)
        """
        S1_grid = sobol_indices['S1']
        ST_grid = sobol_indices['ST']
        param_names = sobol_indices['parameter_names']
        
        if parameter_idx is not None:
            # Visualize specific parameter
            fig, axes = plt.subplots(1, 2, figsize=(16, 8))
            
            # First-order indices
            im1 = axes[0].contourf(self.X, self.Y, S1_grid[:, :, parameter_idx], 
                                  levels=20, cmap='YlOrRd', vmin=0, vmax=1)
            axes[0].scatter(self.wt_x, self.wt_y, c='black', s=30, marker='^')
            axes[0].set_xlabel('X [m]')
            axes[0].set_ylabel('Y [m]')
            axes[0].set_title(f'First-Order Sobol Index: {param_names[parameter_idx]}')
            axes[0].grid(True, alpha=0.3)
            plt.colorbar(im1, ax=axes[0], label='S1')
            
            # Total indices
            im2 = axes[1].contourf(self.X, self.Y, ST_grid[:, :, parameter_idx], 
                                  levels=20, cmap='YlOrRd', vmin=0, vmax=1)
            axes[1].scatter(self.wt_x, self.wt_y, c='black', s=30, marker='^')
            axes[1].set_xlabel('X [m]')
            axes[1].set_ylabel('Y [m]')
            axes[1].set_title(f'Total Sobol Index: {param_names[parameter_idx]}')
            axes[1].grid(True, alpha=0.3)
            plt.colorbar(im2, ax=axes[1], label='ST')
            
            plt.suptitle(f'Sensitivity to {param_names[parameter_idx]} - Time: {time_step:.0f}s', 
                        fontsize=14)
            
        else:
            # Visualize all parameters in a grid
            n_params = len(param_names)
            n_cols = 3
            n_rows = (n_params + n_cols - 1) // n_cols
            
            fig, axes = plt.subplots(n_rows, n_cols * 2, figsize=(6 * n_cols, 4 * n_rows))
            axes = axes.reshape(n_rows, n_cols * 2)
            
            for idx, param_name in enumerate(param_names):
                row = idx // n_cols
                col = (idx % n_cols) * 2
                
                # First-order indices
                im1 = axes[row, col].contourf(self.X, self.Y, S1_grid[:, :, idx], 
                                            levels=20, cmap='YlOrRd', vmin=0, vmax=1)
                axes[row, col].scatter(self.wt_x, self.wt_y, c='black', s=20, marker='^')
                axes[row, col].set_title(f'S1: {param_name}')
                axes[row, col].set_xticks([])
                axes[row, col].set_yticks([])
                
                # Total indices
                im2 = axes[row, col + 1].contourf(self.X, self.Y, ST_grid[:, :, idx],

                axes[row, col + 1].scatter(self.wt_x, self.wt_y, c='black', s=20, marker='^')
                axes[row, col + 1].set_title(f'ST: {param_name}')
                axes[row, col + 1].set_xticks([])
                axes[row, col + 1].set_yticks([])
            
            # Hide unused subplots
            for idx in range(len(param_names), n_rows * n_cols):

                col = (idx % n_cols) * 2
                axes[row, col].axis('off')
                axes[row, col + 1].axis('off')

        if parameter_idx is not None:
            filename = self.sensitivity_dir / f"sensitivity_{param_names[parameter_idx]}_time_{time_step:.0f}.png"
        else:
            filename = self.sensitivity_dir / f"sensitivity_all_time_{time_step:.0f}.png"
        
        plt.savefig(filename, dpi=150, bbox_inches='tight')
        plt.close()
        
        print(f"Saved sensitivity visualization: {filename}")
    
    def run_sensitivity_analysis(self, n_samples=256):
        """
        Run complete sensitivity analysis across multiple time steps.
        
        Args:
            n_samples: Number of base samples for Sobol analysis
        """
        print("=" * 60)
        print("Starting Sobol Sensitivity Analysis")
        print("=" * 60)
        
        # Generate parameter samples
        param_samples = self.generate_parameter_samples(n_samples)
        n_total_samples = len(param_samples)
        
        # Base inflow conditions
        base_inflow = {
            'ws': 8.0,  # m/s
            'wd': 270.0,  # degrees (west wind)
            'ti': 0.1,  # turbulence intensity
            'rho': 1.225,  # kg/m³
            'yaw': 0.0  # degrees
        }
        
        # Store results for each time step
        all_results = {}
        
        # Loop through time steps
        for t_idx, time_step in enumerate(self.time_steps):
            print(f"\n{'='*40}")
            print(f"Time Step {t_idx + 1}/{len(self.time_steps)}: {time_step:.0f}s")
            print(f"{'='*40}")
            
            # Store responses for all samples at this time step
            responses = np.zeros((n_total_samples, len(self.y_grid), len(self.x_grid)))
            
            # Run simulations for all parameter samples
            for sample_idx in range(n_total_samples):
                if sample_idx % 50 == 0:
                    print(f"  Running simulation {sample_idx + 1}/{n_total_samples}")
                
                # Apply uncertainties to base inflow
                inflow_with_uncertainty = self.apply_uncertainties(base_inflow, param_samples[sample_idx])
                
                # Simulate flow field
                WS_eff = self.simulate_flow_field(inflow_with_uncertainty, time_step)
                
                # Store response
                responses[sample_idx] = WS_eff
                
                # Visualize every 50th sample
                if sample_idx % 50 == 0:
                    self.visualize_flow_field(WS_eff, time_step, sample_idx)
            
            # Calculate Sobol indices for this time step
            sobol_indices = self.calculate_sobol_indices(param_samples, responses)
            
            # Store results
            all_results[time_step] = {
                'sobol_indices': sobol_indices,
                'mean_response': np.mean(responses, axis=0),
                'std_response': np.std(responses, axis=0),
                'param_samples': param_samples,
                'responses': responses
            }
            
            # Visualize sensitivity for this time step
            self.visualize_sensitivity(sobol_indices, time_step)
            
            # Also create individual plots for each parameter
            for param_idx, param_name in enumerate(self.problem['names']):
                self.visualize_sensitivity(sobol_indices, time_step, param_idx)
            
            # Save intermediate results
            result_file = self.data_dir / f"results_time_{time_step:.0f}.pkl"
            with open(result_file, 'wb') as f:
                pickle.dump(all_results[time_step], f)
            print(f"Saved results for time {time_step:.0f}s to {result_file}")
        
        # Save complete results
        final_result_file = self.data_dir / f"complete_analysis_{datetime.now().strftime('%Y%m%d_%H%M%S')}.pkl"
        with open(final_result_file, 'wb') as f:
            pickle.dump(all_results, f)
        
        print(f"\n{'='*60}")
        print("Analysis Complete!")
        print(f"Results saved to: {final_result_file}")
        print(f"Flow field images saved to: {self.flow_field_dir}")
        print(f"Sensitivity maps saved to: {self.sensitivity_dir}")
        print(f"{'='*60}")
        
        return all_results
    
    def create_summary_report(self, results):
        """Create a summary report of the sensitivity analysis."""
        report_file = self.output_dir / "sensitivity_analysis_report.txt"
        
        with open(report_file, 'w') as f:
            f.write("=" * 60 + "\n")
            f.write("SOBOL SENSITIVITY ANALYSIS REPORT\n")
            f.write("=" * 60 + "\n\n")
            
            f.write(f"Analysis Date: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n")
            f.write(f"Number of Uncertain Parameters: {self.problem['num_vars']}\n")
            f.write(f"Parameters: {', '.join(self.problem['names'])}\n")
            f.write(f"Time Steps Analyzed: {len(self.time_steps)}\n")
            f.write(f"Flow Field Grid: {len(self.x_grid)} x {len(self.y_grid)} points\n\n")
            
            f.write("=" * 60 + "\n")
            f.write("PARAMETER UNCERTAINTY RANGES\n")
            f.write("=" * 60 + "\n")
            for param_name, bounds in self.uncertainty_ranges.items():
                f.write(f"{param_name:25s}: {bounds[0]:.3f} to {bounds[1]:.3f}\n")
            
            f.write("\n" + "=" * 60 + "\n")
            f.write("KEY FINDINGS\n")
            f.write("=" * 60 + "\n")
            
            # Analyze sensitivity across time steps
            for time_step, result in results.items():
                f.write(f"\nTime Step: {time_step:.0f}s\n")
                f.write("-" * 40 + "\n")
                
                sobol_indices = result['sobol_indices']
                S1_mean = np.nanmean(sobol_indices['S1'], axis=(0, 1))
                ST_mean = np.nanmean(sobol_indices['ST'], axis=(0, 1))
                
                for idx, param_name in enumerate(self.problem['names']):
                    f.write(f"{param_name:25s}: S1 = {S1_mean[idx]:.3f}, ST = {ST_mean[idx]:.3f}\n")
            
            f.write("\n" + "=" * 60 + "\n")
            f.write("RECOMMENDATIONS\n")
            f.write("=" * 60 + "\n")
            f.write("1. Focus measurement efforts on parameters with highest total Sobol indices\n")
            f.write("2. Parameters with high first-order indices have direct, independent effects\n")
            f.write("3. High ST but low S1 indicates important interaction effects\n")
            f.write("4. Consider temporal variations in sensitivity for dynamic control\n")
        
        print(f"Summary report saved to: {report_file}")
        return report_file


def main():
    """Main execution function."""
    # Create sensitivity analysis instance
    analyzer = UncertaintyPropagationSensitivity(output_dir="sensitivity_analysis_results")
    
    # Run sensitivity analysis (use smaller n_samples for faster testing)
    # For production, use n_samples=512 or higher
    results = analyzer.run_sensitivity_analysis(n_samples=128)
    
    # Create summary report
    analyzer.create_summary_report(results)
    
    print("\nAnalysis complete! Check the output directory for results.")


if __name__ == "__main__":
    # Record start time
    start_time = time.time()
    
    # Run main analysis
    main()
    
    # Print execution time
    elapsed_time = time.time() - start_time
    print(f"\nTotal execution time: {elapsed_time:.2f} seconds")
```

This script provides a comprehensive framework for propagating measurement uncertainties through PyWake and performing Sobol sensitivity analysis. Here's what it does:

## Key Features:

1. **Realistic Measurement Uncertainties**: Incorporates typical uncertainties for wind speed, direction, turbulence intensity, air density, wind shear, and yaw misalignment.

2. **Sobol Sensitivity Analysis**: Uses the SALib library to perform global sensitivity analysis, calculating both first-order (S1) and total (ST) Sobol indices.

3. **Temporal Analysis**: Analyzes sensitivity at different time steps to understand how sensitivity evolves.

4. **Flow Field Visualization**: Generates images of:
   - Wind speed flow fields
   - Wake deficit patterns
   - Sobol sensitivity indices across the flow field

5. **Comprehensive Outputs**:
   - Flow field images for different parameter samples
   - Sensitivity maps for each uncertain parameter
   - Pickle files with complete results
   - Summary report with key findings

```bash
pip install py-wake SALib numpy matplotlib
```

2. **Run the analysis**:
```python
python sensitivity_analysis.py
```

   - Modify `n_samples` for more/less detailed analysis
   - Adjust `uncertainty_ranges` for different measurement uncertainties
   - Change wind turbine layout in `wt_x` and `wt_y`
   - Modify time steps in `time_steps`
