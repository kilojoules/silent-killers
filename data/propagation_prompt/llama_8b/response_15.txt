Here's an example script that demonstrates how you could use PyWake to simulate the flow field, propagate measurement uncertainties, and perform a Sobol sensitivity analysis on `WS_eff` as a function of `x` and `y`:

```python
import numpy as np
import pywake as pw
import matplotlib.pyplot as plt
from sobol_distributions import SobolDistribution

# Set the seed for reproducibility
np.random.seed(0)

# Define the wind turbine and atmospheric conditions
wake = pw.Wake(
    x=np.array([0, 10]),  # x-coordinates of the wind turbine
    y=np.array([0, 0]),  # y-coordinates of the wind turbine
    z=np.array([80, 80]),  # z-coordinates of the wind turbine
    wind_direction=270,  # wind direction in degrees
    wind_speed=10,  # wind speed in m/s
    wind_shear=0.15,  # wind shear exponent
    atmospheric_stability='neutral',  # atmospheric stability
    inflow_field=pw.InflowField(
        mean_speed=10,  # mean wind speed in m/s
        turbulence_intensity=0.2,  # turbulence intensity
        I_x=0.1,  # integral length scale in x-direction
        I_y=0.1,  # integral length scale in y-direction
        I_z=0.1,  # integral length scale in z-direction
    ),
)

# Define the inflow uncertainties
inflow_uncertainties = {
   'mean_speed': 1,  # uncertainty in mean wind speed
    'turbulence_intensity': 0.1,  # uncertainty in turbulence intensity
    'I_x': 0.05,  # uncertainty in integral length scale in x-direction
    'I_y': 0.05,  # uncertainty in integral length scale in y-direction
    'I_z': 0.05,  # uncertainty in integral length scale in z-direction
}

# Create a Sobol distribution for the inflow uncertainties
sobol_dist = SobolDistribution(inflow_uncertainties)

# Define the time points at which to evaluate the flow field
times = np.linspace(0, 3600, 10)  # 10 time points from 0 to 3600 seconds

# Initialize arrays to store the results
ws_eff_sobol = np.zeros((len(times), len(sobol_dist), len(sobol_dist)))

# Perform the Sobol sensitivity analysis
for i, t in enumerate(times):
    # Evaluate the flow field at the current time point
    wake.evaluate(t)
    
    # Get the wind speed at the turbine location
    ws_eff = wake.wind_speeds[:, 0, 0]
    
    # Sample the inflow uncertainties using the Sobol distribution
    sobol_samples = sobol_dist.sample((len(sobol_dist),))
    
    # Evaluate the wind speed at the turbine location for each set of inflow uncertainties
    ws_eff_sobol[i] = np.array([np.mean(ws_eff + np.random.normal(0, uncert, size=len(sobol_samples)) for uncert in sobol_samples])

# Plot the Sobol sensitivity indices
plt.figure(figsize=(10, 8))
for i in range(len(sobol_dist)):
    plt.plot(times, ws_eff_sobol[:, i], label=f'Sobol index {i}')
plt.xlabel('Time (s)')
plt.ylabel('Wind speed (m/s)')
plt.title('Sobol sensitivity indices')
plt.legend()
plt.show()

# Save the flow field images
for i, t in enumerate(times):
    wake.evaluate(t)
    wake.plot_flow_field('flow_field_{}.png'.format(i))
```
