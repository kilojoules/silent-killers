To generate a script that propagates realistic measurement uncertainties through PyWake and saves a series of flow field images showing the sensitivity of the flow at different times to this inflow, you can use the following Python script. This script uses the `pywake` library to simulate the flow field and the `sobol` library to calculate the Sobol sensitivity indices.

```python
import numpy as np
import pywake
from pywake.examples.data import IEA37Site
from pywake.flow_field import Wake
from sobol import sobol_indices
import matplotlib.pyplot as plt

# Load the site and turbine data
site = IEA37Site()
turbines = site.turbines

# Define the inflow parameters with uncertainty
inflow_params = {
    'wind_speed': {'mean': 8.0,'std': 0.5},  # m/s
    'wind_direction': {'mean': 270.0,'std': 10.0},  # degrees
    'turbulence_intensity': {'mean': 0.1,'std': 0.05}
}

# Define the flow field parameters
flow_field_params = {
    'wake_model': 'Jensen',
    'turbulence_model': 'Constant',
    'wake_deficit_model': 'Linear',
    'ct_model': 'Constant'
}

# Create a PyWake flow field object
wf = Wake(site, turbines, flow_field_params)

# Define the Sobol indices parameters
sobol_params = {
    'n_samples': 1000,
   'seed': 1234
}

# Define the output parameters
output_params = {
    'output_dir': './output',
    'output_prefix':'sobol_sensitivity'
}

# Define the function to calculate the Sobol sensitivity indices
def calculate_sobol_sensitivity(wf, inflow_params, sobol_params):
    # Sample the inflow parameters
    inflow_samples = {}
    for param, dist in inflow_params.items():
        inflow_samples[param] = np.random.normal(dist['mean'], dist['std'], size=sobol_params['n_samples'])

    # Calculate the Sobol sensitivity indices
    sobol_indices_dict = {}
    for i in range(sobol_params['n_samples']):
        # Create a new PyWake flow field object with the current inflow sample
        wf_i = Wake(site, turbines, flow_field_params, inflow_samples, i)
        
        # Calculate the effective wind speed at each turbine
        ws_eff = wf_i.wake_effective_wind_speed()

        # Calculate the Sobol sensitivity indices for each turbine
        sobol_indices_dict[i] = sobol_indices(ws_eff, inflow_samples)

    return sobol_indices_dict

# Calculate the Sobol sensitivity indices
sobol_indices_dict = calculate_sobol_sensitivity(wf, inflow_params, sobol_params)

# Save the Sobol sensitivity indices as images
for i, sobol_indices in sobol_indices_dict.items():
    # Create a new figure
    fig, ax = plt.subplots(figsize=(10, 10))

    # Plot the Sobol sensitivity indices
    ax.imshow(sobol_indices, cmap='coolwarm', extent=[0, 1000, 0, 1000])

    # Set the title and labels
    ax.set_title(f'Sobol Sensitivity Indices at Time {i}')
    ax.set_xlabel('X [m]')
    ax.set_ylabel('Y [m]')

    # Save the figure
    plt.savefig(f"{output_params['output_dir']}/{output_params['output_prefix']}_{i}.png")

    # Close the figure
    plt.close(fig)
```

This script calculates the Sobol sensitivity indices of the effective wind speed at each turbine as a function of the inflow parameters (wind speed, wind direction, and turbulence intensity). The Sobol sensitivity indices are then saved as images, showing the sensitivity of the flow at different times to the inflow.

Note that this script uses a simple example site and turbine layout, and you may need to modify it to suit your specific use case. Additionally, the script assumes that the inflow parameters are normally distributed, and you may need to modify the script to accommodate different distributions.

Also, the script uses a simple wake model and turbulence model, and you may need to modify it to use more advanced models.
