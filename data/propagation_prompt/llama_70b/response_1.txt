import numpy as np
from pywake import WakeModel

class MyWakeModel(WakeModel):
    def __init__(self, turbine_x, turbine_y, ws_eff):
        super().__init__()
        self.turbine_x = turbine_x
        self.turbine_y = turbine_y
        self.ws_eff = ws_eff

    def calculate_wake(self, x, y):
        # Implement your wake model here
        # For simplicity, let's assume a Gaussian wake model
        wake_deficit = np.exp(-((x - self.turbine_x) ** 2 + (y - self.turbine_y) ** 2) / (2 * self.ws_eff ** 2))
        return wake_deficit
```

import numpy as np
from SALib.analyze import sobol
from pywake_model import MyWakeModel

def propagate_uncertainty(model, x, y, ws_eff, num_samples):

    ws_eff_bounds = [0.1, 0.3]  # 10% and 30% uncertainty

    # Create a Sobol sensitivity analysis object
    sa = sobol.Analyzer(model, ['ws_eff'], ['sobol'])

    # Perform the sensitivity analysis
    results = sa.analyze(x, y, ws_eff, num_samples, bounds=[ws_eff_bounds])

    # Extract the Sobol sensitivity indices
    sobol_indices = results['S1']

    return sobol_indices

Create a Python script (e.g., `generate_images.py`) that generates the flow field images:

import numpy as np
import matplotlib.pyplot as plt
from pywake_model import MyWakeModel
from uncertainty_propagation import propagate_uncertainty

def generate_images(x, y, ws_eff, num_samples):
    # Define the time steps
    time_steps = np.arange(0, 10, 0.1)

    # Create a figure and axis object
    fig, ax = plt.subplots()

    # Loop through each time step
    for t in time_steps:
        # Propagate the uncertainty for this time step
        sobol_indices = propagate_uncertainty(MyWakeModel, x, y, ws_eff, num_samples)

        # Create a contour plot of the Sobol sensitivity indices
        ax.contourf(x, y, sobol_indices, cmap='coolwarm')

        # Add a title and labels
        ax.set_title(f'Sobol Sensitivity at t={t:.2f}')
        ax.set_xlabel('x')
        ax.set_ylabel('y')

        # Save the figure
        plt.savefig(f'sobol_sensitivity_{t:.2f}.png')

        # Clear the axis for the next time step
        ax.clear()
```
