def create_turbine_with_uncertainty(base_d=130, base_h=110, base_rated_power=3.35):
    """
    Create a wind turbine with parameters that will have uncertainty.
    
    Parameters:
    -----------
    base_d : float
        Base rotor diameter (m)
    base_h : float
        Base hub height (m)
    base_rated_power : float
        Base rated power (MW)
    
    Returns:
    --------
    WindTurbine object
    """
    # Define power and thrust coefficient curves
    ws = np.arange(0, 30, 0.1)
    # Simple power curve approximation
    power = np.zeros_like(ws)
    ct = np.zeros_like(ws)
    
    # Cut-in at 4 m/s, rated at 11 m/s, cut-out at 25 m/s
    cut_in = 4.0
    rated_ws = 11.0
    cut_out = 25.0
    
    # Power curve
    power[(ws >= cut_in) & (ws < rated_ws)] = (
        base_rated_power * ((ws[(ws >= cut_in) & (ws < rated_ws)] - cut_in) / 
                           (rated_ws - cut_in))**3 * 1e6
    )
    power[(ws >= rated_ws) & (ws <= cut_out)] = base_rated_power * 1e6
    power[ws > cut_out] = 0
    
    # Thrust coefficient curve
    ct[(ws >= cut_in) & (ws < rated_ws)] = 0.8 - 0.4 * (
        (ws[(ws >= cut_in) & (ws < rated_ws)] - cut_in) / (rated_ws - cut_in)
    )
    ct[(ws >= rated_ws) & (ws <= cut_out)] = 0.3
    ct[ws > cut_out] = 0
    
    # Create wind turbine
    return WindTurbine(
        name='UncertainTurbine',
        diameter=base_d,
        hub_height=base_h,
        powerCtFunction=PowerCtTabular(ws, power, 'kW', ct)
    )

def create_wind_farm_layout(n_turbines=9):
    """
    Create a simple wind farm layout.
    
    Parameters:
    -----------
    n_turbines : int
        Number of turbines (will create n_turbines x n_turbines grid if square)
    
    Returns:
    --------
    tuple: (x positions, y positions)
    """
    # Create a grid layout
    spacing = 500  # meters between turbines
    
    if n_turbines <= 3:
        # Small layout
        x = np.array([0, spacing, 2*spacing])
        y = np.array([0, 0, 0])
    else:
        # Square grid for larger farms
        n_per_side = int(np.sqrt(n_turbines))
        x_grid, y_grid = np.meshgrid(
            np.arange(n_per_side) * spacing,
            np.arange(n_per_side) * spacing
        )
        x = x_grid.flatten()[:n_turbines]
        y = y_grid.flatten()[:n_turbines]
    
    return x, y

def run_pywake_simulation(params, turbine, layout_x, layout_y, flow_grid):
    """
    Run a single PyWake simulation with given parameters.
    
    Parameters:
    -----------
    params : dict
        Dictionary of simulation parameters
    turbine : WindTurbine
        Wind turbine object
    layout_x, layout_y : array
        Turbine positions
    flow_grid : XYGrid
        Grid for flow field calculation
    
    Returns:
    --------
    array: WS_eff values across the flow grid
    """
    # Create site with uncertainty in z0
    site = UniformSite(
        ti=params['ti'],  # turbulence intensity
        ws=params['ws'],  # wind speed (will be overridden in simulation)
        p_wd=None,  # Uniform distribution
        shear=params['shear'],
        roughness_length=params['z0']
    )
    
    # Create wake model with Bastankhah-Gaussian model
    wake_model = IEA37SimpleBastankhahGaussian(
        site=site,
        rotorDiameterModel='square',
        turbulenceModel=STF2017TurbulenceModel()
    )
    
    # Run simulation
    simulation_result = wake_model(
        x=layout_x,
        y=layout_y,
        h=turbine.hub_height * np.ones_like(layout_x),  # All turbines same height
        wd=params['wd'],
        ws=params['ws'],
        D=turbine.diameter * np.ones_like(layout_x),  # All turbines same diameter
        type=[0] * len(layout_x)  # All turbines same type
    )
    
    # Calculate flow field
    flow_map = simulation_result.flow_map(grid=flow_grid)
    
    return flow_map.WS_eff.values

def sobol_sensitivity_analysis(parameters, turbine, layout_x, layout_y, 
                              flow_grid, n_samples=1024):
    """
    Perform Sobol sensitivity analysis on WS_eff across the flow field.
    
    Parameters:
    -----------
    parameters : dict
        Dictionary of uncertain parameters
    turbine : WindTurbine
        Wind turbine object
    layout_x, layout_y : array
        Turbine positions
    flow_grid : XYGrid
        Grid for flow field calculation
    n_samples : int
        Number of samples for Sobol analysis
    
    Returns:
    --------
    dict: Sobol sensitivity indices for each grid point
    """
    # Extract parameter names and bounds for SALib
    param_names = list(parameters.keys())
    bounds = [parameters[name]['bounds'] for name in param_names]
    
    # Define problem for SALib
    problem = {
        'num_vars': len(param_names),
        'names': param_names,
        'bounds': bounds
    }
    
    # Generate samples using Saltelli sequence
    print(f"Generating {n_samples} samples for Sobol analysis...")
    param_values = saltelli.sample(problem, n_samples, calc_second_order=True)
    
    # Convert to dictionary format for simulation
    n_simulations = len(param_values)
    print(f"Running {n_simulations} simulations...")
    
    # Initialize array to store results
    grid_shape = (flow_grid.resolution[0], flow_grid.resolution[1])
    ws_eff_results = np.zeros((n_simulations, grid_shape[0], grid_shape[1]))
    
    # Run simulations
    for i in tqdm(range(n_simulations), desc="Running simulations"):
        # Convert sample to parameter dictionary
        params_dict = {}
        for j, name in enumerate(param_names):
            params_dict[name] = param_values[i, j]
        
        # Create turbine with current diameter
        current_turbine = create_turbine_with_uncertainty(
            base_d=params_dict['diameter'],
            base_h=110,  # Fixed hub height
            base_rated_power=3.35
        )
        
        # Run simulation
        try:
            ws_eff_results[i] = run_pywake_simulation(
                params_dict, current_turbine, layout_x, layout_y, flow_grid
            )
        except Exception as e:
            print(f"Error in simulation {i}: {e}")
            # Use previous successful result or mean
            if i > 0:
                ws_eff_results[i] = ws_eff_results[i-1]
            else:
                ws_eff_results[i] = np.ones(grid_shape) * params_dict['ws']
    
    # Perform Sobol analysis for each grid point
    print("Performing Sobol analysis...")
    sobol_indices = {
        'S1': np.zeros((len(param_names), grid_shape[0], grid_shape[1])),
        'ST': np.zeros((len(param_names), grid_shape[0], grid_shape[1])),
        'S2': np.zeros((len(param_names), len(param_names), grid_shape[0], grid_shape[1]))
    }
    
    # Analyze each grid point
    for i in tqdm(range(grid_shape[0]), desc="Analyzing grid points"):
        for j in range(grid_shape[1]):
            # Extract time series for this grid point
            Y = ws_eff_results[:, i, j]
            
            # Perform Sobol analysis
            try:
                Si = sobol.analyze(problem, Y, print_to_console=False)
                sobol_indices['S1'][:, i, j] = Si['S1']
                sobol_indices['ST'][:, i, j] = Si['ST']
                if 'S2' in Si:
                    sobol_indices['S2'][:, :, i, j] = Si['S2']
            except:
                # If analysis fails, set to zeros
                sobol_indices['S1'][:, i, j] = 0
                sobol_indices['ST'][:, i, j] = 0
    
    return sobol_indices, param_names, ws_eff_results
