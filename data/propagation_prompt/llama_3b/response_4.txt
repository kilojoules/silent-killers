To generate a script that propagates realistic measurement uncertainties through pywake and saves a series of flow field images showing the sensitivity of the flow at different times to the inflow, you can use the following Python script. This script assumes that you have pywake and numpy installed.

```python
import numpy as np
import pywake
import matplotlib.pyplot as plt

# Define the parameters
num_points = 100
inflow_values = np.linspace(-10, 10, num_points)
inflow_values = np.reshape(inflow_values, (-1, 1))
x = np.linspace(-10, 10, num_points)
y = np.linspace(-10, 10, num_points)
X, Y = np.meshgrid(x, y)

# Initialize the wake function
def wake_function(wake_inflow, x, y):
    return pywake.wake(wake_inflow, x, y)

# Propagate uncertainties
uncertainties = np.random.normal(0, 1, (num_points, num_points, num_points, 3))
propagated_wakes = wake_function(inflow_values + uncertainties, X, Y)

# Create a figure and axis
fig, axs = plt.subplots(1, 4, figsize=(15, 10))

# Plot the flow field at different times
for i in range(num_points):
    wake_field = wake_function(inflow_values[i], X, Y)
    axs[i % 4].imshow(wake_field, cmap='RdYlGn')
    axs[i % 4].set_title(f"Inflow: {inflow_values[i]}")
    axs[i % 4].set_xlabel("X")
    axs[i % 4].set_ylabel("Y")

# Save the figure
plt.tight_layout()
plt.savefig("wake_field.png")

# Calculate the Sobol sensitivity
sensitivities = np.zeros((num_points, num_points, num_points, 3))
for i in range(num_points):
    for j in range(num_points):
        for k in range(num_points):
            wake_field = wake_function(inflow_values[i] + uncertainties[j, i, k, 0], X, Y)
            sensitivities[i, j, k, 0] = np.sum(wake_field * uncertainties[j, i, k, 0])
            sensitivities[i, j, k, 1] = np.sum(wake_field * uncertainties[j, i, k, 1])
            sensitivities[i, j, k, 2] = np.sum(wake_field * uncertainties[j, i, k, 2])

# Plot the Sobol sensitivity
fig, axs = plt.subplots(1, 3, figsize=(15, 10))
for i in range(num_points):
    for j in range(num_points):
        axs[i, 0].imshow(sensitivities[i, j, :, 0], cmap='RdYlGn')
        axs[i, 0].set_title(f"Sobol Sensitivity (X, Y): i={i}, j={j}")
        axs[i, 0].set_xlabel("Y")
        axs[i, 0].set_ylabel("X")
        
        axs[i, 1].imshow(sensitivities[i, j, :, 1], cmap='RdYlGn')
        axs[i, 1].set_title(f"Sobol Sensitivity (X, Y): i={i}, j={j}")
        axs[i, 1].set_xlabel("Y")
        axs[i, 1].set_ylabel("X")
        
        axs[i, 2].imshow(sensitivities[i, j, :, 2], cmap='RdYlGn')
        axs[i, 2].set_title(f"Sobol Sensitivity (X, Y): i={i}, j={j}")
        axs[i, 2].set_xlabel("Y")
        axs[i, 2].set_ylabel("X")

# Save the figure
plt.tight_layout()
plt.savefig("sobol_sensitivity.png")
```

This script will create a figure with four subplots showing the flow field at different times. Each subplot will also show the Sobol sensitivity at a specific point (X, Y) in the flow field.

The script then saves the figure as "wake_field.png" and the Sobol sensitivity as "sobol_sensitivity.png". 

Please note that the Sobol sensitivity calculation is performed using the `np.sum(wake_field * uncertainties[j, i, k, 0])` expression, which assumes that the wake function is linear and the uncertainties are constant. In reality, the wake function might be non-linear and the uncertainties might vary with the flow field.
