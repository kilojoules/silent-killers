import numpy as np

def join_h5_files(file1, file2, common_type):
    try:
        with h5py.File(file1, 'r') as f1, h5py.File(file2, 'r') as f2:
            return np.concatenate((f1, f2), axis=0)
    except Exception as e:
        print(f"Error joining files: {e}")
        return None

    common_type = 'float32'

    if joined_data is not None:

if __name__ == "__main__":
    main()
