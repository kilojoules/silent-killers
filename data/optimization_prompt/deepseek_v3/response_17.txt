## Diagnosis

The problem occurs during the **merging phase** where the main process tries to read all worker output files. One worker file is either:
1. **Corrupted** - The worker process was interrupted while writing
2. **Incomplete** - The worker didn't finish writing before the main process tried to read it
3. **Empty** - The worker started but didn't write any data

The repeating "130 possible points..." messages suggest workers are still running/printing while the main process is trying to merge files.

### Option 2: Add timeout and retry logic for file reading

```python
import time
import signal

class TimeoutException(Exception):
    pass

def timeout_handler(signum, frame):
    raise TimeoutException()

def read_file_with_timeout(filepath, timeout=30):
    """Read file with timeout to handle hanging reads."""
    signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(timeout)
    
    try:
        with h5py.File(filepath, 'r') as f:
            data = {}
            for key in f.keys():
                data[key] = f[key][:]
        signal.alarm(0)  # Cancel the alarm
        return data
    except TimeoutException:
        print(f"Timeout reading {filepath} after {timeout} seconds")
        return None
    finally:
        signal.alarm(0)  # Ensure alarm is always cancelled

# In your main merging loop:
for i in range(num_workers):
    worker_file = f"worker_{i}.h5"
    max_retries = 3
    retry_delay = 5
    
    for attempt in range(max_retries):
        try:
            data = read_file_with_timeout(worker_file)
            if data is not None:
                # Process data
                break
            else:
                print(f"Retry {attempt + 1}/{max_retries} for {worker_file}")
                time.sleep(retry_delay)
        except Exception as e:
            print(f"Failed to read {worker_file} on attempt {attempt + 1}: {e}")
            if attempt == max_retries - 1:
                print(f"Skipping {worker_file} after {max_retries} failed attempts")
                break
            time.sleep(retry_delay)
```

```python
def recover_from_partial_results(num_workers, expected_seeds_per_worker):
    """Attempt to recover data from partial worker files."""
    recovered_data = {}
    missing_workers = []
    
    for i in range(num_workers):
        worker_file = f"worker_{i}.h5"
        
        # Check file size - very small files are likely incomplete
        if os.path.exists(worker_file):
            file_size = os.path.getsize(worker_file)
            if file_size < 1024:  # Less than 1KB is suspicious
                print(f"Warning: {worker_file} is very small ({file_size} bytes)")
                os.remove(worker_file)  # Remove incomplete file
                missing_workers.append(i)
                continue
        
        try:
            with h5py.File(worker_file, 'r') as f:
                # Check if file has expected amount of data
                num_completed = 0
                for key in f.keys():
                    if 'farm' in key:
                        num_completed += 1
                
                if num_completed >= expected_seeds_per_worker * 0.8:  # At least 80% complete
                    print(f"Worker {i}: Recovered {num_completed} optimizations")
                    recovered_data[i] = f
                else:
                    print(f"Worker {i}: Only {num_completed}/{expected_seeds_per_worker} complete - skipping")
                    missing_workers.append(i)
        except:
            missing_workers.append(i)
    
    return recovered_data, missing_workers
```
