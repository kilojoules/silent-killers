```python
import os
import h5py
import struct

def is_valid_hdf5_file(filepath):
    """Check if file exists and has a valid HDF5 signature"""
    if not os.path.exists(filepath):
        return False
    
    if os.path.getsize(filepath) == 0:
        return False
    
    # Check HDF5 signature (first 8 bytes)
    try:
        with open(filepath, 'rb') as f:
            signature = f.read(8)
            # HDF5 signature is either:
            # \x89HDF\r\n\x1a\n (superblock version 0)
            # \x89HDF\r\n\x00\n (superblock version 2)
            if signature[:4] != b'\x89HDF':
                return False
            if signature[4:8] not in [b'\r\n\x1a\n', b'\r\n\x00\n']:
                return False
        return True
    except:
        return False
```
