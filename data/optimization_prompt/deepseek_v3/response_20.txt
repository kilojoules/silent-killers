```python
    try:

            # Try to access a key attribute or dataset to ensure it's valid
            if len(f.keys()) == 0:
                logging.warning(f"File {filepath} is empty")
                return False
            return True
    except (OSError, IOError, ValueError) as e:
        logging.error(f"Corrupted file detected: {filepath} - {str(e)}")
```

```python
import time

def safe_file_operation(operation, filepath, max_retries=3, delay=1):
    """Perform file operations with retry logic."""
    for attempt in range(max_retries):
        try:
            return operation(filepath)
        except (OSError, IOError) as e:
            if attempt == max_retries - 1:
                raise
            logging.warning(f"File operation failed (attempt {attempt+1}): {e}")
            time.sleep(delay)
```
