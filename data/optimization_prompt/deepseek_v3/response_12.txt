### Option 2: Add Retry Logic for File Opening

```python
def open_hdf5_with_retry(filepath, max_retries=3, delay=1):
    """Attempt to open an HDF5 file with retries."""
    import time
    
    for attempt in range(max_retries):
        try:
            return h5py.File(filepath, 'r')
        except (OSError, IOError) as e:
            if attempt == max_retries - 1:
                raise
            print(f"Attempt {attempt + 1} failed for {filepath}: {e}")
            time.sleep(delay)
    
    raise OSError(f"Failed to open {filepath} after {max_retries} attempts")

# In your main function:
for worker_file in worker_files:

        continue
```
