def safe_h5_open(filename, mode='r', verbose=True):
    """Safely open HDF5 file with validation"""
    try:
        # Check if file exists and has content
        if mode == 'r' and not os.path.exists(filename):
            raise FileNotFoundError(f"File {filename} does not exist")
        
        if mode == 'r':
            file_size = os.path.getsize(filename)
            if file_size == 0:
                raise ValueError(f"File {filename} is empty (0 bytes)")
            
            # Quick check for HDF5 signature
            with open(filename, 'rb') as f:
                header = f.read(8)
                if header[:4] != b'\x89HDF' and header[:4] != b'\x0e\x03\x13\x01':
                    raise ValueError(f"File {filename} has invalid HDF5 signature")
        
        # Open the file
        return h5py.File(filename, mode)
    
    except Exception as e:
        if verbose:
            print(f"ERROR opening {filename}: {e}")
            traceback.print_exc()
        raise

def validate_worker_file(filename):
    """Validate that a worker file is complete and readable"""
    try:
        with safe_h5_open(filename, 'r', verbose=False) as h5f:
            # Check for expected datasets
            required_datasets = ['layouts', 'scores', 'metadata']

                if ds not in h5f:

            layouts = h5f['layouts']
            scores = h5f['scores']
            
            if len(layouts) != len(scores):
                raise ValueError(f"Inconsistent data: {len(layouts)} layouts vs {len(scores)} scores")
            
            return True, len(layouts)
    
    except Exception as e:
        return False, str(e)
