import os
import h5py

def validate_h5_file(filepath):
    """Check if HDF5 file is valid and complete."""
    if not os.path.exists(filepath):
        return False, "File does not exist"
    
    if os.path.getsize(filepath) == 0:
        return False, "File is empty"
    
    try:
        with h5py.File(filepath, 'r') as f:
            # Check if file has expected structure/datasets
            if len(f.keys()) == 0:
                return False, "File has no datasets"
            return True, "Valid"
    except (OSError, IOError) as e:
        return False, f"Corrupted HDF5: {str(e)}"
    except Exception as e:
        return False, f"Unexpected error: {str(e)}"
```

def save_worker_results(results, output_path):
    """Save results atomically to avoid corruption."""
    temp_path = output_path + ".tmp"
    
    try:
        with h5py.File(temp_path, 'w') as f:
            # Save your data
            f.create_dataset('results', data=results)
        
        # Atomic rename (usually atomic on Unix systems)
        os.rename(temp_path, output_path)
        return True
    except Exception as e:
        # Clean up temp file on error
        if os.path.exists(temp_path):
            os.remove(temp_path)
        logging.error(f"Failed to save {output_path}: {e}")
        return False
```
